194  private static byte[] CreateBitmapFileHeader(int offsetToPixels, int fileSize)
195  {

212  private void FillBitmapInfoHeader(int bitsPerPixel, int cmap_entries, byte[] infoHeader)
213  {
214  /* Fill the info header (Microsoft calls this a BITMAPINFOHEADER) */
215  PUT_2B(infoHeader, 0, infoHeader.Length); /* biSize */
216  PUT_4B(infoHeader, 4, m_parameters.Width); /* biWidth */
217  PUT_4B(infoHeader, 8, m_parameters.Height); /* biHeight */
218  PUT_2B(infoHeader, 12, 1); /* biPlanes - must be 1 */
219  PUT_2B(infoHeader, 14, bitsPerPixel); /* biBitCount */
220  /* we leave biCompression = 0, for none */
221  /* we leave biSizeImage = 0; this is correct for uncompressed data */
222 
223  if (m_parameters.DensityUnit == DensityUnit.DotsCm)
224  {
225  /* if have density in dots/cm, then */
226  PUT_4B(infoHeader, 24, m_parameters.DensityX * 100); /* XPels/M */
227  PUT_4B(infoHeader, 28, m_parameters.DensityY * 100); /* XPels/M */
228  }
229 
230  PUT_2B(infoHeader, 32, cmap_entries); /* biClrUsed */
231  /* we leave biClrImportant = 0 */
232  }

252 
253  int i;
254  if (colormap is object)
255  {

262  Output.WriteByte(colormap[1][i]);
263  Output.WriteByte(colormap[0][i]);
264  if (map_entry_size == 4)

266  Output.WriteByte(0);

268  }
269  }
270  else

272  /* Grayscale colormap (only happens with grayscale quantization) */
273  for (i = 0; i < num_colors; i++)
274  {
275  Output.WriteByte(colormap[0][i]);
276  Output.WriteByte(colormap[0][i]);
277  Output.WriteByte(colormap[0][i]);
278  if (map_entry_size == 4)

280  Output.WriteByte(0);
281  }
282  }
283  }
284  }
285  else
286  {
287  /* If no colormap, must be grayscale data. Generate a linear "map". */
288  for (i = 0; i < 256; i++)
289  {
290  Output.WriteByte((byte)i);
291  Output.WriteByte((byte)i);
292  Output.WriteByte((byte)i);
293  if (map_entry_size == 4)

295  Output.WriteByte(0);

297  }
298  }

301  if (i > map_colors)

303  throw new InvalidOperationException("Too many colors");
304  }
305 
306  for (; i < map_colors; i++)
307  {
308  Output.WriteByte(0);
309  Output.WriteByte(0);
310  Output.WriteByte(0);
