32  public static int[] Palette2rgb(ImageLine line, PngChunkPLTE pal, PngChunkTRNS trns, int[] buf)
33  {
34  bool isalpha = trns != null;
35  int channels = isalpha ? 4 : 3;
36  int nsamples = line.ImgInfo.Cols * channels;
37  if (buf == null || buf.Length < nsamples)
38  buf = new int[nsamples];
39  if (!line.SamplesUnpacked)
40  line = line.unpackToNewImageLine();
41  bool isbyte = line.SampleType == Hjg.Pngcs.ImageLine.ESampleType.BYTE;
42  int nindexesWithAlpha = trns != null ? trns.GetPalletteAlpha().Length : 0;
43  for (int c = 0; c < line.ImgInfo.Cols; c++)
44  {
45  int index = isbyte ? (line.ScanlineB[c] & 0xFF) : line.Scanline[c];
46  pal.GetEntryRgb(index, buf, c * channels);
47  if (isalpha)
48  {
49  int alpha = index < nindexesWithAlpha ? trns.GetPalletteAlpha()[index] : 255;
50  buf[c * channels + 3] = alpha;
51  }
52  }
53  return buf;
54  }
