23  {
24  if (line is null)

29  if (pal is null)

33 
34  var isalpha = trns is object;
35  var channels = isalpha ? 4 : 3;
36  var nsamples = line.ImgInfo.Cols * channels;
37  if (buf is null || buf.Length < nsamples)

42  if (!line.SamplesUnpacked)

46 
47  var isbyte = line.SampleType == Hjg.Pngcs.ImageLine.ESampleType.BYTE;
48  var nindexesWithAlpha = trns?.GetPalletteAlpha().Length ?? 0;
49  for (var c = 0; c < line.ImgInfo.Cols; c++)
50  {
51  var index = isbyte ? (line.ScanlineB[c] & 0xFF) : line.Scanline[c];
52  pal.GetEntryRgb(index, buf, c * channels);
53  if (isalpha)

56  }
57  }
