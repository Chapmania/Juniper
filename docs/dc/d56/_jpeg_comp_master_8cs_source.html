61  /// <summary>
62  /// <para>Per-pass setup.</para>
63  /// <para>
64  /// This is called at the beginning of each pass. We determine which
65  /// modules will be active during this pass and give them appropriate
66  /// start_pass calls.
67  /// We also set is_last_pass to indicate whether any more passes will

295  /// <summary>
296  /// Do computations that are needed before processing a JPEG scan
297  /// cinfo.comps_in_scan and cinfo.cur_comp_info[] are already set
298  /// </summary>
299  private void PerScanSetup()
300  {
301  if (m_cinfo.m_comps_in_scan == 1)
302  {
303  /* Noninterleaved (single-component) scan */
304  var compIndex = m_cinfo.m_cur_comp_info[0];
305  var compptr = m_cinfo.Component_info[compIndex];
306 
307  /* Overall image size in MCUs */
308  m_cinfo.m_MCUs_per_row = compptr.Width_in_blocks;
309  m_cinfo.m_MCU_rows_in_scan = compptr.height_in_blocks;
310 
311  /* For noninterleaved scan, always one block per MCU */
312  compptr.MCU_width = 1;
313  compptr.MCU_height = 1;
314  compptr.MCU_blocks = 1;
315  compptr.MCU_sample_width = compptr.DCT_h_scaled_size;
316  compptr.last_col_width = 1;
317 
318  /* For noninterleaved scans, it is convenient to define last_row_height
319  * as the number of block rows present in the last iMCU row.
320  */
321  var tmp = compptr.height_in_blocks % compptr.V_samp_factor;
322  if (tmp == 0)
323  {
324  tmp = compptr.V_samp_factor;
325  }
326 
327  compptr.last_row_height = tmp;
328 
329  /* Prepare array describing MCU composition */
330  m_cinfo.m_blocks_in_MCU = 1;
331  m_cinfo.m_MCU_membership[0] = 0;
332  }
333  else
334  {
335  /* Interleaved (multi-component) scan */
336  if (m_cinfo.m_comps_in_scan <= 0 || m_cinfo.m_comps_in_scan > JpegConstants.MAX_COMPS_IN_SCAN)
337  {
338  m_cinfo.ErrExit(JMessageCode.JERR_COMPONENT_COUNT, m_cinfo.m_comps_in_scan, JpegConstants.MAX_COMPS_IN_SCAN);
339  }
340 
341  /* Overall image size in MCUs */
342  m_cinfo.m_MCUs_per_row = (int)JpegUtils.jdiv_round_up(
343  m_cinfo.jpeg_width, m_cinfo.m_max_h_samp_factor * m_cinfo.BlockSize);
344 
345  m_cinfo.m_MCU_rows_in_scan = (int)JpegUtils.jdiv_round_up(
346  m_cinfo.jpeg_height, m_cinfo.m_max_v_samp_factor * m_cinfo.BlockSize);
347 
348  m_cinfo.m_blocks_in_MCU = 0;
349 
350  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
351  {
352  var compIndex = m_cinfo.m_cur_comp_info[ci];
353  var compptr = m_cinfo.Component_info[compIndex];
354 
355  /* Sampling factors give # of blocks of component in each MCU */
356  compptr.MCU_width = compptr.H_samp_factor;
357  compptr.MCU_height = compptr.V_samp_factor;
358  compptr.MCU_blocks = compptr.MCU_width * compptr.MCU_height;
359  compptr.MCU_sample_width = compptr.MCU_width * compptr.DCT_h_scaled_size;
360 
361  /* Figure number of non-dummy blocks in last MCU column & row */
362  var tmp = compptr.Width_in_blocks % compptr.MCU_width;
363  if (tmp == 0)
364  {
365  tmp = compptr.MCU_width;
366  }
367 
368  compptr.last_col_width = tmp;
369 
370  tmp = compptr.height_in_blocks % compptr.MCU_height;
371  if (tmp == 0)
372  {
373  tmp = compptr.MCU_height;
374  }
375 
376  compptr.last_row_height = tmp;
377 
378  /* Prepare array describing MCU composition */
379  var mcublks = compptr.MCU_blocks;
380  if (m_cinfo.m_blocks_in_MCU + mcublks > JpegConstants.C_MAX_BLOCKS_IN_MCU)
381  {
382  m_cinfo.ErrExit(JMessageCode.JERR_BAD_MCU_SIZE);
383  }
384 
385  while (mcublks-- > 0)
386  {
387  m_cinfo.m_MCU_membership[m_cinfo.m_blocks_in_MCU++] = ci;
388  }
389  }
390  }
391 
392  /* Convert restart specified in rows to actual MCU count. */
393  /* Note that count must fit in 16 bits, so we provide limiting. */
394  if (m_cinfo.m_restart_in_rows > 0)
395  {
396  var nominal = m_cinfo.m_restart_in_rows * m_cinfo.m_MCUs_per_row;
397  m_cinfo.m_restart_interval = Math.Min(nominal, 65535);
398  }
