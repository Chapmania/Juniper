704  /// </summary>

706  /// deal of work including most of the actual output.</remarks>
707  public void JpegFinishCompress()
708  {
709  int iMCU_row;
710 
711  if (globalState == JpegState.CSTATE_SCANNING || globalState == JpegState.CSTATE_RAW_OK)
712  {
713  /* Terminate first pass */
714  if (m_next_scanline < m_image_height)
715  {
716  ErrExit(JMessageCode.JERR_TOO_LITTLE_DATA);
717  }
718

722  {
723  ErrExit(JMessageCode.JERR_BAD_STATE, (int)globalState);
724  }
725 
726  /* Perform any remaining passes */
727  while (!m_master.IsLastPass())
728  {
729  m_master.PrepareForPass();
730  for (iMCU_row = 0; iMCU_row < m_total_iMCU_rows; iMCU_row++)
731  {

982  /// <summary>

985  /// </summary>
986  /// <param name="colorspace">The required colorspace.</param>
987  /// <remarks>See <see href="../articles/KB/special-color-spaces.html">Special color spaces</see>,
988  /// below, before using this. A large number of parameters, including all per-component parameters,
989  /// are set by this routine; if you want to twiddle individual parameters you should call
990  /// <c>jpeg_set_colorspace</c> before rather than after.</remarks>
991  /// <seealso href="../articles/KB/compression-parameter-selection.html">Compression parameter selection</seealso>
992  /// <seealso href="../articles/KB/special-color-spaces.html">Special color spaces</seealso>
993  public void JpegSetColorspace(JColorSpace colorspace)
994  {
995  int ci;
996 
997  /* Safety check to ensure start_compress not called yet. */
998  if (globalState != JpegState.CSTATE_START)
999  {
1000  ErrExit(JMessageCode.JERR_BAD_STATE, (int)globalState);
1001  }
1002 
1003  /* For all colorspaces, we use Q and Huff tables 0 for luminance components,
1004  * tables 1 for chrominance components.
1005  */
1006 
1007  m_jpeg_color_space = colorspace;
1008 
1009  m_write_JFIF_header = false; /* No marker for non-JFIF colorspaces */
1010  m_write_Adobe_marker = false; /* write no Adobe marker by default */
1011 
1012  switch (colorspace)
1013  {
1014  case JColorSpace.JCS_UNKNOWN:
1015  m_num_components = m_input_components;
1016  if (m_num_components < 1 || m_num_components > JpegConstants.MAX_COMPONENTS)
1017  {
1018  ErrExit(JMessageCode.JERR_COMPONENT_COUNT, m_num_components, JpegConstants.MAX_COMPONENTS);
1019  }
1020 
1021  for (ci = 0; ci < m_num_components; ci++)
1022  {
1023  JpegSetColorspaceSetComp(ci, ci, 1, 1, 0, 0, 0);
1024  }
1025 
1026  break;
1027 
1028  case JColorSpace.JCS_GRAYSCALE:
1029  m_write_JFIF_header = true; /* Write a JFIF marker */
1030  m_num_components = 1;
1031  /* JFIF specifies component ID 1 */
1032  JpegSetColorspaceSetComp(0, 0x01, 1, 1, 0, 0, 0);
1033  break;
1034 
1035  case JColorSpace.JCS_RGB:
1036  m_write_Adobe_marker = true; /* write Adobe marker to flag RGB */
1037  m_num_components = 3;
1038  JpegSetColorspaceSetComp(0, 0x52 /* 'R' */, 1, 1, 0,
1039  ColorTransform == JColorTransform.JCT_SUBTRACT_GREEN ? 1 : 0,
1040  ColorTransform == JColorTransform.JCT_SUBTRACT_GREEN ? 1 : 0);
1041  JpegSetColorspaceSetComp(1, 0x47 /* 'G' */, 1, 1, 0, 0, 0);
1042  JpegSetColorspaceSetComp(2, 0x42 /* 'B' */, 1, 1, 0,
1043  ColorTransform == JColorTransform.JCT_SUBTRACT_GREEN ? 1 : 0,
1044  ColorTransform == JColorTransform.JCT_SUBTRACT_GREEN ? 1 : 0);
1045  break;
1046 
1047  case JColorSpace.JCS_YCbCr:
1048  m_write_JFIF_header = true; /* Write a JFIF marker */
1049  m_num_components = 3;
1050  /* JFIF specifies component IDs 1,2,3 */
1051  /* We default to 2x2 subsamples of chrominance */
1052  JpegSetColorspaceSetComp(0, 0x01, 2, 2, 0, 0, 0);
1053  JpegSetColorspaceSetComp(1, 0x02, 1, 1, 1, 1, 1);
1054  JpegSetColorspaceSetComp(2, 0x03, 1, 1, 1, 1, 1);
1055  break;
1056 
1057  case JColorSpace.JCS_CMYK:
1058  m_write_Adobe_marker = true; /* write Adobe marker to flag CMYK */
1059  m_num_components = 4;
1060  JpegSetColorspaceSetComp(0, 0x43 /* 'C' */, 1, 1, 0, 0, 0);
1061  JpegSetColorspaceSetComp(1, 0x4D /* 'M' */, 1, 1, 0, 0, 0);
1062  JpegSetColorspaceSetComp(2, 0x59 /* 'Y' */, 1, 1, 0, 0, 0);
1063  JpegSetColorspaceSetComp(3, 0x4B /* 'K' */, 1, 1, 0, 0, 0);
1064  break;
1065 
1066  case JColorSpace.JCS_YCCK:
1067  m_write_Adobe_marker = true; /* write Adobe marker to flag YCCK */
1068  m_num_components = 4;
1069  JpegSetColorspaceSetComp(0, 0x01, 2, 2, 0, 0, 0);
1070  JpegSetColorspaceSetComp(1, 0x02, 1, 1, 1, 1, 1);

1273  /// <summary>
1274  /// Converts a value on the IJG-recommended quality scale to a linear scaling percentage.
1275  /// </summary>
1276  /// <param name="quality">The IJG-recommended quality scale. Should be 0 (terrible) to 100 (very good).</param>
1277  /// <returns>The linear scaling percentage.</returns>
1278  /// <seealso href="../articles/KB/compression-parameter-selection.html">Compression parameter selection</seealso>
1279  public static int JpegQualityScaling(int quality)
1280  {
1281  /* Safety limit on quality factor. Convert 0 to 1 to avoid zero divide. */
1282  if (quality <= 0)
1283  {
1284  quality = 1;
1285  }
1286 
1287  if (quality > 100)
1288  {
1289  quality = 100;
1290  }
1291

1296  * Qualities 1..50 are converted to scaling percentage 5000/Q.
1297  */

1309  {

1934  /// <summary>
1935  /// Do computations that are needed before master selection phase
1936  /// </summary>
1937  private void InitialSetup(bool transcode_only)
1938  {
1939  if (transcode_only)
1940  {
1941  JpegCalcTransDimensions();
1942  }
1943  else
1944  {
1945  JpegCalcJpegDimensions();
1946  }
1947 
1948  /* Sanity check on block_size */
1949  if (BlockSize < 1 || BlockSize > 16)
1950  {
1951  ErrExit(JMessageCode.JERR_BAD_DCTSIZE, BlockSize, BlockSize);
1952  }
1953 
1954  /* Derive natural_order from block_size */
1955  natural_order = BlockSize switch
1956  {
1957  2 => JpegUtils.jpeg_natural_order2,
1958  3 => JpegUtils.jpeg_natural_order3,
1959  4 => JpegUtils.jpeg_natural_order4,
1960  5 => JpegUtils.jpeg_natural_order5,
1961  6 => JpegUtils.jpeg_natural_order6,
1962  7 => JpegUtils.jpeg_natural_order7,
1963  _ => JpegUtils.jpeg_natural_order,
1964  };
1965 
1966  /* Derive lim_Se from block_size */
1967  lim_Se = BlockSize < JpegConstants.DCTSIZE
1968  ? (BlockSize * BlockSize) - 1
1969  : JpegConstants.DCTSIZE2 - 1;
1970 
1971  /* Sanity check on image dimensions */
1972  if (jpeg_height <= 0 || jpeg_width <= 0 || m_num_components <= 0)
1973  {
1974  ErrExit(JMessageCode.JERR_EMPTY_IMAGE);
1975  }
1976 
1977  /* Make sure image isn't bigger than I can handle */
1978  if (jpeg_height > JpegConstants.JPEG_MAX_DIMENSION
1979  || jpeg_width > JpegConstants.JPEG_MAX_DIMENSION)
1980  {
1981  ErrExit(JMessageCode.JERR_IMAGE_TOO_BIG, (uint)JpegConstants.JPEG_MAX_DIMENSION);
1982  }
1983 
1984  /* Only 8 to 12 bits data precision are supported for DCT based JPEG */
1985  if (m_data_precision < 8 || m_data_precision > 12)
1986  {
1987  ErrExit(JMessageCode.JERR_BAD_PRECISION, m_data_precision);
1988  }
1989 
1990  /* Check that number of components won't exceed internal array sizes */
1991  if (m_num_components > JpegConstants.MAX_COMPONENTS)
1992  {
1993  ErrExit(JMessageCode.JERR_COMPONENT_COUNT, m_num_components, JpegConstants.MAX_COMPONENTS);
1994  }
1995 
1996  /* Compute maximum sampling factors; check factor validity */
1997  m_max_h_samp_factor = 1;
1998  m_max_v_samp_factor = 1;
1999  for (var ci = 0; ci < m_num_components; ci++)
2000  {
2001  if (compInfo[ci].H_samp_factor <= 0
2002  || compInfo[ci].H_samp_factor > JpegConstants.MAX_SAMP_FACTOR
2003  || compInfo[ci].V_samp_factor <= 0
2004  || compInfo[ci].V_samp_factor > JpegConstants.MAX_SAMP_FACTOR)
2005  {
2006  ErrExit(JMessageCode.JERR_BAD_SAMPLING);
2007  }
2008 
2009  m_max_h_samp_factor = Math.Max(m_max_h_samp_factor, compInfo[ci].H_samp_factor);
2010  m_max_v_samp_factor = Math.Max(m_max_v_samp_factor, compInfo[ci].V_samp_factor);
2011  }
2012 
2013  /* Compute dimensions of components */
2014  for (var ci = 0; ci < m_num_components; ci++)
2015  {
2016  var compptr = compInfo[ci];
2017  /* Fill in the correct component_index value; don't rely on application */
2018  compptr.Component_index = ci;
2019 
2020  /* In selecting the actual DCT scaling for each component, we try to
2021  * scale down the chroma components via DCT scaling rather than downsampling.
2022  * This saves time if the downsampler gets to use 1:1 scaling.
2023  * Note this code adapts subsampling ratios which are powers of 2.
2024  */
2025  var ssize = 1;
2026  var upSampleSize = DoFancyDownsampling
2027  ? JpegConstants.DCTSIZE
2028  : JpegConstants.DCTSIZE / 2;
2029 
2030  while (min_DCT_h_scaled_size * ssize <= upSampleSize
2031  && (m_max_h_samp_factor % (compptr.H_samp_factor * ssize * 2)) == 0)
2032  {
2033  ssize *= 2;
2034  }
2035 
2036  compptr.DCT_h_scaled_size = min_DCT_h_scaled_size * ssize;
2037  ssize = 1;
2038  while (min_DCT_v_scaled_size * ssize <= upSampleSize
2039  && (m_max_v_samp_factor % (compptr.V_samp_factor * ssize * 2)) == 0)
2040  {
2041  ssize *= 2;
2042  }
2043 
2044  compptr.DCT_v_scaled_size = min_DCT_v_scaled_size * ssize;
2045 
2046  /* We don't support DCT ratios larger than 2. */
2047  if (compptr.DCT_h_scaled_size > compptr.DCT_v_scaled_size * 2)
2048  {
2049  compptr.DCT_h_scaled_size = compptr.DCT_v_scaled_size * 2;
2050  }
2051  else if (compptr.DCT_v_scaled_size > compptr.DCT_h_scaled_size * 2)
2052  {
2053  compptr.DCT_v_scaled_size = compptr.DCT_h_scaled_size * 2;
2054  }
2055 
2056  /* Size in DCT blocks */
2057  compptr.Width_in_blocks = (int)JpegUtils.jdiv_round_up(
2058  (long)jpeg_width * compptr.H_samp_factor, (long)m_max_h_samp_factor * BlockSize);
2059 
2060  compptr.height_in_blocks = (int)JpegUtils.jdiv_round_up(
2061  (long)jpeg_height * compptr.V_samp_factor, (long)m_max_v_samp_factor * BlockSize);
2062 
2063  /* Size in samples */
2064  compptr.downsampled_width = (int)JpegUtils.jdiv_round_up(
2065  (long)jpeg_width * compptr.H_samp_factor * compptr.DCT_h_scaled_size,
2066  (long)m_max_h_samp_factor * BlockSize);
2067 
2068  compptr.downsampled_height = (int)JpegUtils.jdiv_round_up(
2069  (long)jpeg_height * compptr.V_samp_factor * compptr.DCT_v_scaled_size,
2070  (long)m_max_v_samp_factor * BlockSize);
2071 
2072  /* Don't need quantization scale after DCT,
2073  * until color conversion says otherwise.
2074  */
2075  compptr.component_needed = false;
2076  }
2077 
2078  /* Compute number of fully interleaved MCU rows (number of times that
2079  * main controller will call coefficient controller).
2080  */
2081  m_total_iMCU_rows = (int)JpegUtils.jdiv_round_up(
2082  jpeg_height, (long)m_max_v_samp_factor * BlockSize);
2083  }

2085  /// <summary>
2086  /// Verify that the scan script in scan_info[] is valid;
2087  /// also determine whether it uses progressive JPEG, and set progressive_mode.
2088  /// </summary>
2089  private void ValidateScript()
2090  {
2091  if (m_num_scans <= 0)
2092  {
2093  ErrExit(JMessageCode.JERR_BAD_SCAN_SCRIPT, 0);
2094  }
2095 
2096  /* For sequential JPEG, all scans must have Ss=0, Se=DCTSIZE2-1;
2097  * for progressive JPEG, no scan can have this.
2098  */
2099  var last_bitpos = new int[JpegConstants.MAX_COMPONENTS][];
2100  for (var i = 0; i < JpegConstants.MAX_COMPONENTS; i++)
2101  {
2102  last_bitpos[i] = new int[JpegConstants.DCTSIZE2];
2103  }
2104 
2105  var component_sent = new bool[JpegConstants.MAX_COMPONENTS];
2106 
2107  /* -1 until that coefficient has been seen; then last Al for it */
2108  if (m_scan_info[0].Ss != 0 || m_scan_info[0].Se != JpegConstants.DCTSIZE2 - 1)
2109  {
2110  m_progressive_mode = true;
2111  for (var ci = 0; ci < m_num_components; ci++)
2112  {
2113  for (var coefi = 0; coefi < JpegConstants.DCTSIZE2; coefi++)
2114  {
2115  last_bitpos[ci][coefi] = -1;
2116  }
2117  }
2118  }
2119  else
2120  {
2121  m_progressive_mode = false;
2122  for (var ci = 0; ci < m_num_components; ci++)
2123  {
2124  component_sent[ci] = false;
2125  }
2126  }
2127 
2128  for (var scanno = 1; scanno <= m_num_scans; scanno++)
2129  {
2130  var scanInfo = m_scan_info[scanno - 1];
2131 
2132  /* Validate component indexes */
2133  var ncomps = scanInfo.comps_in_scan;
2134  if (ncomps <= 0 || ncomps > JpegConstants.MAX_COMPS_IN_SCAN)
2135  {
2136  ErrExit(JMessageCode.JERR_COMPONENT_COUNT, ncomps, JpegConstants.MAX_COMPS_IN_SCAN);
2137  }
2138 
2139  for (var ci = 0; ci < ncomps; ci++)
2140  {
2141  var thisi = scanInfo.component_index[ci];
2142  if (thisi < 0 || thisi >= m_num_components)
2143  {
2144  ErrExit(JMessageCode.JERR_BAD_SCAN_SCRIPT, scanno);
2145  }
2146 
2147  /* Components must appear in SOF order within each scan */
2148  if (ci > 0 && thisi <= scanInfo.component_index[ci - 1])
2149  {
2150  ErrExit(JMessageCode.JERR_BAD_SCAN_SCRIPT, scanno);
2151  }
2152  }
2153 
2154  /* Validate progression parameters */
2155  var Ss = scanInfo.Ss;
2156  var Se = scanInfo.Se;
2157  var Ah = scanInfo.Ah;
2158  var Al = scanInfo.Al;
2159  if (m_progressive_mode)
2160  {
2161  /* The JPEG spec simply gives the ranges 0..13 for Ah and Al, but that
2162  * seems wrong: the upper bound ought to depend on data precision.
2163  * Perhaps they really meant 0..N+1 for N-bit precision.
2164  * Here we allow 0..10 for 8-bit data; Al larger than 10 results in
2165  * out-of-range reconstructed DC values during the first DC scan,
2166  * which might cause problems for some decoders.
2167  */
2168  const int MAX_AH_AL = 10;
