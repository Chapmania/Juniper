9  /// <summary>
10  /// Main buffer control (downsampled-data buffer)
11  /// </summary>
12  internal class JpegCMainController
13  {
14  private readonly JpegCompressStruct m_cinfo;
15 
16  private int m_cur_iMCU_row; /* number of current iMCU row */
17  private int m_rowgroup_ctr; /* counts row groups received in iMCU row */
18  private bool m_suspended; /* remember if we suspended output */
19 
20  /* If using just a strip buffer, this points to the entire set of buffers
21  * (we allocate one for each component). In the full-image case, this
22  * points to the currently accessible strips of the virtual arrays.
23  */
24  private readonly byte[][][] m_buffer = new byte[JpegConstants.MAX_COMPONENTS][][];
25 
26  public JpegCMainController(JpegCompressStruct cinfo)
27  {
28  m_cinfo = cinfo;
29 
30  /* Allocate a strip buffer for each component */
31  for (var ci = 0; ci < cinfo.m_num_components; ci++)
32  {
33  var compptr = cinfo.Component_info[ci];
34  m_buffer[ci] = JpegCommonStruct.AllocJpegSamples(
35  compptr.Width_in_blocks * compptr.DCT_h_scaled_size,
36  compptr.V_samp_factor * compptr.DCT_v_scaled_size);
37  }
38  }
39 
40  // Initialize for a processing pass.
41  public void start_pass(JBufMode pass_mode)
42  {
43  /* Do nothing in raw-data mode. */
44  if (m_cinfo.m_raw_data_in)
45  {
46  return;
47  }
48 
49  m_cur_iMCU_row = 0; /* initialize counters */
50  m_rowgroup_ctr = 0;
51  m_suspended = false;
52 
53  if (pass_mode != JBufMode.PassThrough)
54  {
55  m_cinfo.ErrExit(JMessageCode.JERR_BAD_BUFFER_MODE);
56  }
57  }
58 
59  /// <summary>
60  /// Process some data.
61  /// This routine handles the simple pass-through mode,
62  /// where we have only a strip buffer.
63  /// </summary>
64  public void process_data(byte[][] input_buf, ref int in_row_ctr, int in_rows_avail)
65  {
66  while (m_cur_iMCU_row < m_cinfo.m_total_iMCU_rows)
67  {
68  /* Read input data if we haven't filled the main buffer yet */
69  if (m_rowgroup_ctr < m_cinfo.min_DCT_v_scaled_size)
70  {
71  m_cinfo.m_prep.PreProcessData(input_buf, ref in_row_ctr, in_rows_avail, m_buffer,
72  ref m_rowgroup_ctr, m_cinfo.min_DCT_v_scaled_size);
73  }
74 
75  /* If we don't have a full iMCU row buffered, return to application for
76  * more data. Note that preprocessor will always pad to fill the iMCU row
77  * at the bottom of the image.
78  */
79  if (m_rowgroup_ctr != m_cinfo.min_DCT_v_scaled_size)
80  {
81  return;
82  }
83 
84  /* Send the completed row to the compressor */
85  if (!m_cinfo.m_coef.CompressData(m_buffer))
86  {
87  /* If compressor did not consume the whole row, then we must need to
88  * suspend processing and return to the application. In this situation
89  * we pretend we didn't yet consume the last input row; otherwise, if
90  * it happened to be the last row of the image, the application would
91  * think we were done.
92  */
93  if (!m_suspended)
94  {
95  in_row_ctr--;
96  m_suspended = true;
97  }
98 
99  return;
100  }
101 
102  /* We did finish the row. Undo our little suspension hack if a previous
103  * call suspended; then mark the main buffer empty.
104  */
105  if (m_suspended)
106  {
107  in_row_ctr++;
108  m_suspended = false;
109  }
110 
111  m_rowgroup_ctr = 0;
112  m_cur_iMCU_row++;
113  }
114  }
115  }
116 }
