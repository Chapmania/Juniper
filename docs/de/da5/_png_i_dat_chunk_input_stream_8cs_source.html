104  private void EndChunkGoForNext()
105  {
106  // Called after readging the last byte of chunk
107  // Checks CRC, and read ID from next CHUNK
108  // Those values are left in idLastChunk / lenLastChunk
109  // Skips empty IDATS
110  do
111  {
112  var crc = PngHelperInternal.ReadInt4(inputStream);
113  offset += 4;
114  if (checkCrc)
115  {
116  var crccalc = (int)crcEngine.GetValue();
117  if (lenLastChunk > 0 && crc != crccalc)
118  {
119  throw new PngjBadCrcException("error reading idat; offset: " + offset.ToString(CultureInfo.CurrentCulture));
120  }
121 
122  crcEngine.Reset();
123  }
124 
125  lenLastChunk = PngHelperInternal.ReadInt4(inputStream);
126  if (lenLastChunk < 0)
127  {
128  throw new PngjInputException("invalid len for chunk: " + lenLastChunk.ToString(CultureInfo.CurrentCulture));
129  }

132  PngHelperInternal.ReadBytes(inputStream, idLastChunk, 0, 4);
133  offset += 8;
134

136  if (!ended)

138  foundChunksInfo.Add(new IdatChunkInfo(lenLastChunk, offset - 8));
139  if (checkCrc)

142  }
143  }
