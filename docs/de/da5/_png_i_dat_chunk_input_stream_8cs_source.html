77  private void EndChunkGoForNext() {
78  // Called after readging the last byte of chunk
79  // Checks CRC, and read ID from next CHUNK
80  // Those values are left in idLastChunk / lenLastChunk
81  // Skips empty IDATS
82  do {
83  int crc = Hjg.Pngcs.PngHelperInternal.ReadInt4(inputStream); //
84  offset += 4;
85  if (checkCrc) {
86  int crccalc = (int)crcEngine.GetValue();
87  if (lenLastChunk > 0 && crc != crccalc)
88  throw new PngjBadCrcException("error reading idat; offset: " + offset);
89  crcEngine.Reset();
90  }
91  lenLastChunk = Hjg.Pngcs.PngHelperInternal.ReadInt4(inputStream);
92  if (lenLastChunk < 0)
93  throw new PngjInputException("invalid len for chunk: " + lenLastChunk);
94  toReadThisChunk = lenLastChunk;
95  Hjg.Pngcs.PngHelperInternal.ReadBytes(inputStream, idLastChunk, 0, 4);
96  offset += 8;
97

99  if (!ended) {

101  if (checkCrc)
102  crcEngine.Update(idLastChunk, 0, 4);
103  }
104  // PngHelper.logdebug("IDAT ended. next len= " + lenLastChunk + " idat?" +
105  // (!ended));
106  } while (lenLastChunk == 0 && !ended);
