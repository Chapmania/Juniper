194  /// <summary>
195  /// Performs the min-distance-edit operation, given the new, desired angle value.
196  /// </summary>
197  /// <returns>The update.</returns>
198  /// <param name="v1">V1.</param>
199  private Angle Update(float v1)
200  {
201  float r = rotations,
202  v0 = currentValue;
203 
204  v1 = Repeat(v1);
205 
206  float d1, d2, d3;
207  do
208  {
209  // figure out if it is adding the raw value, or whole rotations of the value, that
210  // results in a smaller magnitude of change.
211  d1 = v1 + r - v0;
212  d2 = Math.Abs(d1 + 360);
213  d3 = Math.Abs(d1 - 360);
214  d1 = Math.Abs(d1);
215  if (d2 < d1 && d2 < d3)
216  {
217  r += 360;
218  }
219  else if (d3 < d1)
220  {
221  r -= 360;
222  }
223  } while (d1 > d2 || d1 > d3);
224 
225  return new Angle(v1 + r, r);
226  }
