317  /// <summary>
318  /// Performs the min-distance-edit operation, given the new, desired angle value.
319  /// </summary>
320  /// <returns>The update.</returns>
321  /// <param name="v1">V1.</param>
322  private Angle Update(float v1)
323  {
324  var r = rotations;
325  var v0 = currentValue;
326 
327  v1 = Repeat(v1);
328 
329  float d1;
330  float d2;
331  float d3;
332  do
333  {
334  // figure out if it is adding the raw value, or whole rotations of the value, that
335  // results in a smaller magnitude of change.
336  d1 = v1 + r - v0;
337  d2 = Abs(d1 + 360);
338  d3 = Abs(d1 - 360);
339  d1 = Abs(d1);
340  if (d2 < d1 && d2 < d3)
341  {
342  r += 360;
343  }
344  else if (d3 < d1)
345  {
346  r -= 360;
347  }
348 
349  } while (d1 > d2 || d1 > d3);
350 
351  return new Angle(v1 + r, r);
352  }
