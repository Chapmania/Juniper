178  /// <summary>
179  /// Initiate a new request for a weather report at a given location.
180  /// </summary>
181  /// <param name="location">The current location for which to retrieve a report. Reports need to be at least <see cref="ReportTTLMinutes"/> minutes or <see cref="ReportRadiusMeters"/> meters apart.</param>
182  /// <param name="force">Force downloading a new report, regardless of how far we are from the last report location.</param>
183  /// <param name="prog">A progress tracker, if any.</param>
184  /// <returns></returns>
185  public async Task<IWeatherReport> GetWeatherReportAsync(LatLngPoint location, bool force, IProgress prog = null)
186  {
187  if (location is null)
188  {
189  throw new ArgumentNullException(nameof(location));
190  }
191 
192  prog.Report(0);
193  if (NeedsNewReport(location) || force)
194  {
195  var url = new Uri($"{serverURI}/data/{version.ToString(2)}/{operation}?lat={location.Latitude.ToString(CultureInfo.InvariantCulture)}&lon={location.Longitude.ToString(CultureInfo.InvariantCulture)}&units={units}&appid={apiKey}");
196  try
197  {
198  var requester = HttpWebRequestExt.Create(url);
199  requester.Accept = MediaType.Application.Json;
200  using var response = await requester
201  .GetAsync()
202  .ConfigureAwait(false);
203  if (weatherFactory.TryDeserialize<WeatherReport>(response, out var report))
204  {
205  LastReport = report;
206  }
207  else
208  {
209  throw ErrorReport = new WeatherReportException("GetNewReport", "No response: " + url);
210  }
211  }
212  catch (Exception exp)
213  {
214  throw ErrorReport = new WeatherReportException("GetNewReport", exp.Message + ": " + url);
215  }
216  }
217 
218  prog.Report(1);
219  return LastReport;
220  }
