124  static internal void unpackInplaceInt(ImageInfo iminfo, int[] src, int[] dst,
125  bool Scale) {
126  int bitDepth = iminfo.BitDepth;
127  if (bitDepth >= 8)
128  return; // nothing to do
129  int mask0 = GetMaskForPackedFormatsLs(bitDepth);
130  int scalefactor = 8 - bitDepth;
131  int offset0 = 8 * iminfo.SamplesPerRowPacked - bitDepth * iminfo.SamplesPerRow;
132  int mask, offset, v;
133  if (offset0 != 8) {
134  mask = mask0 << offset0;
135  offset = offset0; // how many bits to shift the mask to the right to recover mask0
136  } else {
137  mask = mask0;
138  offset = 0;
139  }
140  for (int j = iminfo.SamplesPerRow - 1, i = iminfo.SamplesPerRowPacked - 1; j >= 0; j--) {
141  v = (src[i] & mask) >> offset;
142  if (Scale)
143  v <<= scalefactor;
144  dst[j] = v;
145  mask <<= bitDepth;
146  offset += bitDepth;
147  if (offset == 8) {
148  mask = mask0;
149  offset = 0;
150  i--;
151  }
152  }
153  }

155  static internal void packInplaceInt(ImageInfo iminfo, int[] src, int[] dst,
156  bool scaled) {
157  int bitDepth = iminfo.BitDepth;
158  if (bitDepth >= 8)
159  return; // nothing to do
160  int mask0 = GetMaskForPackedFormatsLs(bitDepth);
161  int scalefactor = 8 - bitDepth;
162  int offset0 = 8 - bitDepth;
163  int v, v0;
164  int offset = 8 - bitDepth;
165  v0 = src[0]; // first value is special for in place
166  dst[0] = 0;
167  if (scaled)
168  v0 >>= scalefactor;
169  v0 = ((v0 & mask0) << offset);
170  for (int i = 0, j = 0; j < iminfo.SamplesPerRow; j++) {
171  v = src[j];
172  if (scaled)
173  v >>= scalefactor;
174  dst[i] |= ((v & mask0) << offset);
175  offset -= bitDepth;
176  if (offset < 0) {
177  offset = offset0;
178  i++;
179  dst[i] = 0;
180  }
181  }
182  dst[0] |= v0;
183  }

217  static internal void packInplaceByte(ImageInfo iminfo, byte[] src, byte[] dst,
218  bool scaled) {
219  int bitDepth = iminfo.BitDepth;
220  if (bitDepth >= 8)
221  return; // nothing to do
222  byte mask0 = (byte)GetMaskForPackedFormatsLs(bitDepth);
223  byte scalefactor = (byte)(8 - bitDepth);
224  byte offset0 = (byte)(8 - bitDepth);

226  int offset = 8 - bitDepth;
227  v0 = src[0]; // first value is special
228  dst[0] = 0;

232  for (int i = 0, j = 0; j < iminfo.SamplesPerRow; j++) {
