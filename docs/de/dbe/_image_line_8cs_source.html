137  internal static void UnpackInplaceInt(ImageInfo iminfo, int[] src, int[] dst,
138  bool Scale)
139  {
140  var bitDepth = iminfo.BitDepth;
141  if (bitDepth >= 8)
142  {
143  return; // nothing to do
144  }
145 
146  var mask0 = GetMaskForPackedFormatsLs(bitDepth);
147  var scalefactor = 8 - bitDepth;
148  var offset0 = (8 * iminfo.SamplesPerRowPacked) - (bitDepth * iminfo.SamplesPerRow);
149  int mask, offset, v;
150  if (offset0 != 8)
151  {
152  mask = mask0 << offset0;
153  offset = offset0; // how many bits to shift the mask to the right to recover mask0
154  }
155  else
156  {
157  mask = mask0;
158  offset = 0;
159  }
160 
161  for (int j = iminfo.SamplesPerRow - 1, i = iminfo.SamplesPerRowPacked - 1; j >= 0; j--)
162  {
163  v = (src[i] & mask) >> offset;
164  if (Scale)
165  {
166  v <<= scalefactor;
167  }
168 
169  dst[j] = v;
170  mask <<= bitDepth;
171  offset += bitDepth;
172  if (offset == 8)
173  {
174  mask = mask0;
175  offset = 0;
176  i--;
177  }
178  }
179  }

181  internal static void PackInplaceInt(ImageInfo iminfo, int[] src, int[] dst,
182  bool scaled)
183  {
184  var bitDepth = iminfo.BitDepth;
185  if (bitDepth >= 8)
186  {
187  return; // nothing to do
188  }
189 
190  var mask0 = GetMaskForPackedFormatsLs(bitDepth);
191  var scalefactor = 8 - bitDepth;
192  var offset0 = 8 - bitDepth;
193  int v, v0;
194  var offset = 8 - bitDepth;
195  v0 = src[0]; // first value is special for in place
196  dst[0] = 0;
197  if (scaled)
198  {
199  v0 >>= scalefactor;
200  }
201 
202  v0 = ((v0 & mask0) << offset);
203  for (int i = 0, j = 0; j < iminfo.SamplesPerRow; j++)
204  {
205  v = src[j];
206  if (scaled)
207  {
208  v >>= scalefactor;
209  }
210 
211  dst[i] |= ((v & mask0) << offset);
212  offset -= bitDepth;
213  if (offset < 0)
214  {
215  offset = offset0;
216  i++;
217  dst[i] = 0;
218  }
219  }
