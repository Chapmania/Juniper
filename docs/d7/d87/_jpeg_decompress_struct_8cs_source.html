789  /// <summary>
790  /// Decompression startup: this will read the source datastream header markers, up to the beginning of the compressed data proper.
791  /// </summary>
792  /// <param name="require_image">Read a description of <b>Return Value</b>.</param>
793  /// <returns>
794  /// <para>
795  /// If you pass <c>require_image=true</c> (normal case), you need not check for a
796  /// <see cref="ReadResult.JPEG_HEADER_TABLES_ONLY"/> return code; an abbreviated file will cause
797  /// an error exit. <see cref="ReadResult.JPEG_SUSPENDED"/> is only possible if you use a data source
798  /// module that can give a suspension return.<br/><br/>
799  /// </para>
800  /// <para>
801  /// This method will read as far as the first SOS marker (ie, actual start of compressed data),
802  /// and will save all tables and parameters in the JPEG object. It will also initialize the
803  /// decompression parameters to default values, and finally return <see cref="ReadResult.JPEG_HEADER_OK"/>.
804  /// On return, the application may adjust the decompression parameters and then call
805  /// <see cref="JpegDecompressStruct.JpegStartDecompress"/>. (Or, if the application only wanted to
806  /// determine the image parameters, the data need not be decompressed. In that case, call
807  /// <see cref="JpegCommonStruct.JpegAbort"/> to release any temporary space.)<br/><br/>
808  /// </para>
809  /// <para>

811  /// <see cref="ReadResult.JPEG_HEADER_TABLES_ONLY"/> upon reaching EOI. The application may then re-use
812  /// the JPEG object to read the abbreviated image datastream(s). It is unnecessary (but OK) to call
813  /// <see cref="JpegCommonStruct.JpegAbort">jpeg_abort</see> in this case.
814  /// The <see cref="ReadResult.JPEG_SUSPENDED"/> return code only occurs if the data source module
815  /// requests suspension of the decompressor. In this case the application should load more source
816  /// data and then re-call <c>jpeg_read_header</c> to resume processing.<br/><br/>
817  /// </para>
818  /// <para>
819  /// If a non-suspending data source is used and <c>require_image</c> is <c>true</c>,
820  /// then the return code need not be inspected since only <see cref="ReadResult.JPEG_HEADER_OK"/> is possible.
821  /// </para>
822  /// </returns>
823  /// <remarks>Need only initialize JPEG object and supply a data source before calling.<br/>
824  /// On return, the image dimensions and other info have been stored in the JPEG object.
825  /// The application may wish to consult this information before selecting decompression parameters.<br/>
826  /// This routine is now just a front end to <see cref="JpegConsumeInput"/>, with some extra error checking.
827  /// </remarks>
828  /// <seealso href="../articles/KB/decompression-details.html">Decompression details</seealso>
829  /// <seealso href="../articles/KB/decompression-parameter-selection.html">Decompression parameter selection</seealso>
830  public ReadResult JpegReadHeader(bool require_image)
831  {
832  if (globalState != JpegState.DSTATE_START && globalState != JpegState.DSTATE_INHEADER)
833  {
834  ErrExit(JMessageCode.JERR_BAD_STATE, (int)globalState);
835  }
836 
837  var retcode = JpegConsumeInput();
838 
839  switch (retcode)
840  {
841  case ReadResult.JPEG_REACHED_SOS:
842  return ReadResult.JPEG_HEADER_OK;
843  case ReadResult.JPEG_REACHED_EOI:
844  if (require_image) /* Complain if application wanted an image */
845  {
846  ErrExit(JMessageCode.JERR_NO_IMAGE);
847  }
848  /* Reset to start state; it would be safer to require the application to
849 * call jpeg_abort, but we can't change it now for compatibility reasons.
850 * A side effect is to free any temporary memory (there shouldn't be any).
851 */
852  JpegAbort(); /* sets state = DSTATE_START */
853  return ReadResult.JPEG_HEADER_TABLES_ONLY;
854 
855  case ReadResult.JPEG_SUSPENDED:
856  /* no work */
857  break;
858  }
859

861  }

866  /// <summary>
867  /// Decompression initialization.
868  /// </summary>
869  /// <returns>Returns <c>false</c> if suspended. The return value need be inspected
870  /// only if a suspending data source is used.
871  /// </returns>
872  /// <remarks><para><see cref="JpegDecompressStruct.JpegReadHeader">jpeg_read_header</see> must be completed before calling this.<br/></para>
873  /// <para>If a multipass operating mode was selected, this will do all but the last pass, and thus may take a great deal of time.</para>
874  /// </remarks>
875  /// <seealso cref="JpegDecompressStruct.JpegFinishDecompress"/>
876  /// <seealso href="../articles/KB/decompression-details.html">Decompression details</seealso>
877  public bool JpegStartDecompress()
878  {
879  if (globalState == JpegState.DSTATE_READY)
880  {
881  /* First call: initialize master control, select active modules */
882  m_master = new JpegDecompMaster(this);
883  if (bufferedImage)
884  {
885  /* No more work here; expecting jpeg_start_output next */
886  globalState = JpegState.DSTATE_BUFIMAGE;
887  return true;
888  }
889 
890  globalState = JpegState.DSTATE_PRELOAD;
891  }
892 
893  if (globalState == JpegState.DSTATE_PRELOAD)
894  {
895  /* If file has multiple scans, absorb them all into the coef buffer */
896  if (m_inputctl.HasMultipleScans())
897  {
898  while (true)
899  {
900  ReadResult retcode;
901  /* Call progress monitor hook if present */
902  prog?.Updated();
903 
904  /* Absorb some more input */
905  retcode = m_inputctl.ConsumeInput();
906  if (retcode == ReadResult.JPEG_SUSPENDED)
907  {
908  return false;
909  }
910 
911  if (retcode == ReadResult.JPEG_REACHED_EOI)
912  {
913  break;
914  }
915 
916  /* Advance progress counter if appropriate */
917  if (prog is object && (retcode == ReadResult.JPEG_ROW_COMPLETED || retcode == ReadResult.JPEG_REACHED_SOS))
918  {
919  prog.PassCounter++;
920  if (prog.PassCounter >= prog.PassLimit)
921  {
922  /* underestimated number of scans; ratchet up one scan */
923  prog.PassLimit += m_total_iMCU_rows;
924  }
925  }
926  }
927  }
928 
929  outputScanNumber = inputScanNumber;
930  }
931  else if (globalState != JpegState.DSTATE_PRESCAN)
932  {
933  ErrExit(JMessageCode.JERR_BAD_STATE, (int)globalState);
934  }
935 
936  /* Perform any dummy output passes, and set up for the final pass */
937  return OutputPassSetup();
938  }

1329 
1330  case JColorSpace.JCS_YCbCr:
1331  case JColorSpace.JCS_BG_YCC:

1333  break;

1335  case JColorSpace.JCS_CMYK:
1336  case JColorSpace.JCS_YCCK:

1338  break;

1343  break;

1348  outputComponents = 1;
1349  }
