15  /// <summary>
16  /// <para>Coefficient buffer control</para>
17  /// <para>
18  /// This code applies interblock smoothing as described by section K.8
19  /// of the JPEG standard: the first 5 AC coefficients are estimated from
20  /// the DC values of a DCT block and its 8 neighboring blocks.
21  /// We apply smoothing only for progressive JPEG decoding, and only if
22  /// the coefficients it can estimate are not yet known to full precision.
23  /// </para>
24  /// </summary>
25  internal class JpegDCoefController
26  {

34  private const int Q02_POS = 2;

55  /* In single-pass modes, it's sufficient to buffer just one MCU.
56  * We allocate a workspace of D_MAX_BLOCKS_IN_MCU coefficient blocks,
57  * and let the entropy decoder write into that workspace each time.
58  * (On 80x86, the workspace is FAR even though it's not really very big;
59  * this is to keep the module interfaces unchanged when a large coefficient

74  public JpegDCoefController(JpegDecompressStruct cinfo, bool need_full_buffer)
75  {
76  m_cinfo = cinfo;
77 
78  /* Create the coefficient buffer. */
79  if (need_full_buffer)
80  {
81  /* Allocate a full-image virtual array for each component, */
82  /* padded to a multiple of samp_factor DCT blocks in each direction. */
83  /* Note we ask for a pre-zeroed array. */
84  for (var ci = 0; ci < cinfo.numComponents; ci++)
85  {

87  JpegUtils.jround_up(cinfo.CompInfo[ci].Width_in_blocks, cinfo.CompInfo[ci].H_samp_factor),
88  JpegUtils.jround_up(cinfo.CompInfo[ci].height_in_blocks, cinfo.CompInfo[ci].V_samp_factor));
89  m_whole_image[ci].ErrorProcessor = cinfo;
90  }

94  m_coef_arrays = m_whole_image; /* link to virtual arrays */
95  }

99  for (var i = 0; i < JpegConstants.D_MAX_BLOCKS_IN_MCU; i++)

105  m_decompressor = DecompressorType.OnePass;

119  /// <summary>
120  /// Consume input data and store it in the full-image coefficient buffer.
121  /// We read as much as one fully interleaved MCU row ("iMCU" row) per call,
122  /// ie, v_samp_factor block rows for each component in the scan.
123  /// </summary>
124  public ReadResult ConsumeData()
125  {
126  if (m_useDummyConsumeData)
127  {
128  return ReadResult.JPEG_SUSPENDED; /* Always indicate nothing was done */
129  }
130 
131  var buffer = new JBlock[JpegConstants.MAX_COMPS_IN_SCAN][][];
132 
133  /* Align the virtual buffers for the components used in this scan. */
134  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
135  {
136  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]];
137 
138  buffer[ci] = m_whole_image[componentInfo.Component_index].Access(
139  m_cinfo.inputIMcuRow * componentInfo.V_samp_factor, componentInfo.V_samp_factor);
140 
141  /* Note: entropy decoder expects buffer to be zeroed,
142  * but this is handled automatically by the memory manager
143  * because we requested a pre-zeroed array.
144  */
145  }
146 
147  /* Loop to process one whole iMCU row */
148  for (var yoffset = m_MCU_vert_offset; yoffset < m_MCU_rows_per_iMCU_row; yoffset++)
149  {
150  for (var MCU_col_num = m_MCU_ctr; MCU_col_num < m_cinfo.m_MCUs_per_row; MCU_col_num++)
151  {
152  /* Construct list of pointers to DCT blocks belonging to this MCU */
153  var blkn = 0; /* index of current DCT block within MCU */
154  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
155  {
156  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]];
157  var start_col = MCU_col_num * componentInfo.MCU_width;
158  for (var yindex = 0; yindex < componentInfo.MCU_height; yindex++)
159  {
160  for (var xindex = 0; xindex < componentInfo.MCU_width; xindex++)
161  {
162  m_MCU_buffer[blkn] = buffer[ci][yindex + yoffset][start_col + xindex];
163  blkn++;
164  }
165  }
166  }
167 
168  /* Try to fetch the MCU. */
169  if (!m_cinfo.m_entropy.decodeMcu(m_MCU_buffer))
170  {
171  /* Suspension forced; update state counters and exit */
172  m_MCU_vert_offset = yoffset;
173  m_MCU_ctr = MCU_col_num;
174  return ReadResult.JPEG_SUSPENDED;
175  }
176  }
177 
178  /* Completed an MCU row, but perhaps not an iMCU row */
179  m_MCU_ctr = 0;
180  }
181 
182  /* Completed the iMCU row, advance counters for next one */
183  m_cinfo.inputIMcuRow++;
184  if (m_cinfo.inputIMcuRow < m_cinfo.m_total_iMCU_rows)
185  {
186  StartiMCURow();
187  return ReadResult.JPEG_ROW_COMPLETED;
188  }
189 
190  /* Completed the scan */
191  m_cinfo.m_inputctl.FinishInputPass();
192  return ReadResult.JPEG_SCAN_COMPLETED;
193  }

240  /// <summary>
241  /// <para>
242  /// Decompress and return some data in the single-pass case.
243  /// Always attempts to emit one fully interleaved MCU row ("iMCU" row).
244  /// Input and output must run in lockstep since we have only a one-MCU buffer.
245  /// Return value is JPEG_ROW_COMPLETED, JPEG_SCAN_COMPLETED, or JPEG_SUSPENDED.
246  /// </para>
247  /// <para>
248  /// NB: output_buf contains a plane for each component in image,
249  /// which we index according to the component's SOF position.
250  /// </para>
251  /// </summary>
252  private ReadResult DecompressOnePass(ComponentBuffer[] output_buf)
253  {
254  var last_MCU_col = m_cinfo.m_MCUs_per_row - 1;
255  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
256 
257  /* Loop to process as much as one whole iMCU row */
258  for (var yoffset = m_MCU_vert_offset; yoffset < m_MCU_rows_per_iMCU_row; yoffset++)
259  {
260  for (var MCU_col_num = m_MCU_ctr; MCU_col_num <= last_MCU_col; MCU_col_num++)
261  {
262  /* Try to fetch an MCU. Entropy decoder expects buffer to be zeroed. */
263  if (m_cinfo.lim_Se != 0)
264  {
265  /* can bypass in DC only case */
266  for (var i = 0; i < m_cinfo.m_blocks_in_MCU; i++)
267  {
268  Array.Clear(m_MCU_buffer[i].data, 0, m_MCU_buffer[i].data.Length);
269  }
270  }
271 
272  if (!m_cinfo.m_entropy.decodeMcu(m_MCU_buffer))
273  {
274  /* Suspension forced; update state counters and exit */
275  m_MCU_vert_offset = yoffset;
276  m_MCU_ctr = MCU_col_num;
277  return ReadResult.JPEG_SUSPENDED;
278  }
279 
280  /* Determine where data should go in output_buf and do the IDCT thing.
281  * We skip dummy blocks at the right and bottom edges (but blkn gets
282  * incremented past them!). Note the inner loop relies on having
283  * allocated the MCU_buffer[] blocks sequentially.
284  */
285  var blkn = 0; /* index of current DCT block within MCU */
286  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
287  {
288  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]];
289 
290  /* Don't bother to IDCT an uninteresting component. */
291  if (!componentInfo.component_needed)
292  {
293  blkn += componentInfo.MCU_blocks;
294  continue;
295  }
296 
297  var useful_width = (MCU_col_num < last_MCU_col) ? componentInfo.MCU_width : componentInfo.last_col_width;
298  var outputIndex = yoffset * componentInfo.DCT_v_scaled_size;
299  var start_col = MCU_col_num * componentInfo.MCU_sample_width;
300  for (var yindex = 0; yindex < componentInfo.MCU_height; yindex++)
301  {
302  if (m_cinfo.inputIMcuRow < last_iMCU_row || yoffset + yindex < componentInfo.last_row_height)
303  {
304  var output_col = start_col;
305  for (var xindex = 0; xindex < useful_width; xindex++)
306  {
307  m_cinfo.m_idct.Inverse(componentInfo.Component_index,
308  m_MCU_buffer[blkn + xindex].data, output_buf[componentInfo.Component_index],
309  outputIndex, output_col);
310 
311  output_col += componentInfo.DCT_h_scaled_size;
312  }
313  }
314 
315  blkn += componentInfo.MCU_width;
316  outputIndex += componentInfo.DCT_v_scaled_size;
317  }
318  }
319  }
320 
321  /* Completed an MCU row, but perhaps not an iMCU row */
322  m_MCU_ctr = 0;
323  }
324 
325  /* Completed the iMCU row, advance counters for next one */
326  m_cinfo.outputIMcuRow++;
327  m_cinfo.inputIMcuRow++;
328  if (m_cinfo.inputIMcuRow < m_cinfo.m_total_iMCU_rows)
329  {
330  StartiMCURow();
331  return ReadResult.JPEG_ROW_COMPLETED;
332  }
333 
334  /* Completed the scan */
335  m_cinfo.m_inputctl.FinishInputPass();
336  return ReadResult.JPEG_SCAN_COMPLETED;
337  }

339  /// <summary>
340  /// <para>
341  /// Decompress and return some data in the multi-pass case.
342  /// Always attempts to emit one fully interleaved MCU row ("iMCU" row).
343  /// Return value is JPEG_ROW_COMPLETED, JPEG_SCAN_COMPLETED, or JPEG_SUSPENDED.
344  /// </para>
345  /// <para>NB: output_buf contains a plane for each component in image.</para>
346  /// </summary>
347  private ReadResult DecompressDataOrdinary(ComponentBuffer[] output_buf)
348  {
349  /* Force some input to be done if we are getting ahead of the input. */
350  while (m_cinfo.inputScanNumber < m_cinfo.outputScanNumber
351  || (m_cinfo.inputScanNumber == m_cinfo.outputScanNumber
352  && m_cinfo.inputIMcuRow <= m_cinfo.outputIMcuRow))
353  {
354  if (m_cinfo.m_inputctl.ConsumeInput() == ReadResult.JPEG_SUSPENDED)
355  {
356  return ReadResult.JPEG_SUSPENDED;
357  }
358  }
359 
360  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
361 
362  /* OK, output from the virtual arrays. */
363  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
364  {
365  var componentInfo = m_cinfo.CompInfo[ci];
366 
367  /* Don't bother to IDCT an uninteresting component. */
368  if (!componentInfo.component_needed)
369  {
370  continue;
371  }
372 
373  /* Align the virtual buffer for this component. */
374  var buffer = m_whole_image[ci].Access(m_cinfo.outputIMcuRow * componentInfo.V_samp_factor,
375  componentInfo.V_samp_factor);
376 
377  /* Count non-dummy DCT block rows in this iMCU row. */
378  int block_rows;
379  if (m_cinfo.outputIMcuRow < last_iMCU_row)
380  {
381  block_rows = componentInfo.V_samp_factor;
382  }
383  else
384  {
385  /* NB: can't use last_row_height here; it is input-side-dependent! */
386  block_rows = componentInfo.height_in_blocks % componentInfo.V_samp_factor;
387  if (block_rows == 0)
388  {
389  block_rows = componentInfo.V_samp_factor;
390  }
391  }
392 
393  /* Loop over all DCT blocks to be processed. */
394  var rowIndex = 0;
395  for (var block_row = 0; block_row < block_rows; block_row++)
396  {
397  var output_col = 0;
398  for (var block_num = 0; block_num < componentInfo.Width_in_blocks; block_num++)
399  {
400  m_cinfo.m_idct.Inverse(componentInfo.Component_index,
401  buffer[block_row][block_num].data, output_buf[ci], rowIndex, output_col);
402 
403  output_col += componentInfo.DCT_h_scaled_size;
404  }
405 
406  rowIndex += componentInfo.DCT_v_scaled_size;
407  }
408  }
409 
410  m_cinfo.outputIMcuRow++;
411  if (m_cinfo.outputIMcuRow < m_cinfo.m_total_iMCU_rows)
412  {
413  return ReadResult.JPEG_ROW_COMPLETED;
414  }
415 
416  return ReadResult.JPEG_SCAN_COMPLETED;
417  }

419  /// <summary>
420  /// Variant of decompress_data for use when doing block smoothing.
421  /// </summary>
422  private ReadResult DecompressSmoothData(ComponentBuffer[] output_buf)
423  {
424  /* Force some input to be done if we are getting ahead of the input. */
425  while (m_cinfo.inputScanNumber <= m_cinfo.outputScanNumber && !m_cinfo.m_inputctl.EOIReached())
426  {
427  if (m_cinfo.inputScanNumber == m_cinfo.outputScanNumber)
428  {
429  /* If input is working on current scan, we ordinarily want it to
430  * have completed the current row. But if input scan is DC,
431  * we want it to keep one row ahead so that next block row's DC
432  * values are up to date.
433  */
434  var delta = (m_cinfo.m_Ss == 0) ? 1 : 0;
435  if (m_cinfo.inputIMcuRow > m_cinfo.outputIMcuRow + delta)
436  {
437  break;
438  }
439  }
440 
441  if (m_cinfo.m_inputctl.ConsumeInput() == ReadResult.JPEG_SUSPENDED)
442  {
443  return ReadResult.JPEG_SUSPENDED;
444  }
445  }
446 
447  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
448 
449  /* OK, output from the virtual arrays. */
450  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
451  {
452  var componentInfo = m_cinfo.CompInfo[ci];
453 
454  /* Don't bother to IDCT an uninteresting component. */
455  if (!componentInfo.component_needed)
456  {
457  continue;
458  }
459 
460  int block_rows;
461  int access_rows;
462  bool last_row;
463  /* Count non-dummy DCT block rows in this iMCU row. */
464  if (m_cinfo.outputIMcuRow < last_iMCU_row)
465  {
466  block_rows = componentInfo.V_samp_factor;
467  access_rows = block_rows * 2; /* this and next iMCU row */
468  last_row = false;
469  }
470  else
471  {
472  /* NB: can't use last_row_height here; it is input-side-dependent! */
473  block_rows = componentInfo.height_in_blocks % componentInfo.V_samp_factor;
474  if (block_rows == 0)
475  {
476  block_rows = componentInfo.V_samp_factor;
477  }
478 
479  access_rows = block_rows; /* this iMCU row only */
480  last_row = true;
481  }
482 
483  bool first_row;
484  var bufferRowOffset = 0;
485 
486  /* Align the virtual buffer for this component. */
487  JBlock[][] buffer;
488  if (m_cinfo.outputIMcuRow > 0)
489  {
490  access_rows += componentInfo.V_samp_factor; /* prior iMCU row too */
491  buffer = m_whole_image[ci].Access((m_cinfo.outputIMcuRow - 1) * componentInfo.V_samp_factor, access_rows);
492  bufferRowOffset = componentInfo.V_samp_factor; /* point to current iMCU row */
493  first_row = false;
494  }
495  else
496  {
497  buffer = m_whole_image[ci].Access(0, access_rows);
498  first_row = true;
499  }
500 
501  /* Fetch component-dependent info */
502  var coefBitsOffset = ci * SAVED_COEFS;
503  int Q00 = componentInfo.quant_table.quantBal[0];
504  int Q01 = componentInfo.quant_table.quantBal[Q01_POS];
505  int Q10 = componentInfo.quant_table.quantBal[Q10_POS];
506  int Q20 = componentInfo.quant_table.quantBal[Q20_POS];
507  int Q11 = componentInfo.quant_table.quantBal[Q11_POS];
508  int Q02 = componentInfo.quant_table.quantBal[Q02_POS];
509  var outputIndex = ci;
510 
511  /* Loop over all DCT blocks to be processed. */
512  for (var block_row = 0; block_row < block_rows; block_row++)
513  {
514  var bufferIndex = bufferRowOffset + block_row;
515 
516  int prev_block_row;
517  if (first_row && block_row == 0)
518  {
519  prev_block_row = bufferIndex;
520  }
521  else
522  {
523  prev_block_row = bufferIndex - 1;
524  }
525 
526  int next_block_row;
527  if (last_row && block_row == block_rows - 1)
528  {
529  next_block_row = bufferIndex;
530  }
531  else
532  {
533  next_block_row = bufferIndex + 1;
534  }
535 
536  /* We fetch the surrounding DC values using a sliding-register approach.
537  * Initialize all nine here so as to do the right thing on narrow pics.
538  */
539  int DC1 = buffer[prev_block_row][0][0];
540  var DC2 = DC1;
541  var DC3 = DC1;
542 
543  int DC4 = buffer[bufferIndex][0][0];
544  var DC5 = DC4;
545  var DC6 = DC4;
546 
547  int DC7 = buffer[next_block_row][0][0];
548  var DC8 = DC7;
549  var DC9 = DC7;
550 
551  var output_col = 0;
552  var last_block_column = componentInfo.Width_in_blocks - 1;
553  for (var block_num = 0; block_num <= last_block_column; block_num++)
554  {
555  /* Fetch current DCT block into workspace so we can modify it. */
556  var workspace = new JBlock();
557  Buffer.BlockCopy(buffer[bufferIndex][0].data, 0, workspace.data, 0, workspace.data.Length * sizeof(short));
558 
559  /* Update DC values */
560  if (block_num < last_block_column)
561  {
562  DC3 = buffer[prev_block_row][1][0];
563  DC6 = buffer[bufferIndex][1][0];
564  DC9 = buffer[next_block_row][1][0];
565  }
566 
567  /* Compute coefficient estimates per K.8.
568  * An estimate is applied only if coefficient is still zero,
569  * and is not known to be fully accurate.
570  */
571  /* AC01 */
572  var Al = m_coef_bits_latch[m_coef_bits_savedOffset + coefBitsOffset + 1];
573  if (Al != 0 && workspace[1] == 0)
574  {
575  int pred;
576  var num = 36 * Q00 * (DC4 - DC6);
577  if (num >= 0)
578  {
579  pred = ((Q01 << 7) + num) / (Q01 << 8);
580  if (Al > 0 && pred >= (1 << Al))
581  {
582  pred = (1 << Al) - 1;
583  }
584  }
585  else
586  {
587  pred = ((Q01 << 7) - num) / (Q01 << 8);
588  if (Al > 0 && pred >= (1 << Al))
589  {
590  pred = (1 << Al) - 1;
591  }
592 
593  pred = -pred;
594  }
595 
596  workspace[1] = (short)pred;
597  }
598 
599  /* AC10 */
600  Al = m_coef_bits_latch[m_coef_bits_savedOffset + coefBitsOffset + 2];
601  if (Al != 0 && workspace[8] == 0)
602  {
603  int pred;
604  var num = 36 * Q00 * (DC2 - DC8);
605  if (num >= 0)
606  {
607  pred = ((Q10 << 7) + num) / (Q10 << 8);
608  if (Al > 0 && pred >= (1 << Al))
609  {
610  pred = (1 << Al) - 1;
611  }
612  }
613  else
614  {
615  pred = ((Q10 << 7) - num) / (Q10 << 8);
616  if (Al > 0 && pred >= (1 << Al))
617  {
618  pred = (1 << Al) - 1;
619  }
620 
621  pred = -pred;
622  }
623 
624  workspace[8] = (short)pred;
625  }
626 
627  /* AC20 */
628  Al = m_coef_bits_latch[m_coef_bits_savedOffset + coefBitsOffset + 3];
629  if (Al != 0 && workspace[16] == 0)
630  {
631  int pred;
632  var num = 9 * Q00 * (DC2 + DC8 - (2 * DC5));
633  if (num >= 0)
634  {
635  pred = ((Q20 << 7) + num) / (Q20 << 8);
636  if (Al > 0 && pred >= (1 << Al))
637  {
638  pred = (1 << Al) - 1;
639  }
640  }
641  else
642  {
643  pred = ((Q20 << 7) - num) / (Q20 << 8);
644  if (Al > 0 && pred >= (1 << Al))
645  {
646  pred = (1 << Al) - 1;
647  }
648 
649  pred = -pred;
650  }
651 
652  workspace[16] = (short)pred;
653  }
654 
655  /* AC11 */
656  Al = m_coef_bits_latch[m_coef_bits_savedOffset + coefBitsOffset + 4];
657  if (Al != 0 && workspace[9] == 0)
658  {
659  int pred;
660  var num = 5 * Q00 * (DC1 - DC3 - DC7 + DC9);
661  if (num >= 0)
662  {
663  pred = ((Q11 << 7) + num) / (Q11 << 8);
664  if (Al > 0 && pred >= (1 << Al))
665  {
666  pred = (1 << Al) - 1;
667  }
668  }
669  else
670  {
671  pred = ((Q11 << 7) - num) / (Q11 << 8);
672  if (Al > 0 && pred >= (1 << Al))
673  {
674  pred = (1 << Al) - 1;
675  }
676 
677  pred = -pred;
678  }
679 
680  workspace[9] = (short)pred;
681  }
682 
683  /* AC02 */
684  Al = m_coef_bits_latch[m_coef_bits_savedOffset + coefBitsOffset + 5];
685  if (Al != 0 && workspace[2] == 0)
686  {
687  int pred;
688  var num = 9 * Q00 * (DC4 + DC6 - (2 * DC5));
689  if (num >= 0)
690  {
691  pred = ((Q02 << 7) + num) / (Q02 << 8);
692  if (Al > 0 && pred >= (1 << Al))
693  {
694  pred = (1 << Al) - 1;
695  }
696  }
697  else
698  {
699  pred = ((Q02 << 7) - num) / (Q02 << 8);
700  if (Al > 0 && pred >= (1 << Al))
701  {
702  pred = (1 << Al) - 1;
703  }
704 
705  pred = -pred;
706  }
707 
708  workspace[2] = (short)pred;
709  }
710 
711  /* OK, do the IDCT */
712  m_cinfo.m_idct.Inverse(componentInfo.Component_index, workspace.data, output_buf[outputIndex], 0, output_col);
713 
714  /* Advance for next column */
715  DC1 = DC2;
716  DC2 = DC3;
717  DC4 = DC5;
718  DC5 = DC6;
719  DC7 = DC8;
720  DC8 = DC9;
721 
722  bufferIndex++;
723  prev_block_row++;
724  next_block_row++;
725 
726  output_col += componentInfo.DCT_h_scaled_size;
727  }
728 
729  outputIndex += componentInfo.DCT_v_scaled_size;
730  }
731  }
732 
733  m_cinfo.outputIMcuRow++;
734  if (m_cinfo.outputIMcuRow < m_cinfo.m_total_iMCU_rows)
735  {
736  return ReadResult.JPEG_ROW_COMPLETED;
737  }
738 
739  return ReadResult.JPEG_SCAN_COMPLETED;
740  }

742  /// <summary>
743  /// Determine whether block smoothing is applicable and safe.
744  /// We also latch the current states of the coef_bits[] entries for the
745  /// AC coefficients; otherwise, if the input side of the decompressor
746  /// advances into a new scan, we might think the coefficients are known
747  /// more accurately than they really are.
748  /// </summary>
749  private bool SmoothingOK()
750  {
751  if (!m_cinfo.progressiveMode || m_cinfo.coefBits is null)
752  {
753  return false;
754  }
755 
756  /* Allocate latch area if not already done */
757  if (m_coef_bits_latch is null)
758  {
759  m_coef_bits_latch = new int[m_cinfo.numComponents * SAVED_COEFS];
760  m_coef_bits_savedOffset = 0;
761  }
762 
763  var smoothing_useful = false;
764  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
765  {
766  /* All components' quantization values must already be latched. */
767  var qtable = m_cinfo.CompInfo[ci].quant_table;
768  if (qtable is null)
769  {
770  return false;
771  }
772 
773  /* Verify DC & first 5 AC quantizers are nonzero to avoid zero-divide. */
774  if (qtable.quantBal[0] == 0
775  || qtable.quantBal[Q01_POS] == 0
776  || qtable.quantBal[Q10_POS] == 0
777  || qtable.quantBal[Q20_POS] == 0
778  || qtable.quantBal[Q11_POS] == 0
779  || qtable.quantBal[Q02_POS] == 0)
780  {
781  return false;
782  }
783 
784  /* DC values must be at least partly known for all components. */
785  if (m_cinfo.coefBits[ci][0] < 0)
786  {
787  return false;
788  }
789 
790  /* Block smoothing is helpful if some AC coefficients remain inaccurate. */
791  for (var coefi = 1; coefi <= 5; coefi++)
792  {
793  m_coef_bits_latch[m_coef_bits_savedOffset + coefi] = m_cinfo.coefBits[ci][coefi];
794  if (m_cinfo.coefBits[ci][coefi] != 0)
795  {
796  smoothing_useful = true;
797  }
798  }
799 
800  m_coef_bits_savedOffset += SAVED_COEFS;
801  }
802 
803  return smoothing_useful;
804  }

807  /// Reset within-iMCU-row counters for a new row (input side)
808  /// </summary>
809  private void StartiMCURow()
810  {
811  /* In an interleaved scan, an MCU row is the same as an iMCU row.
812  * In a noninterleaved scan, an iMCU row has v_samp_factor MCU rows.
813  * But at the bottom of the image, process only what's left.
814  */
815  if (m_cinfo.m_comps_in_scan > 1)
816  {
817  m_MCU_rows_per_iMCU_row = 1;
818  }
819  else
820  {

822

824  {
825  m_MCU_rows_per_iMCU_row = componentInfo.V_samp_factor;
826  }
827  else
828  {
829  m_MCU_rows_per_iMCU_row = componentInfo.last_row_height;
830  }
831  }
832 
833  m_MCU_ctr = 0;
834  m_MCU_vert_offset = 0;
835  }
