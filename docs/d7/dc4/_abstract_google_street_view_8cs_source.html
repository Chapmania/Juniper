557  private bool ParseSearchParams(string searchLocation, out string searchPano, out LatLngPoint searchPoint)
558  {
559  searchPano = null;
560  searchPoint = null;
561 
562  if (!string.IsNullOrEmpty(searchLocation))
563  {
564  var panoMatch = GMAPS_URL_PANO_PATTERN.Match(searchLocation);
565  var latLngMatch = GMAPS_URL_LATLNG_PATTERN.Match(searchLocation);
566 
567  if (panoMatch.Success && MetadataResponse.IsPano(panoMatch.Groups[1].Value))
568  {
569  searchPano = panoMatch.Groups[1].Value;
570  }
571  else if (MetadataResponse.IsPano(searchLocation))
572  {
573  searchPano = searchLocation;
574  }
575 
576  if (latLngMatch.Success && LatLngPoint.TryParse(latLngMatch.Groups[1].Value, out var point))
577  {
578  searchPoint = point;
579  }
580  else if (LatLngPoint.TryParse(searchLocation, out var point2))
581  {
582  searchPoint = point2;
583  }
584  }
585 
586  return metadata == null
587  || searchPano != null
588  && metadata.Pano_ID != searchPano
589  || searchPoint != null
590  && searchPoint.Distance(metadata.Location) > 3f;
591  }
