161  protected static Vector2 Round2Square(Vector2 point)
162  {
163  var pLen = point.magnitude;
164  if (pLen > 0)
165  {
166  var onUnitCircle = point / pLen;
167  float mx = Mathf.Abs(onUnitCircle.x),
168  my = Mathf.Abs(onUnitCircle.y);
169 
170  var onUnitSquare = onUnitCircle;
171  if (mx > my)
172  {
173  onUnitSquare /= mx;
174  }
175  else if (my > 0)
176  {
177  onUnitSquare /= my;
178  }
179 
180  point *= onUnitSquare.magnitude;
181  }
182  return point;
183  }
