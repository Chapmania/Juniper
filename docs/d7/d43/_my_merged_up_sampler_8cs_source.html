1 /*
2  * This file contains code for merged upsampling/color conversion.
3  *
4  * This file combines functions from my_upsampler and jpeg_color_deconverter;
5  * read those files first to understand what's going on.
6  *
7  * When the chroma components are to be upsampled by simple replication
8  * (ie, box filtering), we can save some work in color conversion by
9  * calculating all the output pixels corresponding to a pair of chroma
10  * samples at one time. In the conversion equations
11  * R = Y + K1 * Cr
12  * G = Y + K2 * Cb + K3 * Cr
13  * B = Y + K4 * Cb
14  * only the Y term varies among the group of pixels corresponding to a pair
15  * of chroma samples, so the rest of the terms can be calculated just once.
16  * At typical sampling ratios, this eliminates half or three-quarters of the
17  * multiplications needed for color conversion.
18  *
19  * This file currently provides implementations for the following cases:
20  * YCC => RGB color conversion only (YCbCr or BG_YCC).
21  * Sampling ratios of 2h1v or 2h2v.
22  * No scaling needed at upsample time.
23  * Corner-aligned (non-CCIR601) sampling alignment.
24  * Other special cases could be added, but in most applications these are
25  * the only common cases. (For uncommon cases we fall back on the more
26  * general code in my_upsampler and jpeg_color_deconverter)
27  */

132  if (m_spare_full)
133  {
134  /* If we have a spare row saved from a previous cycle, just return it. */
135  var temp = new byte[1][];
136  temp[0] = m_spare_row;

138  num_rows = 1;

140  }
141  else
142  {
143  /* Figure number of rows to return to caller. */
144  num_rows = 2;
145 
146  /* Not more than the distance to the end of the image. */
147  if (num_rows > m_rows_to_go)
148  {
149  num_rows = m_rows_to_go;
150  }
151 
152  /* And not more than what the client can accept: */
153  out_rows_avail -= out_row_ctr;
154  if (num_rows > out_rows_avail)
155  {
156  num_rows = out_rows_avail;
157  }
158

160  var work_ptrs = new byte[2][];
161  work_ptrs[0] = output_buf[out_row_ctr];
162  if (num_rows > 1)
163  {
164  work_ptrs[1] = output_buf[out_row_ctr + 1];
165  }
166  else
167  {
168  work_ptrs[1] = m_spare_row;
169  m_spare_full = true;
170  }
171 
172  /* Now do the upsampling. */
173  H2V2MergedUpSample(input_buf, in_row_group_ctr, work_ptrs);
174  }
175 
176  /* Adjust counts */
177  out_row_ctr += num_rows;
178  m_rows_to_go -= num_rows;
179 
180  /* When the buffer is emptied, declare this input row group consumed */
181  if (!m_spare_full)

183  in_row_group_ctr++;

196  /// <summary>
197  /// Upsample and color convert for the case of 2:1 horizontal and 1:1 vertical.
198  /// </summary>
199  private void H2V1MergedUpSample(ComponentBuffer[] input_buf, int in_row_group_ctr, byte[][] output_buf, int outRow)
200  {
201  var inputIndex0 = 0;
202  var inputIndex1 = 0;
203  var inputIndex2 = 0;
204  var outputIndex = 0;
205 
206  var limit = m_cinfo.m_sample_range_limit;
207  var limitOffset = m_cinfo.m_sampleRangeLimitOffset;
208 
209  /* Loop for each pair of output pixels */
210  for (var col = m_cinfo.outputWidth >> 1; col > 0; col--)
211  {
212  /* Do the chroma part of the calculation */
213  int cb = input_buf[1][in_row_group_ctr][inputIndex1];
214  inputIndex1++;
215 
216  int cr = input_buf[2][in_row_group_ctr][inputIndex2];
217  inputIndex2++;
218 
219  var cred = m_Cr_r_tab[cr];
220  var cgreen = JpegUtils.RIGHT_SHIFT(m_Cb_g_tab[cb] + m_Cr_g_tab[cr], SCALEBITS);
221  var cblue = m_Cb_b_tab[cb];
222 
223  /* Fetch 2 Y values and emit 2 pixels */
224  int y = input_buf[0][in_row_group_ctr][inputIndex0];
225  inputIndex0++;
226 
227  output_buf[outRow][outputIndex + JpegConstants.RGB_RED] = limit[limitOffset + y + cred];
228  output_buf[outRow][outputIndex + JpegConstants.RGB_GREEN] = limit[limitOffset + y + cgreen];
229  output_buf[outRow][outputIndex + JpegConstants.RGB_BLUE] = limit[limitOffset + y + cblue];
230  outputIndex += JpegConstants.RGB_PIXELSIZE;
231 
232  y = input_buf[0][in_row_group_ctr][inputIndex0];
233  inputIndex0++;
234 
235  output_buf[outRow][outputIndex + JpegConstants.RGB_RED] = limit[limitOffset + y + cred];
236  output_buf[outRow][outputIndex + JpegConstants.RGB_GREEN] = limit[limitOffset + y + cgreen];
237  output_buf[outRow][outputIndex + JpegConstants.RGB_BLUE] = limit[limitOffset + y + cblue];
238  outputIndex += JpegConstants.RGB_PIXELSIZE;
239  }
240 
241  /* If image width is odd, do the last output column separately */
242  if ((m_cinfo.outputWidth & 1) != 0)
243  {
244  int cb = input_buf[1][in_row_group_ctr][inputIndex1];
245  int cr = input_buf[2][in_row_group_ctr][inputIndex2];
246  var cred = m_Cr_r_tab[cr];
247  var cgreen = JpegUtils.RIGHT_SHIFT(m_Cb_g_tab[cb] + m_Cr_g_tab[cr], SCALEBITS);
248  var cblue = m_Cb_b_tab[cb];
249 
250  int y = input_buf[0][in_row_group_ctr][inputIndex0];
251  output_buf[outRow][outputIndex + JpegConstants.RGB_RED] = limit[limitOffset + y + cred];
252  output_buf[outRow][outputIndex + JpegConstants.RGB_GREEN] = limit[limitOffset + y + cgreen];
253  output_buf[outRow][outputIndex + JpegConstants.RGB_BLUE] = limit[limitOffset + y + cblue];

346  /// <summary>
347  /// Initialize tables for YCbCr->RGB colorspace conversion.
348  /// This is taken directly from jpeg_color_deconverter; see that file for more info.
349  /// </summary>
350  private void BuildYccRgbTable()
351  {
352  /* Normal case, sYCC */
353 
354  m_Cr_r_tab = new int[JpegConstants.MAXJSAMPLE + 1];
355  m_Cb_b_tab = new int[JpegConstants.MAXJSAMPLE + 1];
356  m_Cr_g_tab = new int[JpegConstants.MAXJSAMPLE + 1];
357  m_Cb_g_tab = new int[JpegConstants.MAXJSAMPLE + 1];
358 
359  for (int i = 0, x = -JpegConstants.CENTERJSAMPLE; i <= JpegConstants.MAXJSAMPLE; i++, x++)
360  {
361  /* i is the actual input pixel value, in the range 0..MAXJSAMPLE */
362  /* The Cb or Cr value we are thinking of is x = i - CENTERJSAMPLE */
363  /* Cr=>R value is nearest int to 1.402 * x */
364  m_Cr_r_tab[i] = JpegUtils.RIGHT_SHIFT((Fix(1.402) * x) + ONE_HALF, SCALEBITS);
365 
366  /* Cb=>B value is nearest int to 1.772 * x */
367  m_Cb_b_tab[i] = JpegUtils.RIGHT_SHIFT((Fix(1.772) * x) + ONE_HALF, SCALEBITS);
368 
369  /* Cr=>G value is scaled-up -0.714136286 * x */
370  m_Cr_g_tab[i] = -Fix(0.714136286) * x;
371 
372  /* Cb=>G value is scaled-up -0.344136286 * x */
373  /* We also add in ONE_HALF so that need not do it in inner loop */
374  m_Cb_g_tab[i] = (-Fix(0.344136286) * x) + ONE_HALF;
375  }
376  }
