415  {
416  if (other is null)
417  {
418  // sort null values to the end of the collection.
419  return -1;
420  }
421  else
422  {
423  // smaller Priority numbers before larger Priority numbers (i.e. int.MaxValue being the "Highest Priority")
424  var priorityCompare = ((double)Priority - other.Priority).Clamp();
425  var patternCompare = string.CompareOrdinal(RegexSource, other.RegexSource).Clamp();
426

433  {
434  protocolCompare = 1;

436

438  comparison = (comparison << 1) + patternCompare;
439  comparison = (comparison << 1) + protocolCompare;
