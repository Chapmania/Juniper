25  /* Fetching the next N bits from the input stream is a time-critical operation
26  * for the Huffman decoders. We implement it with a combination of inline
27  * macros and out-of-line subroutines. Note that N (the number of bits
28  * demanded at one time) never exceeds 15 for JPEG use.
29  *
30  * We read source bytes into get_buffer and dole out bits as needed.
31  * If get_buffer already contains enough bits, they are fetched in-line
32  * by the macros CHECK_BIT_BUFFER and GET_BITS. When there aren't enough
33  * bits, jpeg_fill_bit_buffer is called; it will attempt to fill get_buffer
34  * as full as possible (not just to the number of bits needed; this
35  * prefetching reduces the overhead cost of calling jpeg_fill_bit_buffer).
36  * Note that jpeg_fill_bit_buffer may return false to indicate suspension.
37  * On true return, jpeg_fill_bit_buffer guarantees that get_buffer contains
38  * at least the requested number of bits --- dummy zeroes are inserted if
39  * necessary.
40  */
41  private const int BIT_BUF_SIZE = 32; /* size of buffer in bits */

43  /*
44  * Out-of-line code for bit fetching.
45  * Note: current values of get_buffer and bits_left are passed as parameters,
46  * but are returned in the corresponding fields of the state struct.
47  *
48  * On most machines MIN_GET_BITS should be 25 to allow the full 32-bit width
49  * of get_buffer to be used. (On machines with wider words, an even larger
50  * buffer could be used.) However, on some machines 32-bit shifts are
51  * quite slow and take time proportional to the number of places shifted.

222  {

224  /* Validate progressive scan parameters */
225  if (cinfo.m_Ss == 0)
226  {
227  if (cinfo.m_Se != 0)
228  {
229  bad = true;
230  }
231  }
232  else
233  {
234  /* need not check Ss/Se < 0 since they came from unsigned bytes */
235  if (cinfo.m_Se < cinfo.m_Ss || cinfo.m_Se > cinfo.lim_Se)
236  {
237  bad = true;
238  }
239 
240  /* AC scans may have only one component */
241  if (cinfo.m_comps_in_scan != 1)
242  {
243  bad = true;
244  }
245  }
246 
247  if (cinfo.m_Ah != 0)
248  {
249  /* Successive approximation refinement scan: must have Al = Ah-1. */
250  if (cinfo.m_Ah - 1 != cinfo.m_Al)
251  {
252  bad = true;
253  }
254  }
255 
256  if (cinfo.m_Al > 13)
257  {
258  /* need not check for < 0 */
259  /* Arguably the maximum Al value should be less than 13 for 8-bit precision,
260  * but the spec doesn't say so, and we try to be liberal about what we
261  * accept. Note: large Al values could result in out-of-range DC
262  * coefficients during early scans, leading to bizarre displays due to
263  * overflows in the IDCT math. But we won't crash.
264  */
265  bad = true;
266  }
267 
268  if (bad)
269  {
270  cinfo.ErrExit(JMessageCode.JERR_BAD_PROGRESSION,
271  cinfo.m_Ss, cinfo.m_Se, cinfo.m_Ah, cinfo.m_Al);
272  }
273 
274  /* Update progression status, and verify that scan order is legal.
275  * Note that inter-scan inconsistencies are treated as warnings
276  * not fatal errors ... not clear if this is right way to behave.
277  */
278  for (var ci = 0; ci < cinfo.m_comps_in_scan; ci++)
279  {
280  var cindex = cinfo.CompInfo[cinfo.m_cur_comp_info[ci]].Component_index;
281  if (cinfo.m_Ss != 0 && cinfo.coefBits[cindex][0] < 0)
282  {
283  /* AC without prior DC scan */
284  cinfo.WarnMS(JMessageCode.JWRN_BOGUS_PROGRESSION, cindex, 0);
285  }
286 
287  for (var coefi = cinfo.m_Ss; coefi <= cinfo.m_Se; coefi++)
288  {
289  var expected = cinfo.coefBits[cindex][coefi];
290  if (expected < 0)
291  {
292  expected = 0;
293  }
294 
295  if (cinfo.m_Ah != expected)
296  {
297  cinfo.WarnMS(JMessageCode.JWRN_BOGUS_PROGRESSION, cindex, coefi);
298  }
299

301  }
302  }
303

305  if (cinfo.m_Ah == 0)
306  {
307  if (cinfo.m_Ss == 0)
308  {
309  decodeMcu = DecodeMcuDCFirst;
310  }
311  else
312  {
313  decodeMcu = DecodeMcuACFirst;
314  }
315  }
316  else if (cinfo.m_Ss == 0)
317  {
318  decodeMcu = DecodeMcuDCRefine;
319  }
320  else
321  {
322  decodeMcu = DecodeMcuACRefine;
323  }
324

326  {
327  var compptr = cinfo.CompInfo[cinfo.m_cur_comp_info[ci]];
328 
329  /* Make sure requested tables are present, and compute derived tables.
330  * We may build same derived table more than once, but it's not expensive.
331  */
332  if (cinfo.m_Ss == 0)
333  {
334  if (cinfo.m_Ah == 0)
335  {
336  /* DC refinement needs no table */
337  var tbl = compptr.Dc_tbl_no;
338  JpegMakeDDerivedTable(true, tbl, ref DerivedTables[tbl]);

344  JpegMakeDDerivedTable(false, tbl, ref DerivedTables[tbl]);
345

347  acDerivedTable = DerivedTables[tbl];
348  }

351  saved.last_dc_val[ci] = 0;
352  }
353

355  saved.EOBRUN = 0;
356  }
357  else
358  {
359  /* Check that the scan parameters Ss, Se, Ah/Al are OK for sequential JPEG.
360  * This ought to be an error condition, but we make it a warning because
361  * there are some baseline files out there with all zeroes in these bytes.
362  */

364  || cinfo.m_Ah != 0
365  || cinfo.m_Al != 0

367  || cinfo.m_Se < JpegConstants.DCTSIZE2)

546  /// <summary>
547  /// Decode one MCU's worth of Huffman-compressed coefficients, full-size blocks.
548  /// </summary>
549  private bool DecodeMcuFull(JBlock[] MCU_data)
550  {
551  /* Process restart marker if needed; may have to suspend */
552  if (cinfo.m_restart_interval != 0)
553  {
554  if (restartsToGo == 0)
555  {
556  if (!ProcessRestart())
557  {
558  return false;
559  }
560  }
561  }
562 
563  /* If we've run out of data, just leave the MCU set to zeroes.
564  * This way, we return uniform gray for the remainder of the segment.
565  */
566  if (!insufficientData)
567  {
568  /* Load up working state */
569  var br_state = new BitreadWorkingState();
570  BITREAD_LOAD_STATE(bitState, out var get_buffer, out var bits_left, ref br_state);
571  var state = new SavableState();
572  state.Assign(saved);
573 
574  /* Outer loop handles each block in the MCU */
575 
576  for (var blkn = 0; blkn < cinfo.m_blocks_in_MCU; blkn++)
577  {
578  /* Decode a single block's worth of coefficients */
579 
580  /* Section F.2.2.1: decode the DC coefficient difference */
581  var htbl = dcCurTables[blkn];
582  HUFF_DECODE(out var s, ref br_state, htbl, ref get_buffer, ref bits_left);
583 
584  htbl = acCurTables[blkn];
585  var k = 1;
586  var coef_limit = coefLimit[blkn];
587  var endThisBlock = false;
588 
589  if (coef_limit != 0)
590  {
591  /* Convert DC difference to actual value, update last_dc_val */
592  if (s != 0)
593  {
594  CHECK_BIT_BUFFER(ref br_state, s, ref get_buffer, ref bits_left);
595 
596  var r = GET_BITS(s, get_buffer, ref bits_left);
597  s = HUFF_EXTEND(r, s);
598  }
599 
600  var ci = cinfo.m_MCU_membership[blkn];
601  s += state.last_dc_val[ci];
602  state.last_dc_val[ci] = s;
603 
604  /* Output the DC coefficient */
605  MCU_data[blkn][0] = (short)s;
606 
607  /* Section F.2.2.2: decode the AC coefficients */
608  /* Since zeroes are skipped, output area must be cleared beforehand */
609  for (; k < coef_limit; k++)
610  {
611  HUFF_DECODE(out s, ref br_state, htbl, ref get_buffer, ref bits_left);
612 
613  var r = s >> 4;
614  s &= 15;
615 
616  if (s != 0)
617  {
618  k += r;
619  CHECK_BIT_BUFFER(ref br_state, s, ref get_buffer, ref bits_left);
620  r = GET_BITS(s, get_buffer, ref bits_left);
621  s = HUFF_EXTEND(r, s);
622 
623  /* Output coefficient in natural (dezigzagged) order.
624  * Note: the extra entries in jpeg_natural_order[] will save us
625  * if k >= DCTSIZE2, which could happen if the data is corrupted.
626  */
627  MCU_data[blkn][JpegUtils.jpeg_natural_order[k]] = (short)s;
628  }
629  else
630  {
631  if (r != 15)
632  {
633  endThisBlock = true;
634  break;
635  }
636 
637  k += 15;
638  }
639  }
640  }
641  else if (s != 0)
642  {
643  CHECK_BIT_BUFFER(ref br_state, s, ref get_buffer, ref bits_left);
644  DROP_BITS(s, ref bits_left);
645  }
646 
647  if (endThisBlock)
648  {
649  continue;
650  }
651 
652  /* Section F.2.2.2: decode the AC coefficients */
653  /* In this path we just discard the values */
654  for (; k < JpegConstants.DCTSIZE2; k++)
655  {
656  HUFF_DECODE(out s, ref br_state, htbl, ref get_buffer, ref bits_left);
657 
658  var r = s >> 4;
659  s &= 15;
660 
661  if (s != 0)
662  {
663  k += r;
664  CHECK_BIT_BUFFER(ref br_state, s, ref get_buffer, ref bits_left);
665  DROP_BITS(s, ref bits_left);
666  }
667  else
668  {
669  if (r != 15)
670  {
671  break;
672  }
673 
674  k += 15;
675  }
676  }
677  }
678 
679  /* Completed MCU, so update state */
680  BITREAD_SAVE_STATE(ref bitState, get_buffer, bits_left);
681  saved.Assign(state);
682  }
683 
684  /* Account for restart interval (no-op if not using restarts) */
685  restartsToGo--;
686 
687  return true;
688  }

1070  if (cinfo.m_restart_interval != 0)
1071  {
1072  if (restartsToGo == 0)
1073  {
1074  if (!ProcessRestart())
1075  {
1076  return false;
1077  }
1078  }
1079  }
1080 
1081  /* If we've run out of data, don't modify the MCU.
1082  */
1083  if (!insufficientData)
1084  {
1085  var p1 = 1 << cinfo.m_Al; /* 1 in the bit position being coded */
1086  var m1 = -1 << cinfo.m_Al; /* -1 in the bit position being coded */
1087  var natural_order = cinfo.natural_order;
1088 
1089  /* Load up working state */
1090  var br_state = new BitreadWorkingState();
1091  BITREAD_LOAD_STATE(bitState, out var get_buffer, out var bits_left, ref br_state);
1092  var EOBRUN = saved.EOBRUN; /* only part of saved state we need */
1093 
1094  /* If we are forced to suspend, we must undo the assignments to any newly
1095  * nonzero coefficients in the block, because otherwise we'd get confused
1096  * next time about which coefficients were already nonzero.
1097  * But we need not undo addition of bits to already-nonzero coefficients;
1098  * instead, we can test the current bit to see if we already did it.
1099  */
1100  var num_newnz = 0;
1101  var newnz_pos = new int[JpegConstants.DCTSIZE2];
1102 
1103  /* initialize coefficient loop counter to start of band */
1104  var k = cinfo.m_Ss;
1105 
1106  if (EOBRUN == 0)
1107  {
1108  do
1109  {
1110  HUFF_DECODE(out var s, ref br_state, acDerivedTable, ref get_buffer, ref bits_left);
1111 
1112  var r = s >> 4;
1113  s &= 15;
1114  if (s != 0)
1115  {
1116  if (s != 1)
1117  {

1119  cinfo.WarnMS(JMessageCode.JWRN_HUFF_BAD_CODE);

1121

1123 
1124  if (GET_BITS(1, get_buffer, ref bits_left) != 0)
1125  {

1127  s = p1;
1128  }
1129  else
1130  {
1131  /* newly nonzero coef is negative */
1132  s = m1;
1133  }
1134  }
1135  else
1136  {
1137  if (r != 15)
1138  {
1139  EOBRUN = (uint)(1 << r); /* EOBr, run length is 2^r + appended bits */
1140  if (r != 0)
1141  {
1142  CHECK_BIT_BUFFER(ref br_state, r, ref get_buffer, ref bits_left);
1143 
1144  r = GET_BITS(r, get_buffer, ref bits_left);
1145  EOBRUN += (uint)r;
1146  }
1147 
1148  break; /* rest of block is handled by EOB logic */
1149  }
1150  /* note s = 0 for processing ZRL */
1151  }
1152  /* Advance over already-nonzero coefs and r still-zero coefs,
1153  * appending correction bits to the nonzeroes. A correction bit is 1
1154  * if the absolute value of the coefficient must be increased.
1155  */
1156  do
1157  {

1170  {
1171  MCU_data[0][blockIndex] += (short)p1;
1172  }
1173  else
1174  {
1175  MCU_data[0][blockIndex] += (short)m1;
1176  }
1177  }
1178  }
1179  }
1180  else if (--r < 0)
1181  {
1182  break; /* reached target zero coefficient */
1183  }
1184 
1185  k++;
1186  }
1187  while (k <= cinfo.m_Se);
1188 
1189  if (s != 0)
1190  {
1191  var pos = natural_order[k];
1192 
1193  /* Output newly nonzero coefficient */
1194  MCU_data[0][pos] = (short)s;
1195 
1196  /* Remember its position in case we have to suspend */
1197  newnz_pos[num_newnz++] = pos;
1198  }
1199 
1200  k++;
1201 
1202  } while (k <= cinfo.m_Se);
1203  }
1204 
1205  if (EOBRUN != 0)
1206  {
1207  /* Scan any remaining coefficient positions after the end-of-band
1208  * (the last newly nonzero coefficient, if any). Append a correction
1209  * bit to each already-nonzero coefficient. A correction bit is 1
1210  * if the absolute value of the coefficient must be increased.
1211  */
1212  do
1213  {
1214  var blockIndex = natural_order[k];
1215  var thiscoef = MCU_data[0][blockIndex];
1216  if (thiscoef != 0)
1217  {
1218  CHECK_BIT_BUFFER(ref br_state, 1, ref get_buffer, ref bits_left);
1219

1221  {
1222  if ((thiscoef & p1) == 0)
1223  {
1224  /* do nothing if already changed it */
1225  if (thiscoef >= 0)
1226  {
1227  MCU_data[0][blockIndex] += (short)p1;
1228  }
1229  else
1230  {
1231  MCU_data[0][blockIndex] += (short)m1;
1232  }
1233  }
1234  }
1235  }
1236 
1237  k++;
1238 
1239  } while (k <= cinfo.m_Se);
1240

1242  EOBRUN--;
1243  }
1244 
1245  /* Completed MCU, so update state */

1248  }

1366  /*
1367  * Code for extracting next Huffman-coded symbol from input bit stream.
1368  * Again, this is time-critical and we make the main paths be macros.
1369  *
1370  * We use a lookahead table to process codes of up to HUFF_LOOKAHEAD bits
1371  * without looping. Usually, more than 95% of the Huffman codes will be 8
1372  * or fewer bits long. The few overlength codes are handled with a loop,
1373  * which need not be inline code.
1374  *
1375  * Notes about the HUFF_DECODE macro:
1376  * 1. Near the end of the data segment, we may fail to get enough bits
1377  * for a lookahead. In that case, we do it the hard way.
1378  * 2. If the lookahead table contains no entry, the next code must be
1379  * more than HUFF_LOOKAHEAD bits long.
1380  * 3. jpeg_huff_decode returns -1 if forced to suspend.

1419  /* Load up the bit buffer to a depth of at least nbits */
1420  private static void JpegFillBitBuffer(ref BitreadWorkingState state, int get_buffer, int bits_left, int nbits)
1421  {
1422  /* Attempt to load at least MIN_GET_BITS bits into get_buffer. */
1423  /* (It is assumed that no request will be for more than that many bits.) */
1424  /* We fail to do so only if we hit a marker or are forced to suspend. */
1425 
1426  var noMoreBytes = false;
1427 
1428  if (state.cinfo.m_unreadMarker == 0)
1429  {
1430  /* cannot advance past a marker */
1431  while (bits_left < MIN_GET_BITS)
1432  {
1433  _ = state.cinfo.m_src.GetByte(out var c);
1434 
1435  /* If it's 0xFF, check and discard stuffed zero byte */
1436  if (c == 0xFF)
1437  {
1438  /* Loop here to discard any padding FF's on terminating marker,
1439  * so that we can save a valid unread_marker value. NOTE: we will
1440  * accept multiple FF's followed by a 0 as meaning a single FF data
1441  * byte. This data pattern is not valid according to the standard.
1442  */
1443  do
1444  {
1445  _ = state.cinfo.m_src.GetByte(out c);
1446  }
1447  while (c == 0xFF);
1448 
1449  if (c == 0)
1450  {
1451  /* Found FF/00, which represents an FF data byte */
1452  c = 0xFF;
1453  }
1454  else
1455  {
1456  /* Oops, it's actually a marker indicating end of compressed data.
1457  * Save the marker code for later use.
1458  * Fine point: it might appear that we should save the marker into
1459  * bitread working state, not straight into permanent state. But
1460  * once we have hit a marker, we cannot need to suspend within the
1461  * current MCU, because we will read no more bytes from the data
1462  * source. So it is OK to update permanent state right away.
1463  */
1464  state.cinfo.m_unreadMarker = c;
1465  /* See if we need to insert some fake zero bits. */
1466  noMoreBytes = true;
1467  break;
1468  }
1469  }
1470 
1471  /* OK, load c into get_buffer */
1472  get_buffer = (get_buffer << 8) | c;
1473  bits_left += 8;
1474  } /* end while */
1475  }
1476  else
1477  {
1478  noMoreBytes = true;
1479  }
1480 
1481  if (noMoreBytes)
1482  {
1483  /* We get here if we've read the marker that terminates the compressed
1484  * data segment. There should be enough bits in the buffer register
1485  * to satisfy the request; if so, no problem.
1486  */
1487  if (nbits > bits_left)
1488  {
1489  /* Uh-oh. Report corrupted data to user and stuff zeroes into
1490  * the data stream, so that we can produce some kind of image.
1491  * We use a nonvolatile flag to ensure that only one warning message
1492  * appears per data segment.
1493  */
1494  var entropy = (HuffmanEntropyDecoder)state.cinfo.m_entropy;
1495  if (!entropy.insufficientData)
1496  {
1497  state.cinfo.WarnMS(JMessageCode.JWRN_HIT_MARKER);
1498  entropy.insufficientData = true;
1499  }
1500 
1501  /* Fill the buffer with zero bits */
1502  get_buffer <<= MIN_GET_BITS - bits_left;
1503  bits_left = MIN_GET_BITS;
1504  }
1505  }
1506 
1507  /* Unload the local registers */
1508  state.getBuffer = get_buffer;
1509  state.bitsLeft = bits_left;
1510  }

1512  /// <summary>
1513  /// Expand a Huffman table definition into the derived format
1514  /// This routine also performs some validation checks on the table.
1515  /// </summary>
1516  private void JpegMakeDDerivedTable(bool isDC, int tblno, ref DDerivedTable dtbl)
1517  {
1518  /* Note that huffsize[] and huffcode[] are filled in code-length order,
1519  * paralleling the order of the symbols themselves in htbl.huffval[].
1520  */
1521 
1522  /* Find the input Huffman table */
1523  if (tblno < 0 || tblno >= JpegConstants.NUM_HUFF_TBLS)
1524  {
1525  cinfo.ErrExit(JMessageCode.JERR_NO_HUFF_TABLE, tblno);
1526  }
1527 
1528  var htbl = isDC ? cinfo.m_dc_huff_tbl_ptrs[tblno] : cinfo.m_ac_huff_tbl_ptrs[tblno];
1529  if (htbl is null)
1530  {
1531  cinfo.ErrExit(JMessageCode.JERR_NO_HUFF_TABLE, tblno);
1532  }
1533 
1534  /* Allocate a workspace if we haven't already done so. */
1535  if (dtbl is null)
1536  {
1537  dtbl = new DDerivedTable();
1538  }
1539 
1540  dtbl.pub = htbl; /* fill in back link */
1541 
1542  /* Figure C.1: make table of Huffman code length for each symbol */
1543 
1544  var p = 0;
1545  var huffsize = new char[257];
1546  for (var l = 1; l <= 16; l++)
1547  {
1548  int i = htbl.Bits[l];
1549  if (p + i > 256) /* protect against table overrun */
1550  {
1551  cinfo.ErrExit(JMessageCode.JERR_BAD_HUFF_TABLE);
1552  }
1553 
1554  while (i-- != 0)
1555  {
1556  huffsize[p++] = (char)l;
1557  }
1558  }
1559 
1560  huffsize[p] = (char)0;
1561  var numsymbols = p;
1562 
1563  /* Figure C.2: generate the codes themselves */
1564  /* We also validate that the counts represent a legal Huffman code tree. */
1565 
1566  var code = 0;
1567  int si = huffsize[0];
1568  var huffcode = new int[257];
1569  p = 0;
1570  while (huffsize[p] != 0)
1571  {
1572  while (huffsize[p] == si)
1573  {
1574  huffcode[p++] = code;
1575  code++;
1576  }
1577 
1578  /* code is now 1 more than the last code used for codelength si; but
1579  * it must still fit in si bits, since no code is allowed to be all ones.
1580  */
1581  if (code >= (1 << si))
1582  {
1583  cinfo.ErrExit(JMessageCode.JERR_BAD_HUFF_TABLE);
1584  }
1585 
1586  code <<= 1;
1587  si++;
1588  }
1589 
1590  /* Figure F.15: generate decoding tables for bit-sequential decoding */
1591 
1592  p = 0;
1593  for (var l = 1; l <= 16; l++)
1594  {
1595  if (htbl.Bits[l] != 0)
1596  {
1597  /* valoffset[l] = huffval[] index of 1st symbol of code length l,
1598  * minus the minimum code of length l
1599  */
1600  dtbl.valOffset[l] = p - huffcode[p];
1601  p += htbl.Bits[l];
1602  dtbl.maxCode[l] = huffcode[p - 1]; /* maximum code of length l */
1603  }
1604  else
1605  {
1606  /* -1 if no codes of this length */
1607  dtbl.maxCode[l] = -1;
1608  }
1609  }
1610 
1611  dtbl.maxCode[17] = 0xFFFFF; /* ensures jpeg_huff_decode terminates */
1612 
1613  /* Compute lookahead tables to speed up decoding.
1614  * First we set all the table entries to 0, indicating "too long";
1615  * then we iterate through the Huffman codes that are short enough and
1616  * fill in all the entries that correspond to bit sequences starting
1617  * with that code.
1618  */
1619 
1620  Array.Clear(dtbl.lookNBits, 0, dtbl.lookNBits.Length);
1621  p = 0;
1622  for (var l = 1; l <= JpegConstants.HUFF_LOOKAHEAD; l++)
1623  {
1624  for (var i = 1; i <= htbl.Bits[l]; i++, p++)
1625  {
1626  /* l = current code's length, p = its index in huffcode[] & huffval[]. */
1627  /* Generate left-justified code followed by all possible bit sequences */
1628  var lookbits = huffcode[p] << (JpegConstants.HUFF_LOOKAHEAD - l);
1629  for (var ctr = 1 << (JpegConstants.HUFF_LOOKAHEAD - l); ctr > 0; ctr--)
1630  {
1631  dtbl.lookNBits[lookbits] = l;
1632  dtbl.lookSym[lookbits] = htbl.Huffval[p];
1633  lookbits++;
1634  }
1635  }
1636  }
1637 
1638  /* Validate symbols as being reasonable.
1639  * For AC tables, we make no check, but accept all byte values 0..255.
1640  * For DC tables, we require the symbols to be in range 0..15.
1641  * (Tighter bounds could be applied depending on the data depth and mode,
1642  * but this is sufficient to ensure safe decoding.)
1643  */
1644  if (isDC)
1645  {
1646  for (var i = 0; i < numsymbols; i++)
1647  {
1648  int sym = htbl.Huffval[i];
1649  if (sym > 15)
1650  {
1651  cinfo.ErrExit(JMessageCode.JERR_BAD_HUFF_TABLE);
1652  }
1653  }
1654  }
1655  }

1657  /* Out-of-line case for Huffman code fetching */
1658  protected static int JpegHuffmanDecode(ref BitreadWorkingState state, int get_buffer, int bits_left, DDerivedTable htbl, int min_bits)
1659  {
1660  /* HUFF_DECODE has determined that the code is at least min_bits */
1661  /* bits long, so fetch that many bits in one swoop. */
1662  var l = min_bits;
1663  CHECK_BIT_BUFFER(ref state, l, ref get_buffer, ref bits_left);
1664 
1665  var code = GET_BITS(l, get_buffer, ref bits_left);
1666 
1667  /* Collect the rest of the Huffman code one bit at a time. */
1668  /* This is per Figure F.16 in the JPEG spec. */
1669 
1670  while (code > htbl.maxCode[l])
1671  {
1672  code <<= 1;
1673  CHECK_BIT_BUFFER(ref state, 1, ref get_buffer, ref bits_left);
1674 
1675  code |= GET_BITS(1, get_buffer, ref bits_left);
1676  l++;
1677  }
1678 
1679  /* Unload the local registers */
1680  state.getBuffer = get_buffer;
1681  state.bitsLeft = bits_left;
1682 
1683  /* With garbage input we may reach the sentinel value l = 17. */
1684 
1685  if (l > 16)
1686  {
1687  state.cinfo.WarnMS(JMessageCode.JWRN_HUFF_BAD_CODE);
1688  /* fake a zero as the safest result */
1689  return 0;
1690  }
1691 
1692  return htbl.pub.Huffval[code + htbl.valOffset[l]];
1693  }
1694  }
