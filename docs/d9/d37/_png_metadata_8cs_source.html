28  /// <summary>Queues the chunk at the writer</summary>
29  /// <param name="chunk">Chunk, ready for write</param>
30  /// <param name="lazyOverwrite">Ovewrite lazily equivalent chunks</param>
31  /// <remarks>Warning: the overwriting applies to equivalent chunks, see <c>ChunkPredicateEquiv</c>
32  /// and will only make sense for queued (not yet writen) chunks
33  /// </remarks>
34  public void QueueChunk(PngChunk chunk, bool lazyOverwrite) {
35  ChunksListForWrite cl = getChunkListW();
36  if (ReadOnly)
37  throw new PngjException("cannot set chunk : readonly metadata");
38  if (lazyOverwrite) {
39  ChunkHelper.TrimList(cl.GetQueuedChunks(), new ChunkPredicateEquiv(chunk));
40  }
41  cl.Queue(chunk);
42  }

146  /// <summary>
147  /// Creates a text chunk and enqueues it
148  /// </summary>
149  /// <param name="key">Key. Short and ASCII string</param>
150  /// <param name="val">Text.</param>
151  /// <param name="useLatin1">Flag. If false, will use UTF-8 (iTXt)</param>
152  /// <param name="compress">Flag. Uses zTXt chunk.</param>
153  /// <returns>The created and enqueued chunk</returns>
154  public PngChunkTextVar SetText(String key, String val, bool useLatin1, bool compress) {
155  if (compress && !useLatin1)
156  throw new PngjException("cannot compress non latin text");
157  PngChunkTextVar c;
158  if (useLatin1) {
159  if (compress) {
160  c = new PngChunkZTXT(chunkList.imageInfo);
161  } else {
162  c = new PngChunkTEXT(chunkList.imageInfo);
163  }
164  } else {
165  c = new PngChunkITXT(chunkList.imageInfo);
166  ((PngChunkITXT)c).SetLangtag(key); // we use the same orig tag (this is not quite right)
167  }
168  c.SetKeyVal(key, val);
169  QueueChunk(c, true);
170  return c;
171  }

183  /// <summary>

189  public List<PngChunkTextVar> GetTxtsForKey(String key) {
190  List<PngChunkTextVar> li = new List<PngChunkTextVar>();
191  foreach (PngChunk c in chunkList.GetById(ChunkHelper.tEXt, key))
192  li.Add((PngChunkTextVar)c);
193  foreach (PngChunk c in chunkList.GetById(ChunkHelper.zTXt, key))
194  li.Add((PngChunkTextVar)c);
195  foreach (PngChunk c in chunkList.GetById(ChunkHelper.iTXt, key))
196  li.Add((PngChunkTextVar)c);
197  return li;
198  }

200  /// <summary>
201  /// Joins all strings for a given key
202  /// </summary>
203  /// <param name="key">Key</param>
204  /// <returns>Concatenated (with newlines) if several found, empty string if none</returns>
205  /// <remarks>You'd perhaps prefer GetTxtsForKey</remarks>
206  public String GetTxtForKey(String key) {
207  String t = "";
208  List<PngChunkTextVar> li = GetTxtsForKey(key);
209  if (li.Count == 0)
210  return t;
211  foreach (PngChunkTextVar c in li)
212  t = t + c.GetVal() + "\n";
