1 /*
2  * This file contains upsampling routines.
3  *
4  * Upsampling input data is counted in "row groups". A row group
5  * is defined to be (v_samp_factor * DCT_v_scaled_size / min_DCT_v_scaled_size)
6  * sample rows of each component. Upsampling will normally produce
7  * max_v_samp_factor pixel rows from each row group (but this could vary
8  * if the upsampler is applying a scale factor of its own).
9  *
10  * An excellent reference for image resampling is
11  * Digital Image Warping, George Wolberg, 1990.
12  * Pub. by IEEE Computer Society Press, Los Alamitos, CA. ISBN 0-8186-8944-7.
13  */

59  public MyUpSampler(JpegDecompressStruct cinfo)
60  {
61  m_cinfo = cinfo;
62  m_need_context_rows = false; /* until we find out differently */
63 
64  if (cinfo.m_CCIR601_sampling) /* this isn't supported */
65  {
66  cinfo.ErrExit(JMessageCode.JERR_CCIR601_NOTIMPL);
67  }
68 
69  /* Verify we can handle the sampling factors, select per-component methods,
70  * and create storage as needed.
71  */
72  for (var ci = 0; ci < cinfo.numComponents; ci++)
73  {
74  var componentInfo = cinfo.CompInfo[ci];
75 
76  /* Compute size of an "input group" after IDCT scaling. This many samples
77  * are to be converted to max_h_samp_factor * max_v_samp_factor pixels.
78  */
79  var h_in_group = (componentInfo.H_samp_factor * componentInfo.DCT_h_scaled_size) / cinfo.min_DCT_h_scaled_size;
80  var v_in_group = (componentInfo.V_samp_factor * componentInfo.DCT_v_scaled_size) / cinfo.min_DCT_v_scaled_size;
81  var h_out_group = cinfo.m_max_h_samp_factor;
82  var v_out_group = cinfo.m_maxVSampleFactor;
83 
84  /* save for use later */
85  m_rowgroup_height[ci] = v_in_group;
86 
87  if (!componentInfo.component_needed)
88  {
89  /* Don't bother to upsample an uninteresting component. */
90  m_upsampleMethods[ci] = ComponentUpsampler.noop_upsampler;
91  continue; /* don't need to allocate buffer */
92  }
93 
94  if (h_in_group == h_out_group && v_in_group == v_out_group)
95  {
96  /* Fullsize components can be processed without any work. */
97  m_upsampleMethods[ci] = ComponentUpsampler.fullsize_upsampler;
98  continue; /* don't need to allocate buffer */
99  }
100 
101  if (h_in_group * 2 == h_out_group && v_in_group == v_out_group)
102  {
103  /* Special case for 2h1v upsampling */
104  m_upsampleMethods[ci] = ComponentUpsampler.h2v1_upsampler;
105  }
106  else if (h_in_group * 2 == h_out_group && v_in_group * 2 == v_out_group)
107  {
108  /* Special case for 2h2v upsampling */
109  m_upsampleMethods[ci] = ComponentUpsampler.h2v2_upsampler;
110  }
111  else if ((h_out_group % h_in_group) == 0 && (v_out_group % v_in_group) == 0)
112  {
113  /* Generic integral-factors upsampling method */
114  m_upsampleMethods[ci] = ComponentUpsampler.int_upsampler;
115  m_h_expand[ci] = (byte)(h_out_group / h_in_group);
116  m_v_expand[ci] = (byte)(v_out_group / v_in_group);
117  }
118  else
119  {
120  cinfo.ErrExit(JMessageCode.JERR_FRACT_SAMPLE_NOTIMPL);
121  }
122 
123  var cb = new ComponentBuffer();
124  cb.SetBuffer(JpegCommonStruct.AllocJpegSamples(JpegUtils.jround_up(cinfo.outputWidth,
125  cinfo.m_max_h_samp_factor), cinfo.m_maxVSampleFactor), null, 0);
126 
127  m_color_buf[ci] = cb;
128  }
129  }

143  /// <summary>
144  /// <para>Control routine to do upsampling (and color conversion).</para>
145  /// <para>
146  /// In this version we upsample each component independently.
147  /// We upsample one row group into the conversion buffer, then apply
148  /// color conversion a row at a time.
149  /// </para>
150  /// </summary>
151  public override void UpSample(ComponentBuffer[] input_buf, ref int in_row_group_ctr, int in_row_groups_avail, byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
152  {
153  /* Fill the conversion buffer, if it's empty */
154  if (m_next_row_out >= m_cinfo.m_maxVSampleFactor)
155  {
156  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
157  {
158  m_perComponentOffsets[ci] = 0;
159 
160  /* Invoke per-component upsample method.*/
161  m_currentComponent = ci;
162  m_upsampleRowOffset = in_row_group_ctr * m_rowgroup_height[ci];
163  UpSampleComponent(ref input_buf[ci]);
164  }
165 
166  m_next_row_out = 0;
167  }
168 
169  /* Color-convert and emit rows */
170 
171  /* How many we have in the buffer: */
172  var num_rows = m_cinfo.m_maxVSampleFactor - m_next_row_out;
173 
174  /* Not more than the distance to the end of the image. Need this test
175  * in case the image height is not a multiple of max_v_samp_factor:
176  */
177  if (num_rows > m_rows_to_go)
178  {
179  num_rows = m_rows_to_go;
180  }
181 
182  /* And not more than what the client can accept: */
183  out_rows_avail -= out_row_ctr;
184  if (num_rows > out_rows_avail)
185  {
186  num_rows = out_rows_avail;
187  }
188 
189  m_cinfo.m_cconvert.ColorConvert(m_color_buf, m_perComponentOffsets, m_next_row_out, output_buf, out_row_ctr, num_rows);
190 
191  /* Adjust counts */
192  out_row_ctr += num_rows;
193  m_rows_to_go -= num_rows;
194  m_next_row_out += num_rows;
195 
196  /* When the buffer is emptied, declare this input row group consumed */
197  if (m_next_row_out >= m_cinfo.m_maxVSampleFactor)
198  {
199  in_row_group_ctr++;
200  }
201  }

203  private void UpSampleComponent(ref ComponentBuffer input_data)
204  {
205  switch (m_upsampleMethods[m_currentComponent])
206  {
207  case ComponentUpsampler.noop_upsampler:
208  NoOpUpSample();
209  break;
210  case ComponentUpsampler.fullsize_upsampler:
211  FullSizeUpSample(ref input_data);
212  break;
213  case ComponentUpsampler.h2v1_upsampler:
214  H2V1UpSample(ref input_data);
215  break;
216  case ComponentUpsampler.h2v2_upsampler:
217  H2V2UpSample(ref input_data);
218  break;
219  case ComponentUpsampler.int_upsampler:
220  IntUpSample(ref input_data);
221  break;
222  default:
223  m_cinfo.ErrExit(JMessageCode.JERR_NOTIMPL);
224  break;
225  }
226  }

254  /// <summary>
255  /// Fast processing for the common case of 2:1 horizontal and 1:1 vertical.
256  /// It's still a box filter.
257  /// </summary>
258  private void H2V1UpSample(ref ComponentBuffer input_data)
259  {
260  var output_data = m_color_buf[m_currentComponent];
261 
262  for (var inrow = 0; inrow < m_cinfo.m_maxVSampleFactor; inrow++)
263  {
264  var row = m_upsampleRowOffset + inrow;
265  var outIndex = 0;
266 
267  for (var col = 0; outIndex < m_cinfo.outputWidth; col++)
268  {
269  var invalue = input_data[row][col]; /* don't need GETJSAMPLE() here */
270  output_data[inrow][outIndex] = invalue;
271  outIndex++;
272  output_data[inrow][outIndex] = invalue;
273  outIndex++;
274  }
275  }
276  }

308  /// <summary>
309  /// This version handles any integral sampling ratios.
310  /// This is not used for typical JPEG files, so it need not be fast.
311  /// Nor, for that matter, is it particularly accurate: the algorithm is
312  /// simple replication of the input pixel onto the corresponding output
313  /// pixels. The hi-falutin sampling literature refers to this as a
314  /// "box filter". A box filter tends to introduce visible artifacts,
315  /// so if you are actually going to use 3:1 or 4:1 sampling ratios
316  /// you would be well advised to improve this code.
317  /// </summary>
318  private void IntUpSample(ref ComponentBuffer input_data)
319  {
320  var output_data = m_color_buf[m_currentComponent];
321  int h_expand = m_h_expand[m_currentComponent];
322  int v_expand = m_v_expand[m_currentComponent];
323 
324  var inrow = 0;
325  var outrow = 0;
326  while (outrow < m_cinfo.m_maxVSampleFactor)
327  {
328  /* Generate one output row with proper horizontal expansion */
329  var row = m_upsampleRowOffset + inrow;
330  for (var col = 0; col < m_cinfo.outputWidth; col++)
331  {
332  var invalue = input_data[row][col]; /* don't need GETJSAMPLE() here */
333  var outIndex = 0;
334  for (var h = h_expand; h > 0; h--)
335  {
336  output_data[outrow][outIndex] = invalue;
337  outIndex++;
338  }
339  }
340 
341  /* Generate any additional output rows by duplicating the first one */
342  if (v_expand > 1)
343  {
344  JpegUtils.jcopy_sample_rows(output_data, outrow, output_data,
345  outrow + 1, v_expand - 1, m_cinfo.outputWidth);
346  }
347 
348  inrow++;
349  outrow += v_expand;
350  }
351  }
352  }
