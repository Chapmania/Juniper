281  /// <summary>
282  /// <para>
283  /// Read JPEG markers before, between, or after compressed-data scans.
284  /// Change state as necessary when a new scan is reached.
285  /// Return value is JPEG_SUSPENDED, JPEG_REACHED_SOS, or JPEG_REACHED_EOI.
286  /// </para>
287  /// <para>
288  /// The consume_input method pointer points either here or to the
289  /// coefficient controller's consume_data routine, depending on whether
290  /// we are reading a compressed data segment or inter-segment markers.
291  /// </para>
292  /// <para>
293  /// Note: This function should NOT return a pseudo SOS marker(with zero
294  /// component number) to the caller.A pseudo marker received by
295  /// read_markers is processed and then skipped for other markers.
296  /// </para>
297  /// </summary>
298  private ReadResult ConsumeMarkers()
299  {
300  ReadResult val;
301 
302  if (m_eoi_reached) /* After hitting EOI, read no further */
303  {
304  return ReadResult.JPEG_REACHED_EOI;
305  }
306 
307  /* Loop to pass pseudo SOS marker */
308  for (; ; )
309  {
310  val = m_cinfo.m_marker.ReadMarkers();
311 
312  switch (val)
313  {
314  case ReadResult.JPEG_REACHED_SOS:
315  /* Found SOS */
316  if (m_inheaders != 0)
317  {
318  /* 1st SOS */
319  if (m_inheaders == 1)
320  {
321  InitialSetup();
322  }
323 
324  if (m_cinfo.m_comps_in_scan == 0)
325  {
326  /* pseudo SOS marker */
327  m_inheaders = 2;
328  break;
329  }
330 
331  m_inheaders = 0;
332 
333  /* Note: start_input_pass must be called by jpeg_decomp_master
334  * before any more input can be consumed.
335  */
336  }
337  else
338  {
339  /* 2nd or later SOS marker */
340  if (!m_has_multiple_scans)
341  {
342  /* Oops, I wasn't expecting this! */
343  m_cinfo.ErrExit(JMessageCode.JERR_EOI_EXPECTED);
344  }
345 
346  if (m_cinfo.m_comps_in_scan == 0)
347  {
348  /* unexpected pseudo SOS marker */
349  break;
350  }
351 
352  m_cinfo.m_inputctl.StartInputPass();
353  }
354  return val;
355 
356  case ReadResult.JPEG_REACHED_EOI:
357  /* Found EOI */
358  m_eoi_reached = true;
359  if (m_inheaders != 0)
360  {
361  /* Tables-only datastream, apparently */
362  if (m_cinfo.m_marker.SawSOF())
363  {
364  m_cinfo.ErrExit(JMessageCode.JERR_SOF_NO_SOS);
365  }
366  }
367  else
368  {
369  /* Prevent infinite loop in coef ctlr's decompress_data routine
370  * if user set output_scan_number larger than number of scans.
371  */
372  if (m_cinfo.outputScanNumber > m_cinfo.inputScanNumber)
373  {
374  m_cinfo.outputScanNumber = m_cinfo.inputScanNumber;
375  }
376  }
377 
378  return val;
379 
380  case ReadResult.JPEG_SUSPENDED:
381  default:
382  return val;
383  }
384  }
385  }

581  /// <summary>
582  /// <para>
583  /// Save away a copy of the Q-table referenced by each component present
584  /// in the current scan, unless already saved during a prior scan.
585  /// </para>
586  /// <para>
587  /// In a multiple-scan JPEG file, the encoder could assign different components
588  /// the same Q-table slot number, but change table definitions between scans
589  /// so that each component uses a different Q-table. (The IJG encoder is not
590  /// currently capable of doing this, but other encoders might.) Since we want
591  /// to be able to dequantize all the components at the end of the file, this
592  /// means that we have to save away the table actually used for each component.
593  /// We do this by copying the table at the start of the first scan containing
594  /// the component.
595  /// The JPEG spec prohibits the encoder from changing the contents of a Q-table
596  /// slot between scans of a component using that slot. If the encoder does so
597  /// anyway, this decoder will simply use the Q-table values that were current
598  /// at the start of the first scan for the component.
599  /// </para>
600  /// <para>
601  /// The decompressor output side looks only at the saved quant tables,
602  /// not at the current Q-table slots.
603  /// </para>
604  /// </summary>
605  private void LatchQuantTables()
606  {
607  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
608  {
609  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]];
610 
611  /* No work if we already saved Q-table for this component */
612  if (componentInfo.quant_table is object)
613  {
614  continue;
615  }
616 
617  /* Make sure specified quantization table is present */
618  var qtblno = componentInfo.Quant_tbl_no;
619  if (qtblno < 0 || qtblno >= JpegConstants.NUM_QUANT_TBLS || m_cinfo.m_quant_tbl_ptrs[qtblno] is null)
620  {
621  m_cinfo.ErrExit(JMessageCode.JERR_NO_QUANT_TABLE, qtblno);
622  }
623 
624  /* OK, save away the quantization table */
625  var qtbl = new JQuantTable();
626  Buffer.BlockCopy(m_cinfo.m_quant_tbl_ptrs[qtblno].quantBal, 0,
627  qtbl.quantBal, 0, qtbl.quantBal.Length * sizeof(short));
628  qtbl.SentTable = m_cinfo.m_quant_tbl_ptrs[qtblno].SentTable;
629  componentInfo.quant_table = qtbl;
630  m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]] = componentInfo;
631  }
632  }

634  /// <summary>
635  /// Do computations that are needed before processing a JPEG scan
636  /// cinfo.comps_in_scan and cinfo.cur_comp_info[] were set from SOS marker
637  /// </summary>
638  private void PerScanSetup()
639  {
640  if (m_cinfo.m_comps_in_scan == 1)
641  {
642  /* Noninterleaved (single-component) scan */
643  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[0]];
644 
645  /* Overall image size in MCUs */
646  m_cinfo.m_MCUs_per_row = componentInfo.Width_in_blocks;
647  m_cinfo.m_MCU_rows_in_scan = componentInfo.height_in_blocks;
648 
649  /* For noninterleaved scan, always one block per MCU */
650  componentInfo.MCU_width = 1;
651  componentInfo.MCU_height = 1;
652  componentInfo.MCU_blocks = 1;
653  componentInfo.MCU_sample_width = componentInfo.DCT_h_scaled_size;
654  componentInfo.last_col_width = 1;
655 
656  /* For noninterleaved scans, it is convenient to define last_row_height
657  * as the number of block rows present in the last iMCU row.
658  */
659  var tmp = componentInfo.height_in_blocks % componentInfo.V_samp_factor;
660  if (tmp == 0)
661  {
662  tmp = componentInfo.V_samp_factor;
663  }
664 
665  componentInfo.last_row_height = tmp;
666  m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[0]] = componentInfo;
667 
668  /* Prepare array describing MCU composition */
669  m_cinfo.m_blocks_in_MCU = 1;
670  m_cinfo.m_MCU_membership[0] = 0;
671  }
672  else
673  {
674  /* Interleaved (multi-component) scan */
675  if (m_cinfo.m_comps_in_scan <= 0 || m_cinfo.m_comps_in_scan > JpegConstants.MAX_COMPS_IN_SCAN)
676  {
677  m_cinfo.ErrExit(JMessageCode.JERR_COMPONENT_COUNT, m_cinfo.m_comps_in_scan, JpegConstants.MAX_COMPS_IN_SCAN);
678  }
679 
680  /* Overall image size in MCUs */
681  m_cinfo.m_MCUs_per_row = (int)JpegUtils.jdiv_round_up(
682  m_cinfo.imageWidth, m_cinfo.m_max_h_samp_factor * m_cinfo.block_size);
683 
684  m_cinfo.m_MCU_rows_in_scan = (int)JpegUtils.jdiv_round_up(
685  m_cinfo.imageHeight, m_cinfo.m_maxVSampleFactor * m_cinfo.block_size);
686 
687  m_cinfo.m_blocks_in_MCU = 0;
688 
689  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
690  {
691  var componentInfo = m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]];
692 
693  /* Sampling factors give # of blocks of component in each MCU */
694  componentInfo.MCU_width = componentInfo.H_samp_factor;
695  componentInfo.MCU_height = componentInfo.V_samp_factor;
696  componentInfo.MCU_blocks = componentInfo.MCU_width * componentInfo.MCU_height;
697  componentInfo.MCU_sample_width = componentInfo.MCU_width * componentInfo.DCT_h_scaled_size;
698 
699  /* Figure number of non-dummy blocks in last MCU column & row */
700  var tmp = componentInfo.Width_in_blocks % componentInfo.MCU_width;
701  if (tmp == 0)
702  {
703  tmp = componentInfo.MCU_width;
704  }
705 
706  componentInfo.last_col_width = tmp;
707 
708  tmp = componentInfo.height_in_blocks % componentInfo.MCU_height;
709  if (tmp == 0)
710  {
711  tmp = componentInfo.MCU_height;
712  }
713 
714  componentInfo.last_row_height = tmp;
715 
716  /* Prepare array describing MCU composition */
717  var mcublks = componentInfo.MCU_blocks;
718  if (m_cinfo.m_blocks_in_MCU + mcublks > JpegConstants.D_MAX_BLOCKS_IN_MCU)
719  {
720  m_cinfo.ErrExit(JMessageCode.JERR_BAD_MCU_SIZE);
721  }
722 
723  m_cinfo.CompInfo[m_cinfo.m_cur_comp_info[ci]] = componentInfo;
724 
725  while (mcublks-- > 0)
726  {
727  m_cinfo.m_MCU_membership[m_cinfo.m_blocks_in_MCU++] = ci;
728  }
729  }
730  }
731  }
732  }
