181  internal static byte[] compressBytes(byte[] ori, int offset, int len, bool compress) {
182  try {
183  MemoryStream inb = new MemoryStream(ori, offset, len);
184  Stream inx = inb;
185  if (!compress) inx = ZlibStreamFactory.createZlibInputStream(inb);
186  MemoryStream outb = new MemoryStream();
187  Stream outx = outb;
188  if (compress) outx = ZlibStreamFactory.createZlibOutputStream(outb);
189  shovelInToOut(inx, outx);
190  inx.Close();
191  outx.Close();
192  byte[] res = outb.ToArray();
193  return res;
194  } catch (Exception e) {
195  throw new PngjException(e);
196  }
197  }

246  /// <summary>
247  /// Ad-hoc criteria for 'equivalent' chunks.
248  /// </summary>
249  /// <remarks>
250  /// Two chunks are equivalent if they have the same Id AND either:
251  /// 1. they are Single
252  /// 2. both are textual and have the same key
253  /// 3. both are SPLT and have the same palette name
254  /// Bear in mind that this is an ad-hoc, non-standard, nor required (nor wrong)
255  /// criterion. Use it only if you find it useful. Notice that PNG allows to have
256  /// repeated textual keys with same keys.
257  /// </remarks>
258  /// <param name="c1">Chunk1</param>
259  /// <param name="c2">Chunk1</param>
260  /// <returns>true if equivalent</returns>
261  public static bool Equivalent(PngChunk c1, PngChunk c2) {
262  if (c1 == c2)
263  return true;
264  if (c1 == null || c2 == null || !c1.Id.Equals(c2.Id))
265  return false;
266  // same id
267  if (c1.GetType() != c2.GetType())
268  return false; // should not happen
269  if (!c2.AllowsMultiple())
270  return true;
271  if (c1 is PngChunkTextVar) {
272  return ((PngChunkTextVar)c1).GetKey().Equals(((PngChunkTextVar)c2).GetKey());
273  }
274  if (c1 is PngChunkSPLT) {
275  return ((PngChunkSPLT)c1).PalName.Equals(((PngChunkSPLT)c2).PalName);
276  }
277  // unknown chunks that allow multiple? consider they don't match
278  return false;
279  }
