345  /// <summary>
346  /// Ad-hoc criteria for 'equivalent' chunks.
347  /// </summary>
348  /// <remarks>
349  /// Two chunks are equivalent if they have the same Id AND either:
350  /// 1. they are Single
351  /// 2. both are textual and have the same key
352  /// 3. both are SPLT and have the same palette name
353  /// Bear in mind that this is an ad-hoc, non-standard, nor required (nor wrong)
354  /// criterion. Use it only if you find it useful. Notice that PNG allows to have
355  /// repeated textual keys with same keys.
356  /// </remarks>
357  /// <param name="c1">Chunk1</param>
358  /// <param name="c2">Chunk1</param>
359  /// <returns>true if equivalent</returns>
360  public static bool Equivalent(AbstractPngChunk c1, AbstractPngChunk c2)
361  {
362  if (c1 == c2)
363  {
364  return true;
365  }
366 
367  if (c1 is null
368  || c2 is null
369  || !c1.Id.Equals(c2.Id, StringComparison.Ordinal))
370  {
371  return false;
372  }
373 
374  // same id
375  if (c1.GetType() != c2.GetType())
376  {
377  return false; // should not happen
378  }
379 
380  if (!c2.AllowsMultiple())
381  {
382  return true;
383  }

386  {
