174  internal int WriteChunks(Stream os, int currentGroup)
175  {
176  var written = new List<int>();
177  for (var i = 0; i < queuedChunks.Count; i++)
178  {
179  var c = queuedChunks[i];
180  if (!ShouldWrite(c, currentGroup))
181  {
182  continue;
183  }
184 
185  if (ChunkHelper.IsCritical(c.Id)
186  && !c.Id.Equals(ChunkHelper.PLTE, System.StringComparison.Ordinal))
187  {
188  throw new PngjOutputException("bad chunk queued: " + c);
189  }
190 
191  if (alreadyWrittenKeys.ContainsKey(c.Id) && !c.AllowsMultiple())
192  {
193  throw new PngjOutputException("duplicated chunk does not allow multiple: " + c);
194  }
195 
196  c.Write(os);
197  PngChunks.Add(c);
198  alreadyWrittenKeys[c.Id] = alreadyWrittenKeys.ContainsKey(c.Id) ? alreadyWrittenKeys[c.Id] + 1 : 1;
199  written.Add(i);
200  c.ChunkGroup = currentGroup;
201  }
202 
203  for (var k = written.Count - 1; k >= 0; k--)
204  {
205 
206  queuedChunks.RemoveAt(written[k]);
207  }
208 
209  return written.Count;
210  }
