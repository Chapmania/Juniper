107  private static bool shouldWrite(PngChunk c, int currentGroup) {
108  if (currentGroup == CHUNK_GROUP_2_PLTE)
109  return c.Id.Equals(ChunkHelper.PLTE);
110  if (currentGroup % 2 == 0)
111  throw new PngjOutputException("bad chunk group?");
112  int minChunkGroup, maxChunkGroup;
113  if (c.mustGoBeforePLTE())
114  minChunkGroup = maxChunkGroup = ChunksList.CHUNK_GROUP_1_AFTERIDHR;
115  else if (c.mustGoBeforeIDAT()) {
116  maxChunkGroup = ChunksList.CHUNK_GROUP_3_AFTERPLTE;
117  minChunkGroup = c.mustGoAfterPLTE() ? ChunksList.CHUNK_GROUP_3_AFTERPLTE
118  : ChunksList.CHUNK_GROUP_1_AFTERIDHR;
119  } else {
120  maxChunkGroup = ChunksList.CHUNK_GROUP_5_AFTERIDAT;
121  minChunkGroup = ChunksList.CHUNK_GROUP_1_AFTERIDHR;
122  }
123 
124  int preferred = maxChunkGroup;
125  if (c.Priority)
126  preferred = minChunkGroup;
127  if (ChunkHelper.IsUnknown(c) && c.ChunkGroup > 0)
128  preferred = c.ChunkGroup;
129  if (currentGroup == preferred)
130  return true;
131  if (currentGroup > preferred && currentGroup <= maxChunkGroup)
132  return true;
133  return false;
134  }

136  internal int writeChunks(Stream os, int currentGroup) {
137  List<int> written = new List<int>();
138  for (int i = 0; i < queuedChunks.Count; i++) {
139  PngChunk c = queuedChunks[i];
140  if (!shouldWrite(c, currentGroup))
141  continue;
142  if (ChunkHelper.IsCritical(c.Id) && !c.Id.Equals(ChunkHelper.PLTE))
143  throw new PngjOutputException("bad chunk queued: " + c);
144  if (alreadyWrittenKeys.ContainsKey(c.Id) && !c.AllowsMultiple())
145  throw new PngjOutputException("duplicated chunk does not allow multiple: " + c);
146  c.write(os);
147  chunks.Add(c);
148  alreadyWrittenKeys[c.Id] = alreadyWrittenKeys.ContainsKey(c.Id) ? alreadyWrittenKeys[c.Id] + 1 : 1;
149  written.Add(i);
150  c.ChunkGroup = currentGroup;
151  }
152  for (int k = written.Count - 1; k >= 0; k--) {
153  queuedChunks.RemoveAt(written[k]);
154  }
155  return written.Count;
156  }
