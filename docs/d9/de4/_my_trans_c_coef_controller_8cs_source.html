8  /// <summary>
9  /// This is a special implementation of the coefficient
10  /// buffer controller. This is similar to jccoefct.c, but it handles only
11  /// output from presupplied virtual arrays. Furthermore, we generate any
12  /// dummy padding blocks on-the-fly rather than expecting them to be present
13  /// in the arrays.
14  /// </summary>
15  internal class MyTransCCoefController : JpegCCoefController
16  {
17  private readonly JpegCompressStruct m_cinfo;
18 
19  private int m_iMCU_row_num; /* iMCU row # within image */
20  private int m_mcu_ctr; /* counts MCUs processed in current row */
21  private int m_MCU_vert_offset; /* counts MCU rows within iMCU row */
22  private int m_MCU_rows_per_iMCU_row; /* number of such rows needed */
23 
24  /* Virtual block array for each component. */
25  private readonly JVirtArray<JBlock>[] m_whole_image;
26 
27  /* Workspace for constructing dummy blocks at right/bottom edges. */
28  private readonly JBlock[][] m_dummy_buffer = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU][];

30  /// <summary>
31  /// <para>Initialize coefficient buffer controller.</para>
32  /// <para>
33  /// Each passed coefficient array must be the right size for that
34  /// coefficient: width_in_blocks wide and height_in_blocks high,
35  /// with unit height at least v_samp_factor.
36  /// </para>
37  /// </summary>
38  public MyTransCCoefController(JpegCompressStruct cinfo, JVirtArray<JBlock>[] coef_arrays)
39  {
40  m_cinfo = cinfo;
41 
42  /* Save pointer to virtual arrays */
43  m_whole_image = coef_arrays;
44 
45  /* Allocate and pre-zero space for dummy DCT blocks. */
46  var buffer = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU];
47  for (var i = 0; i < JpegConstants.C_MAX_BLOCKS_IN_MCU; i++)
48  {
49  buffer[i] = new JBlock();
50  }
51 
52  for (var i = 0; i < JpegConstants.C_MAX_BLOCKS_IN_MCU; i++)
53  {
54  m_dummy_buffer[i] = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU - i];
55  for (var j = i; j < JpegConstants.C_MAX_BLOCKS_IN_MCU; j++)
56  {
57  m_dummy_buffer[i][j - i] = buffer[j];
58  }
59  }
60  }

76  /// <summary>
77  /// <para>
78  /// Process some data.
79  /// We process the equivalent of one fully interleaved MCU row ("iMCU" row)
80  /// per call, ie, v_samp_factor block rows for each component in the scan.
81  /// The data is obtained from the virtual arrays and fed to the entropy coder.
82  /// Returns true if the iMCU row is completed, false if suspended.
83  /// </para>
84  /// <para>NB: input_buf is ignored; it is likely to be a null pointer.</para>
85  /// </summary>
86  public virtual bool CompressData(byte[][][] input_buf)
87  {
88  /* Align the virtual buffers for the components used in this scan. */
89  var buffer = new JBlock[JpegConstants.MAX_COMPS_IN_SCAN][][];
90  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
91  {
92  var componentInfo = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[ci]];
93  buffer[ci] = m_whole_image[componentInfo.Component_index].Access(
94  m_iMCU_row_num * componentInfo.V_samp_factor, componentInfo.V_samp_factor);
95  }
96 
97  /* Loop to process one whole iMCU row */
98  var last_MCU_col = m_cinfo.m_MCUs_per_row - 1;
99  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
100  var MCU_buffer = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU][];
101  for (var yoffset = m_MCU_vert_offset; yoffset < m_MCU_rows_per_iMCU_row; yoffset++)
102  {
103  for (var MCU_col_num = m_mcu_ctr; MCU_col_num < m_cinfo.m_MCUs_per_row; MCU_col_num++)
104  {
105  /* Construct list of pointers to DCT blocks belonging to this MCU */
106  var blkn = 0; /* index of current DCT block within MCU */
107  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
108  {
109  var componentInfo = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[ci]];
110  var start_col = MCU_col_num * componentInfo.MCU_width;
111  var blockcnt = (MCU_col_num < last_MCU_col) ? componentInfo.MCU_width : componentInfo.last_col_width;
112  for (var yindex = 0; yindex < componentInfo.MCU_height; yindex++)
113  {
114  int xindex;
115  if (m_iMCU_row_num < last_iMCU_row || yindex + yoffset < componentInfo.last_row_height)
116  {
117  /* Fill in pointers to real blocks in this row */
118  for (xindex = 0; xindex < blockcnt; xindex++)
119  {
120  var bufLength = buffer[ci][yindex + yoffset].Length;
121  var start = start_col + xindex;
122  MCU_buffer[blkn] = new JBlock[bufLength - start];
123  for (var j = start; j < bufLength; j++)
124  {
125  MCU_buffer[blkn][j - start] = buffer[ci][yindex + yoffset][j];
126  }
127 
128  blkn++;
129  }
130  }
131  else
132  {
133  /* At bottom of image, need a whole row of dummy blocks */
134  xindex = 0;
135  }
136 
137  /* Fill in any dummy blocks needed in this row.
138  * Dummy blocks are filled in the same way as in jccoefct.c:
139  * all zeroes in the AC entries, DC entries equal to previous
140  * block's DC value. The init routine has already zeroed the
141  * AC entries, so we need only set the DC entries correctly.
142  */
143  for (; xindex < componentInfo.MCU_width; xindex++)
144  {
145  MCU_buffer[blkn] = m_dummy_buffer[blkn];
146  MCU_buffer[blkn][0][0] = MCU_buffer[blkn - 1][0][0];
147  blkn++;
148  }
149  }
150  }
151 
152  /* Try to write the MCU. */
153  if (!m_cinfo.m_entropy.encodeMcu(MCU_buffer))
154  {
155  /* Suspension forced; update state counters and exit */
156  m_MCU_vert_offset = yoffset;
157  m_mcu_ctr = MCU_col_num;
158  return false;
159  }
160  }
161 
162  /* Completed an MCU row, but perhaps not an iMCU row */
163  m_mcu_ctr = 0;
164  }
165 
166  /* Completed the iMCU row, advance counters for next one */
167  m_iMCU_row_num++;
168  StartIMcuRow();
169  return true;
170  }
171 
172  /// <summary>
173  /// Reset within-iMCU-row counters for a new row
174  /// </summary>
175  private void StartIMcuRow()
176  {
177  /* In an interleaved scan, an MCU row is the same as an iMCU row.
178  * In a noninterleaved scan, an iMCU row has v_samp_factor MCU rows.
179  * But at the bottom of the image, process only what's left.
180  */
181  if (m_cinfo.m_comps_in_scan > 1)
182  {
183  m_MCU_rows_per_iMCU_row = 1;
184  }
185  else if (m_iMCU_row_num < (m_cinfo.m_total_iMCU_rows - 1))
186  {
187  m_MCU_rows_per_iMCU_row = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[0]].V_samp_factor;
188  }
189  else
190  {
191  m_MCU_rows_per_iMCU_row = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[0]].last_row_height;
192  }
193 
194  m_mcu_ctr = 0;
195  m_MCU_vert_offset = 0;
196  }
197  }
