543  private static void Recompile(bool advanceBuildStep, params PlatformConfiguration[] platforms)
544  {
545  if (advanceBuildStep)
546  {
547  ++BuildProgress;
548  }
549 
550  var prog = PrepareBuildStep(1);
551 
552  var commonDefines = platforms
553  .SelectMany(p => p.CompilerDefines)
554  .GroupBy(d => d)
555  .Where(defs => defs.Count() == platforms.Length)
556  .SelectMany(defs => defs)
557  .Distinct()
558  .ToList();
559 
560  var currentDefines = CleanupDefines(PlayerSettings
561  .GetScriptingDefineSymbolsForGroup(platforms[0].TargetGroup)
562  .Split(';'))
563  .ToList();
564 
565  var nonJuniper = currentDefines
566  .Exclude(JUNIPER_DEFINES)
567  .Where(def => !def.StartsWith("NO_UNITY"))
568  .Distinct()
569  .ToList();
570 
571  var nextDefines = nonJuniper
572  .Union(commonDefines)
573  .ToList();
574 
575  nextDefines = CleanupDefines(nextDefines);
576 
577  if (nextDefines.Matches(currentDefines))
578  {
579  if (nextDefines.Contains(RECOMPILE_SLUG))
580  {
581  nextDefines.Remove(RECOMPILE_SLUG);
582  }
583  else
584  {
585  nextDefines.Add(RECOMPILE_SLUG);
586  }
587  }
588 
589  var nextDefinesString = string.Join(";", nextDefines);
590 
591  for (var i = platforms.Length - 1; i >= 0; --i)
592  {
593  PlayerSettings.SetScriptingDefineSymbolsForGroup(platforms[i].TargetGroup, nextDefinesString);
594  }
595 
596  DelayedUpdate(prog, 30, () => Debug.LogWarning("Timeout!!!"));
597  }
