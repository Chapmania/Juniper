561  private static void InternalCompile(bool advanceBuildStep, params PlatformConfiguration[] platforms)
562  {
563  if (advanceBuildStep)
564  {
565  ++BuildProgress;
566  }
567 
568  var commonDefines = platforms
569  .SelectMany(p => p.CompilerDefines)
570  .GroupBy(d => d)
571  .Where(defs => defs.Count() == platforms.Length)
572  .SelectMany(defs => defs)
573  .Distinct()
574  .ToList();
575 
576  var currentDefines = CleanupDefines(PlayerSettings
577  .GetScriptingDefineSymbolsForGroup(platforms[0].TargetGroup)
578  .Split(';'))
579  .ToList();
580 
581  var nonJuniper = currentDefines
582  .Exclude(JUNIPER_DEFINES)
583  .Where(def => !def.StartsWith("NO_UNITY"))
584  .Distinct()
585  .ToList();
586 
587  var nextDefines = nonJuniper
588  .Union(commonDefines)
589  .ToList();
590 
591  nextDefines = CleanupDefines(nextDefines);
592 
593  if (nextDefines.Matches(currentDefines))
594  {
595  if (nextDefines.Contains(RECOMPILE_SLUG))
596  {
597  nextDefines.Remove(RECOMPILE_SLUG);
598  }
599  else
600  {
601  nextDefines.Add(RECOMPILE_SLUG);
602  }
603  }
604 
605  var nextDefinesString = string.Join(";", nextDefines);
606 
607  var targetGroups = platforms
608  .Select(p => p.TargetGroup)
609  .Distinct()
610  .Reverse()
611  .ToArray();
612 
613  foreach(var targetGroup in targetGroups)
614  {
615  PlayerSettings.SetScriptingDefineSymbolsForGroup(targetGroup, nextDefinesString);
616  }
617  }
