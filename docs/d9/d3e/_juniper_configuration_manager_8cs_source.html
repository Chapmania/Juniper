546  private static void Recompile(bool advanceBuildStep, string nextDefinesString = null)
547  {
548  if (advanceBuildStep)
549  {
550  ++BuildProgress;
551  }
552  var targetGroup = BuildPipeline.GetBuildTargetGroup(EditorUserBuildSettings.activeBuildTarget);
553  var prog = PrepareBuildStep(1);
554  var currentDefinesString = PlayerSettings.GetScriptingDefineSymbolsForGroup(targetGroup);
555  var currentDefines = FormatDefines(ref currentDefinesString);
556  var nextDefines = FormatDefines(ref nextDefinesString);
557 
558  if (nextDefines.Matches(currentDefines))
559  {
560  if (nextDefines.Contains(RECOMPILE_SLUG))
561  {
562  nextDefines.Remove(RECOMPILE_SLUG);
563  }
564  else
565  {
566  nextDefines.Add(RECOMPILE_SLUG);
567  }
568  }
569 
570  nextDefines.RemoveAll(string.IsNullOrWhiteSpace);
571 
572  nextDefinesString = string.Join(";", nextDefines);
573 
574  PlayerSettings.SetScriptingDefineSymbolsForGroup(targetGroup, nextDefinesString);
