79  /// <summary>
80  /// Conditionally emit a trace or warning message.
81  /// </summary>
82  /// <param name="msg_level">The message severity level.<br/>
83  /// Values are:<br/>
84  /// -1: recoverable corrupt-data warning, may want to abort.<br/>
85  /// 0: important advisory messages (always display to user).<br/>
86  /// 1: first level of tracing detail.<br/>
87  /// 2, 3, ...: successively more detailed tracing messages.
88  /// </param>
89  /// <remarks><para>
90  /// The main reason for overriding this method would be to abort on warnings.
91  /// This method calls <see cref="JpegErrorMessage.OutputMessage">output_message</see> for message showing.<br/>
92  /// </para>
93  /// <para>
94  /// An application might override this method if it wanted to abort on
95  /// warnings or change the policy about which messages to display.
96  /// </para>
97  /// </remarks>
98  /// <seealso href="41dc1a3b-0dea-4594-87d2-c213ab1049e1.htm" target="_self">Error handling</seealso>
99  public virtual void EmitMessage(int msg_level)
100  {
101  if (msg_level < 0)
102  {
103  /* It's a warning message. Since corrupt files may generate many warnings,
104  * the policy implemented here is to show only the first warning,
105  * unless trace_level >= 3.
106  */
107  if (m_numWarnings == 0 || m_traceLevel >= 3)
108  {
109  OutputMessage();
110  }
111 
112  /* Always count warnings in num_warnings. */
113  m_numWarnings++;
114  }
115  else
116  {
117  /* It's a trace message. Show it if trace_level >= msg_level. */
118  if (m_traceLevel >= msg_level)
119  {
120  OutputMessage();
121  }
122  }
123  }

125  /// <summary>
126  /// Actual output of any JPEG message.
127  /// </summary>
128  /// <remarks>Override this to send messages somewhere other than Console.
129  /// Note that this method does not know how to generate a message, only where to send it.
130  /// For extending a generation of messages see <see cref="JpegErrorMessage.FormatMessage">format_message</see>.
131  /// </remarks>

133  public virtual void OutputMessage()

135  // Create the message

137 
138  // Send it to console, adding a newline */
139  Console.WriteLine(buffer);
140  }

142  /// <summary>
143  /// Constructs a readable error message string.
144  /// </summary>
145  /// <remarks>This method is called by <see cref="JpegErrorMessage.OutputMessage">output_message</see>.
146  /// Few applications should need to override this method. One possible reason for doing so is to
147  /// implement dynamic switching of error message language.</remarks>
148  /// <returns>The formatted message</returns>
149  /// <seealso href="41dc1a3b-0dea-4594-87d2-c213ab1049e1.htm" target="_self">Error handling</seealso>
150  public virtual string FormatMessage()
151  {
152  var msgtext = GetMessageText(msgCode);
153  if (msgtext is null)
154  {
155  msgParam = new object[] { msgCode };
156  msgtext = GetMessageText(0);
157  }
158 
159  /* Format the message into the passed buffer */
160  return string.Format(CultureInfo.CurrentCulture, msgtext, msgParam);
161  }
