15  /// <summary>
16  /// <para>Compression preprocessing (downsampling input buffer control).</para>
17  /// <para>
18  /// For the simple (no-context-row) case, we just need to buffer one
19  /// row group's worth of pixels for the downsampling step. At the bottom of
20  /// the image, we pad to a full row group by replicating the last pixel row.
21  /// The downsampler's last output row is then replicated if needed to pad
22  /// out to a full iMCU row.
23  /// </para>
24  /// <para>
25  /// When providing context rows, we must buffer three row groups' worth of
26  /// pixels. Three row groups are physically allocated, but the row pointer
27  /// arrays are made five row groups high, with the extra pointers above and
28  /// below "wrapping around" to point to the last and first real row groups.
29  /// This allows the downsampler to access the proper context rows.
30  /// At the top and bottom of the image, we create dummy context rows by
31  /// copying the first or last real pixel row. This copying could be avoided
32  /// by pointer hacking as is done in jdmainct.c, but it doesn't seem worth the
33  /// trouble on the compression side.
34  /// </para>
35  /// </summary>
36  internal class JpegCPrepController
37  {
38  private readonly JpegCompressStruct cinfo;

40  /* Downsampling input buffer. This buffer holds color-converted data
41  * until we have enough to do a downsample step.
42  */
43  private readonly byte[][][] colorBuf = new byte[JpegConstants.MAX_COMPONENTS][][];
44  private int colorBufRowsOffset;
45 
46  private int rowsToGo; /* counts rows remaining in source image */
47  private int nextBufRow; /* index of next row to store in color_buf */
48 
49  private int thisRowGroup; /* starting row index of group to process */
50  private int nextBufStop; /* downsample when we reach this index */

52  public JpegCPrepController(JpegCompressStruct cinfo)
53  {
54  this.cinfo = cinfo;
55 
56  /* Allocate the color conversion buffer.
57  * We make the buffer wide enough to allow the downsampler to edge-expand
58  * horizontally within the buffer, if it so chooses.
59  */
60  if (cinfo.m_downsample.NeedContextRows())
61  {
62  /* Set up to provide context rows */
63  CreateContextBuffer();
64  }
65  else
66  {
67  /* No context, just make it tall enough for one row group */
68  for (var ci = 0; ci < cinfo.m_num_components; ci++)
69  {
70  colorBufRowsOffset = 0;
71  colorBuf[ci] = JpegCommonStruct.AllocJpegSamples(
72  (cinfo.Component_info[ci].Width_in_blocks *
73  cinfo.min_DCT_h_scaled_size * cinfo.m_max_h_samp_factor) /
74  cinfo.Component_info[ci].H_samp_factor,
75  cinfo.m_max_v_samp_factor);
76  }
77  }

117  /// <summary>
118  /// Create the wrapped-around downsampling input buffer needed for context mode.
119  /// </summary>
120  private void CreateContextBuffer()
121  {
122  var rgroup_height = cinfo.m_max_v_samp_factor;
123  for (var ci = 0; ci < cinfo.m_num_components; ci++)
124  {
125  var samplesPerRow = (cinfo.Component_info[ci].Width_in_blocks *
126  cinfo.min_DCT_h_scaled_size * cinfo.m_max_h_samp_factor) /
127  cinfo.Component_info[ci].H_samp_factor;
128 
129  var fake_buffer = new byte[5 * rgroup_height][];
130  for (var i = 1; i < 4 * rgroup_height; i++)
131  {
132  fake_buffer[i] = new byte[samplesPerRow];
133  }
134 
135  /* Allocate the actual buffer space (3 row groups) for this component.
136  * We make the buffer wide enough to allow the downsampler to edge-expand
137  * horizontally within the buffer, if it so chooses.
138  */
139  var true_buffer = JpegCommonStruct.AllocJpegSamples(samplesPerRow, 3 * rgroup_height);
140 
141  /* Copy true buffer row pointers into the middle of the fake row array */
142  for (var i = 0; i < 3 * rgroup_height; i++)
143  {
144  fake_buffer[rgroup_height + i] = true_buffer[i];
145  }
146 
147  /* Fill in the above and below wraparound pointers */
148  for (var i = 0; i < rgroup_height; i++)
149  {
150  fake_buffer[i] = true_buffer[(2 * rgroup_height) + i];
151  fake_buffer[(4 * rgroup_height) + i] = true_buffer[i];
152  }
153 
154  colorBuf[ci] = fake_buffer;
155  colorBufRowsOffset = rgroup_height;
156  }
157  }

159  /// <summary>
160  /// <para>Process some data in the simple no-context case.</para>
161  /// <para>
162  /// Preprocessor output data is counted in "row groups". A row group
163  /// is defined to be v_samp_factor sample rows of each component.
164  /// Downsampling will produce this much data from each max_v_samp_factor
165  /// input rows.
166  /// </para>
167  /// </summary>
168  private void PreProcessWithoutContext(byte[][] input_buf, ref int in_row_ctr, int in_rows_avail, byte[][][] output_buf, ref int out_row_group_ctr, int out_row_groups_avail)
169  {
170  while (in_row_ctr < in_rows_avail && out_row_group_ctr < out_row_groups_avail)
171  {
172  /* Do color conversion to fill the conversion buffer. */
173  var inrows = in_rows_avail - in_row_ctr;
174  var numrows = cinfo.m_max_v_samp_factor - nextBufRow;
175  numrows = Math.Min(numrows, inrows);
176  cinfo.m_cconvert.color_convert(input_buf, in_row_ctr, colorBuf, colorBufRowsOffset + nextBufRow, numrows);
177  in_row_ctr += numrows;
178  nextBufRow += numrows;
179  rowsToGo -= numrows;
180 
181  /* If at bottom of image, pad to fill the conversion buffer. */
182  if (rowsToGo == 0 && nextBufRow < cinfo.m_max_v_samp_factor)
183  {
184  for (var ci = 0; ci < cinfo.m_num_components; ci++)
185  {
186  ExpandBottomEdge(colorBuf[ci], colorBufRowsOffset, cinfo.m_image_width, nextBufRow, cinfo.m_max_v_samp_factor);
187  }
188 
189  nextBufRow = cinfo.m_max_v_samp_factor;
190  }
191 
192  /* If we've filled the conversion buffer, empty it. */
193  if (nextBufRow == cinfo.m_max_v_samp_factor)
194  {
195  cinfo.m_downsample.Downsample(colorBuf, colorBufRowsOffset, output_buf, out_row_group_ctr);
196  nextBufRow = 0;
197  out_row_group_ctr++;
198  }
199 
200  /* If at bottom of image, pad the output to a full iMCU height.
201  * Note we assume the caller is providing a one-iMCU-height output buffer!
202  */
203  if (rowsToGo == 0 && out_row_group_ctr < out_row_groups_avail)
204  {
205  for (var ci = 0; ci < cinfo.m_num_components; ci++)
206  {
207  var componentInfo = cinfo.Component_info[ci];
208  numrows = (componentInfo.V_samp_factor * componentInfo.DCT_v_scaled_size) /
209  cinfo.min_DCT_v_scaled_size;
210 
211  ExpandBottomEdge(output_buf[ci], 0,
212  componentInfo.Width_in_blocks * componentInfo.DCT_h_scaled_size,
213  out_row_group_ctr * numrows,
214  out_row_groups_avail * numrows);
215  }
216 
217  out_row_group_ctr = out_row_groups_avail;
218  break; /* can exit outer loop without test */
219  }
220  }
221  }

223  /// <summary>
224  /// Process some data in the context case.
225  /// </summary>
226  private void PreProcessContext(byte[][] input_buf, ref int in_row_ctr, int in_rows_avail, byte[][][] output_buf, ref int out_row_group_ctr, int out_row_groups_avail)
227  {
228  while (out_row_group_ctr < out_row_groups_avail)
229  {
230  if (in_row_ctr < in_rows_avail)
231  {
232  /* Do color conversion to fill the conversion buffer. */
233  var inrows = in_rows_avail - in_row_ctr;
234  var numrows = nextBufStop - nextBufRow;
235  numrows = Math.Min(numrows, inrows);
236  cinfo.m_cconvert.color_convert(input_buf, in_row_ctr, colorBuf, colorBufRowsOffset + nextBufRow, numrows);
237 
238  /* Pad at top of image, if first time through */
239  if (rowsToGo == cinfo.m_image_height)
240  {
241  for (var ci = 0; ci < cinfo.m_num_components; ci++)
242  {
243  for (var row = 1; row <= cinfo.m_max_v_samp_factor; row++)
244  {
245  JpegUtils.jcopy_sample_rows(colorBuf[ci], colorBufRowsOffset, colorBuf[ci], colorBufRowsOffset - row, 1, cinfo.m_image_width);
246  }
247  }
248  }
249 
250  in_row_ctr += numrows;
251  nextBufRow += numrows;
252  rowsToGo -= numrows;
253  }
254  else
255  {
256  /* Return for more data, unless we are at the bottom of the image. */
257  if (rowsToGo != 0)
258  {
259  break;
260  }
261 
262  /* When at bottom of image, pad to fill the conversion buffer. */
263  if (nextBufRow < nextBufStop)
264  {
265  for (var ci = 0; ci < cinfo.m_num_components; ci++)
266  {
267  ExpandBottomEdge(colorBuf[ci], colorBufRowsOffset, cinfo.m_image_width, nextBufRow, nextBufStop);
268  }
269 
270  nextBufRow = nextBufStop;
271  }
272  }
273 
274  /* If we've gotten enough data, downsample a row group. */
275  if (nextBufRow == nextBufStop)
276  {
277  cinfo.m_downsample.Downsample(colorBuf, colorBufRowsOffset + thisRowGroup, output_buf, out_row_group_ctr);
278  out_row_group_ctr++;
279 
280  /* Advance pointers with wraparound as necessary. */
281  thisRowGroup += cinfo.m_max_v_samp_factor;
282  var buf_height = cinfo.m_max_v_samp_factor * 3;
283 
284  if (thisRowGroup >= buf_height)
285  {
286  thisRowGroup = 0;
287  }
288 
289  if (nextBufRow >= buf_height)
290  {
291  nextBufRow = 0;
292  }
293 
294  nextBufStop = nextBufRow + cinfo.m_max_v_samp_factor;
295  }
296  }
297  }
