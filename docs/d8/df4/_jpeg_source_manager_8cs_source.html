69  /// This is the default resync_to_restart method for data source
70  /// managers to use if they don't have any better approach.
71  /// </summary>
72  /// <param name="cinfo">An instance of <see cref="JpegDecompressStruct"/></param>
73  /// <param name="desired">The desired</param>
74  /// <returns><c>false</c> if suspension is required.</returns>
75  /// <remarks><para>
76  /// That method assumes that no backtracking is possible.
77  /// Some data source managers may be able to back up, or may have
78  /// additional knowledge about the data which permits a more
79  /// intelligent recovery strategy; such managers would
80  /// presumably supply their own resync method.<br/><br/>
81  /// </para>
82  /// <para>
83  /// read_restart_marker calls resync_to_restart if it finds a marker other than
84  /// the restart marker it was expecting. (This code is *not* used unless
85  /// a nonzero restart interval has been declared.) cinfo.unread_marker is
86  /// the marker code actually found (might be anything, except 0 or FF).
87  /// The desired restart marker number (0..7) is passed as a parameter.<br/><br/>
88  /// </para>
89  /// <para>
90  /// This routine is supposed to apply whatever error recovery strategy seems
91  /// appropriate in order to position the input stream to the next data segment.
92  /// Note that cinfo.unread_marker is treated as a marker appearing before
93  /// the current data-source input point; usually it should be reset to zero
94  /// before returning.<br/><br/>
95  /// </para>
96  /// <para>
97  /// This implementation is substantially constrained by wanting to treat the
98  /// input as a data stream; this means we can't back up. Therefore, we have
99  /// only the following actions to work with:<br/>
100  /// 1. Simply discard the marker and let the entropy decoder resume at next
101  /// byte of file.<br/>
102  /// 2. Read forward until we find another marker, discarding intervening
103  /// data. (In theory we could look ahead within the current bufferload,
104  /// without having to discard data if we don't find the desired marker.
105  /// This idea is not implemented here, in part because it makes behavior
106  /// dependent on buffer size and chance buffer-boundary positions.)<br/>
107  /// 3. Leave the marker unread (by failing to zero cinfo.unread_marker).
108  /// This will cause the entropy decoder to process an empty data segment,
109  /// inserting dummy zeroes, and then we will reprocess the marker.<br/>
110  /// </para>
111  /// <para>
112  /// #2 is appropriate if we think the desired marker lies ahead, while #3 is
113  /// appropriate if the found marker is a future restart marker (indicating
114  /// that we have missed the desired restart marker, probably because it got
115  /// corrupted).<br/>
116  /// We apply #2 or #3 if the found marker is a restart marker no more than
117  /// two counts behind or ahead of the expected one. We also apply #2 if the
118  /// found marker is not a legal JPEG marker code (it's certainly bogus data).
119  /// If the found marker is a restart marker more than 2 counts away, we do #1
120  /// (too much risk that the marker is erroneous; with luck we will be able to
121  /// resync at some future point).<br/>
122  /// For any valid non-restart JPEG marker, we apply #3. This keeps us from
123  /// overrunning the end of a scan. An implementation limited to single-scan
124  /// files might find it better to apply #2 for markers other than EOI, since
125  /// any other marker would have to be bogus data in that case.
126  /// </para></remarks>
127  public virtual bool ReSyncToRestart(JpegDecompressStruct cinfo, int desired)
128  {
129  if (cinfo is null)
130  {
131  throw new System.ArgumentNullException(nameof(cinfo));
132  }
133  /* Always put up a warning. */
134  cinfo.WarnMS(JMessageCode.JWRN_MUST_RESYNC, cinfo.m_unreadMarker, desired);
135 
136  while (true)
137  {
138  /* Outer loop handles repeated decision after scanning forward. */
139  int action;
140  if (cinfo.m_unreadMarker < (int)JpegMarker.SOF0)
141  {
142  /* invalid marker */
143  action = 2;
144  }
145  else if (cinfo.m_unreadMarker < (int)JpegMarker.RST0
146  || cinfo.m_unreadMarker > (int)JpegMarker.RST7)
147  {
148  /* valid non-restart marker */
149  action = 3;
150  }
151  else if (cinfo.m_unreadMarker == ((int)JpegMarker.RST0 + ((desired + 1) & 7))
152  || cinfo.m_unreadMarker == ((int)JpegMarker.RST0 + ((desired + 2) & 7)))
153  {
154  /* one of the next two expected restarts */
155  action = 3;
156  }
157  else if (cinfo.m_unreadMarker == ((int)JpegMarker.RST0 + ((desired - 1) & 7))
158  || cinfo.m_unreadMarker == ((int)JpegMarker.RST0 + ((desired - 2) & 7)))
159  {
160  /* a prior restart, so advance */
161  action = 2;
162  }
163  else
164  {
165  /* desired restart or too far away */
166  action = 1;
167  }
168 
169  cinfo.TraceMS(4, JMessageCode.JTRC_RECOVERY_ACTION, cinfo.m_unreadMarker, action);
170 
171  switch (action)
172  {
173  case 1:
174  /* Discard marker and let entropy decoder resume processing. */
175  cinfo.m_unreadMarker = 0;
176  return true;
177  case 2:
178  /* Scan to the next marker, and repeat the decision loop. */
179  if (!cinfo.m_marker.NextMarker())
180  {
181  return false;
182  }

184  break;
185  case 3:

187  /* Entropy decoder will be forced to process an empty segment. */
188  return true;
