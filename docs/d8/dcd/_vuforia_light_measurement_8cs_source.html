124  void GetAveragePixelIntensity(Vuforia.Image image, GetPixelFromBuffer pixelGetter)
125  {
126  float avgH, avgS, avgV;
127  avgH = avgS = avgV = 0;
128 
129  int bytesPerPixel = image.Stride / image.Width;
130  int numPixels = image.Width * image.Height;
131  for (int r = 0; r < NumSampleRows; ++r)
132  {
133  int y = Random.Range(0, image.Height);
134  for (int c = 0; c < NumSamplesPerRow; ++c)
135  {
136  int x = Random.Range(0, image.Width);
137  int i = y * image.Stride + x * bytesPerPixel;
138 
139  Color pixel = pixelGetter(image.Pixels, i);
140 
141  float h, s, v;
142  Color.RGBToHSV(pixel, out h, out s, out v);
143 
144  avgH += h / numPixels;
145  avgS += s / numPixels;
146  avgV += v / numPixels;
147  }
148  }
149 
150  lastIntensity = avgV;
151  lastColor = Color.HSVToRGB(avgH, avgS, avgV);
