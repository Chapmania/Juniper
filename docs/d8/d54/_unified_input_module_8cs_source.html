365  public override void Process()
366  {
367  Devices.AddRange(newDevices);
368  newDevices.Clear();
369 
370  var motion = mode.HasFlag(Mode.Motion);
371  var hands = mode.HasFlag(Mode.Hands);
372  var mouse = mode.HasFlag(Mode.Mouse)
373  && this.mouse.IsConnected;
374  var touch = mode.HasFlag(Mode.Touch)
375  && TouchConnected
376  && !motion
377  && !hands;
378  var gaze = mode.HasFlag(Mode.Gaze)
379  && !mouse
380  && !TouchConnected
381  && !MotionConnected
382  && !HandsConnected;
383 
384  EnableControllers(motion, false);
385  EnableHands(hands, false);
386  EnableMouse(mouse, false);
387  EnableTouch(touch, false);
388  EnableGaze(gaze, false);
389 
390  foreach (var pointer in Devices)
391  {
392  pointer.Layer = ControllerLayer;
393  if (pointer.IsEnabled)
394  {
395  PointerEventData evtData;
396  GetPointerData(pointer.PointerDataID, out evtData, true);
397  evtData.delta = pointer.ScreenDelta;
398  evtData.position = pointer.ScreenPoint;
399  evtData.scrollDelta = pointer.ScrollDelta;
400  evtData.useDragThreshold = eventSystem.pixelDragThreshold > 0;
401 
402  UpdateRay(pointer, evtData);
403 
404  pointer.Process(evtData, eventSystem.pixelDragThreshold * eventSystem.pixelDragThreshold);
405  }
406  }

409  private void UpdateRay(IPointerDevice pointer, PointerEventData evtData)
410  {
411  var ray = evtData.pointerCurrentRaycast;
412 
413  if (pointer.IsDragging)
414  {
415  var rot = Quaternion.FromToRotation(pointer.LastDirection, pointer.Direction);
416 
417  ray.gameObject = evtData.pointerPress;
418  ray.worldPosition = pointer.Origin + ray.distance * pointer.Direction;
419  ray.worldNormal = rot * ray.worldNormal;
420  ray.screenPosition = pointer.ScreenPoint;
421  }
422  else
423  {
424  RaycastAll(pointer, evtData);
425  if (m_RaycastResultCache.Count > 0)
426  {
427  ray = m_RaycastResultCache[0];
428  evtData.hovered.Clear();
429  foreach (var r in m_RaycastResultCache)
430  {
431  evtData.hovered.Add(r.gameObject);
432  }
433  }
434  else
435  {
436  ray.Clear();
437  ray.worldPosition = pointer.SmoothedWorldPoint;
438  ray.distance = pointer.MinimumPointerDistance;
439  ray.worldNormal = -pointer.Direction;
440  ray.screenPosition = pointer.ScreenPoint;
441  }
442  }
443 
444  evtData.pointerCurrentRaycast = ray;
445  }

450  /// <summary>
451  /// Fire a raycast using all of the GraphicRaycasters in the system, plus the one
452  /// PhysicsRaycaster that is associated with the event Camera.
453  /// </summary>
454  /// <param name="pointer">Pointer.</param>
455  /// <param name="eventData">Event data.</param>
456  private void RaycastAll(IPointerDevice pointer, PointerEventData eventData)
457  {
458  DisplayManager.MoveEventCameraToPointer(pointer);
459 
460  eventSystem.RaycastAll(eventData, m_RaycastResultCache);
461 
462 #if UNITY_MODULES_UI
463  for (var i = 0; i < m_RaycastResultCache.Count; ++i)
464  {
465  var ray = m_RaycastResultCache[i];
466  if (ray.module is GraphicRaycaster)
467  {
468  var gfr = (GraphicRaycaster)ray.module;
469  var canv = gfr.GetComponent<Canvas>();
470  ray.worldNormal = -canv.transform.forward;
471 
472  var pos = pointer.WorldFromScreen(ray.screenPosition);
473  if (canv.renderMode == RenderMode.WorldSpace)
474  {
475  var delta = pos - pointer.Origin;
476  delta *= ray.distance / delta.magnitude + canvasOffset;
477  ray.distance = delta.magnitude;
478  pos = pointer.Origin + delta;
479  }
480  ray.worldPosition = pos;
481 
482  m_RaycastResultCache[i] = ray;
483  }
484  }
485 #endif
