523  private void UpdateRay(IPointerDevice pointer, PointerEventData evtData)
524  {
525  var ray = evtData.pointerCurrentRaycast;
526 
527  if (pointer.IsDragging)
528  {
529  var rot = Quaternion.FromToRotation(pointer.LastDirection, pointer.Direction);
530 
531  ray.gameObject = evtData.pointerPress;
532  ray.worldPosition = pointer.Origin + ray.distance * pointer.Direction;
533  ray.worldNormal = rot * ray.worldNormal;
534  ray.screenPosition = pointer.ScreenPoint;
535  }
536  else
537  {
538  RaycastAll(pointer, evtData);
539  if (m_RaycastResultCache.Count > 0)

542  evtData.hovered.Clear();

544  {
545  evtData.hovered.Add(r.gameObject);
546  }
547  }
548  else
549  {
550  ray.Clear();
551  ray.worldPosition = pointer.SmoothedWorldPoint;
552  ray.distance = pointer.MinimumPointerDistance;
553  ray.worldNormal = -pointer.Direction;
554  ray.screenPosition = pointer.ScreenPoint;
555  }
556  }
