128  private static int RaycastComparer(RaycastResult lhs, RaycastResult rhs)
129  {
130  if (lhs.module.eventCamera != null
131  && rhs.module.eventCamera != null)
132  {
133  return lhs.distance.CompareTo(rhs.distance);
134  }
135  else if (lhs.module.sortOrderPriority != rhs.module.sortOrderPriority)
136  {
137  return lhs.module.sortOrderPriority.CompareTo(rhs.module.sortOrderPriority);
138  }
139  else if (lhs.module.renderOrderPriority != rhs.module.renderOrderPriority)
140  {
141  return lhs.module.renderOrderPriority.CompareTo(rhs.module.renderOrderPriority);
142  }
143  else if (lhs.depth != rhs.depth)
144  {
145  return rhs.depth.CompareTo(lhs.depth);
146  }
147  else if (Mathf.Abs(lhs.distance - rhs.distance) > 0.00001f)
148  {
149  return lhs.distance.CompareTo(rhs.distance);
150  }
151  else
152  {
153  return lhs.index.CompareTo(rhs.index);
154  }

213  private RaycastResult UpdateRay(IPointerDevice pointer, Vector3 lastPointerPosition, PointerEventData evtData)
214  {
215  var ray = evtData.pointerCurrentRaycast;
216  if (pointer.LockedOnTarget)
217  {
218  var from = ray.worldPosition - lastPointerPosition;
219  var to = ray.distance * (pointer.InteractionEndPoint - pointer.transform.position);
220  var delta = to - from;
221  var rot = Quaternion.FromToRotation(from.normalized, to.normalized);
222 
223  ray.gameObject = evtData.pointerPress;
224  ray.worldPosition = (rot * ray.worldPosition) + delta;
225  ray.worldNormal = rot * ray.worldNormal;
226  ray.screenPosition = evtData.position;
227  }
228  else
229  {
230  RaycastAll(pointer, evtData);
231  if (m_RaycastResultCache.Count > 0)
232  {
233  ray = m_RaycastResultCache[0];
234  evtData.hovered.Clear();
235  foreach (var r in m_RaycastResultCache)
236  {
237  evtData.hovered.Add(r.gameObject);
238  }
239  }
240  else
241  {
242  ray.Clear();
243  ray.worldPosition = pointer.InteractionEndPoint;
244  ray.distance = pointer.MinimumPointerDistance;
245  ray.worldNormal = -pointer.InteractionDirection;
246  ray.screenPosition = evtData.position;
247  }
248  }
