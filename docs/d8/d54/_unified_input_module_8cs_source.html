115  /// <param name="rhs">Rhs.</param>
116  private static int RaycastComparer(RaycastResult lhs, RaycastResult rhs)
117  {
118  if (lhs.module.eventCamera != null
119  && rhs.module.eventCamera != null)
120  {
121  return lhs.distance.CompareTo(rhs.distance);
122  }
123  else if (lhs.module.sortOrderPriority != rhs.module.sortOrderPriority)
124  {
125  return lhs.module.sortOrderPriority.CompareTo(rhs.module.sortOrderPriority);
126  }
127  else if (lhs.module.renderOrderPriority != rhs.module.renderOrderPriority)
128  {
129  return lhs.module.renderOrderPriority.CompareTo(rhs.module.renderOrderPriority);
130  }
131  else if (lhs.depth != rhs.depth)
132  {
133  return rhs.depth.CompareTo(lhs.depth);
134  }
135  else if (Mathf.Abs(lhs.distance - rhs.distance) > 0.00001f)
136  {
137  return lhs.distance.CompareTo(rhs.distance);
138  }
139  else
140  {
141  return lhs.index.CompareTo(rhs.index);
142  }
143  }

201  private RaycastResult UpdateRay(IPointerDevice pointer, Vector3 lastPointerPosition, PointerEventData evtData)
202  {
203  var ray = evtData.pointerCurrentRaycast;
204  if (pointer.LockedOnTarget)
205  {
206  var from = ray.worldPosition - lastPointerPosition;
207  var to = ray.distance * (pointer.InteractionEndPoint - pointer.transform.position);
208  var delta = to - from;
209  var rot = Quaternion.FromToRotation(from.normalized, to.normalized);
210 
211  ray.gameObject = evtData.pointerPress;
212  ray.worldPosition = (rot * ray.worldPosition) + delta;
213  ray.worldNormal = rot * ray.worldNormal;
214  ray.screenPosition = evtData.position;
215  }
216  else
217  {
218  RaycastAll(pointer, evtData);
219  if (m_RaycastResultCache.Count > 0)
220  {
221  ray = m_RaycastResultCache[0];
222  evtData.hovered.Clear();
223  foreach (var r in m_RaycastResultCache)
224  {
225  evtData.hovered.Add(r.gameObject);
226  }
227  }
228  else
229  {
230  ray.Clear();
231  ray.worldPosition = pointer.InteractionEndPoint;
232  ray.distance = pointer.MinimumPointerDistance;
233  ray.worldNormal = -pointer.InteractionDirection;
234  ray.screenPosition = evtData.position;
235  }
236  }
