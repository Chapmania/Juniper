27  /* For single-pass compression, it's sufficient to buffer just one MCU
28  * (although this may prove a bit slow in practice). We allocate a
29  * workspace of C_MAX_BLOCKS_IN_MCU coefficient blocks, and reuse it for each
30  * MCU constructed and sent. (On 80x86, the workspace is FAR even though
31  * it's not really very big; this is to keep the module interfaces unchanged
32  * when a large coefficient buffer is necessary.)
33  * In multi-pass modes, this array points to the current MCU's blocks
34  * within the virtual arrays.
35  */
36  private readonly JBlock[][] m_MCU_buffer = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU][];
37 
38  /* In multi-pass modes, we need a virtual block array for each component. */
39  private readonly JVirtArray<JBlock>[] m_whole_image = new JVirtArray<JBlock>[JpegConstants.MAX_COMPONENTS];

41  public MyCCoefController(JpegCompressStruct cinfo, bool need_full_buffer)
42  {
43  m_cinfo = cinfo;
44 
45  /* Create the coefficient buffer. */
46  if (need_full_buffer)
47  {
48  /* Allocate a full-image virtual array for each component, */
49  /* padded to a multiple of samp_factor DCT blocks in each direction. */
50  for (var ci = 0; ci < cinfo.m_num_components; ci++)
51  {
52  m_whole_image[ci] = JpegCommonStruct.CreateBlocksArray(
53  JpegUtils.jround_up(cinfo.Component_info[ci].Width_in_blocks, cinfo.Component_info[ci].H_samp_factor),
54  JpegUtils.jround_up(cinfo.Component_info[ci].height_in_blocks, cinfo.Component_info[ci].V_samp_factor));
55  m_whole_image[ci].ErrorProcessor = cinfo;
56  }
57  }
58  else
59  {
60  /* We only need a single-MCU buffer. */
61  var buffer = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU];
62  for (var i = 0; i < JpegConstants.C_MAX_BLOCKS_IN_MCU; i++)
63  {
64  buffer[i] = new JBlock();
65  }
66 
67  for (var i = 0; i < JpegConstants.C_MAX_BLOCKS_IN_MCU; i++)
68  {
69  m_MCU_buffer[i] = new JBlock[JpegConstants.C_MAX_BLOCKS_IN_MCU - i];
70  for (var j = i; j < JpegConstants.C_MAX_BLOCKS_IN_MCU; j++)
71  {
72  m_MCU_buffer[i][j - i] = buffer[j];
73  }
74  }
75 
76  /* flag for no virtual arrays */
77  m_whole_image[0] = null;
78  }
79  }

125  /// <summary>
126  /// <para>
127  /// Process some data in the single-pass case.
128  /// We process the equivalent of one fully interleaved MCU row ("iMCU" row)
129  /// per call, ie, v_samp_factor block rows for each component in the image.
130  /// Returns true if the iMCU row is completed, false if suspended.
131  /// </para>
132  /// <para>
133  /// NB: input_buf contains a plane for each component in image,
134  /// which we index according to the component's SOF position.
135  /// </para>
136  /// </summary>
137  private bool CompressDataImpl(byte[][][] input_buf)
138  {
139  var last_MCU_col = m_cinfo.m_MCUs_per_row - 1;
140  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
141 
142  /* Loop to write as much as one whole iMCU row */
143  for (var yoffset = m_MCU_vert_offset; yoffset < m_MCU_rows_per_iMCU_row; yoffset++)
144  {
145  for (var MCU_col_num = m_mcu_ctr; MCU_col_num <= last_MCU_col; MCU_col_num++)
146  {
147  /* Determine where data comes from in input_buf and do the DCT thing.
148  * Each call on forward_DCT processes a horizontal row of DCT blocks
149  * as wide as an MCU; we rely on having allocated the MCU_buffer[] blocks
150  * sequentially. Dummy blocks at the right or bottom edge are filled in
151  * specially. The data in them does not matter for image reconstruction,
152  * so we fill them with values that will encode to the smallest amount of
153  * data, viz: all zeroes in the AC entries, DC entries equal to previous
154  * block's DC value. (Thanks to Thomas Kinsman for this idea.)
155  */
156  var blkn = 0;
157  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
158  {
159  var componentInfo = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[ci]];
160  var forward_DCT = m_cinfo.m_fdct.forward_DCT[componentInfo.Component_index];
161 
162  var blockcnt = (MCU_col_num < last_MCU_col) ? componentInfo.MCU_width : componentInfo.last_col_width;
163  var xpos = MCU_col_num * componentInfo.MCU_sample_width;
164  var ypos = yoffset * componentInfo.DCT_v_scaled_size;
165 
166  for (var yindex = 0; yindex < componentInfo.MCU_height; yindex++)
167  {
168  if (m_iMCU_row_num < last_iMCU_row || yoffset + yindex < componentInfo.last_row_height)
169  {
170  forward_DCT(componentInfo, input_buf[componentInfo.Component_index],
171  m_MCU_buffer[blkn], ypos, xpos, blockcnt);
172 
173  if (blockcnt < componentInfo.MCU_width)
174  {
175  /* Create some dummy blocks at the right edge of the image. */
176  for (var i = 0; i < (componentInfo.MCU_width - blockcnt); i++)
177  {
178  Array.Clear(m_MCU_buffer[blkn + blockcnt][i].data, 0, m_MCU_buffer[blkn + blockcnt][i].data.Length);
179  }
180 
181  for (var bi = blockcnt; bi < componentInfo.MCU_width; bi++)
182  {
183  m_MCU_buffer[blkn + bi][0][0] = m_MCU_buffer[blkn + bi - 1][0][0];
184  }
185  }
186  }
187  else
188  {
189  /* Create a row of dummy blocks at the bottom of the image. */
190  for (var i = 0; i < componentInfo.MCU_width; i++)
191  {
192  Array.Clear(m_MCU_buffer[blkn][i].data, 0, m_MCU_buffer[blkn][i].data.Length);
193  }
194 
195  for (var bi = 0; bi < componentInfo.MCU_width; bi++)
196  {
197  m_MCU_buffer[blkn + bi][0][0] = m_MCU_buffer[blkn - 1][0][0];
198  }
199  }
200 
201  blkn += componentInfo.MCU_width;
202  ypos += componentInfo.DCT_v_scaled_size;
203  }
204  }
205 
206  /* Try to write the MCU. In event of a suspension failure, we will
207  * re-DCT the MCU on restart (a bit inefficient, could be fixed...)
208  */
209  if (!m_cinfo.m_entropy.encodeMcu(m_MCU_buffer))
210  {
211  /* Suspension forced; update state counters and exit */
212  m_MCU_vert_offset = yoffset;
213  m_mcu_ctr = MCU_col_num;
214  return false;
215  }
216  }
217 
218  /* Completed an MCU row, but perhaps not an iMCU row */
219  m_mcu_ctr = 0;
220  }
221 
222  /* Completed the iMCU row, advance counters for next one */
223  m_iMCU_row_num++;
224  StartIMcuRow();
225  return true;
226  }

228  /// <summary>
229  /// <para>
230  /// Process some data in the first pass of a multi-pass case.
231  /// We process the equivalent of one fully interleaved MCU row ("iMCU" row)
232  /// per call, ie, v_samp_factor block rows for each component in the image.
233  /// This amount of data is read from the source buffer, DCT'd and quantized,
234  /// and saved into the virtual arrays. We also generate suitable dummy blocks
235  /// as needed at the right and lower edges. (The dummy blocks are constructed
236  /// in the virtual arrays, which have been padded appropriately.) This makes
237  /// it possible for subsequent passes not to worry about real vs. dummy blocks.
238  /// </para>
239  /// <para>
240  /// We must also emit the data to the entropy encoder. This is conveniently
241  /// done by calling compress_output() after we've loaded the current strip
242  /// of the virtual arrays.
243  /// </para>
244  /// <para>
245  /// NB: input_buf contains a plane for each component in image. All
246  /// components are DCT'd and loaded into the virtual arrays in this pass.
247  /// However, it may be that only a subset of the components are emitted to
248  /// the entropy encoder during this first pass; be careful about looking
249  /// at the scan-dependent variables (MCU dimensions, etc).
250  /// </para>
251  /// </summary>
252  private bool CompressFirstPass(byte[][][] input_buf)
253  {
254  var last_iMCU_row = m_cinfo.m_total_iMCU_rows - 1;
255 
256  for (var ci = 0; ci < m_cinfo.m_num_components; ci++)
257  {
258  var componentInfo = m_cinfo.Component_info[ci];
259 
260  /* Align the virtual buffer for this component. */
261  var buffer = m_whole_image[ci].Access(m_iMCU_row_num * componentInfo.V_samp_factor,
262  componentInfo.V_samp_factor);
263 
264  /* Count non-dummy DCT block rows in this iMCU row. */
265  int block_rows;
266  if (m_iMCU_row_num < last_iMCU_row)
267  {
268  block_rows = componentInfo.V_samp_factor;
269  }
270  else
271  {
272  /* NB: can't use last_row_height here, since may not be set! */
273  block_rows = componentInfo.height_in_blocks % componentInfo.V_samp_factor;
274  if (block_rows == 0)
275  {
276  block_rows = componentInfo.V_samp_factor;
277  }
278  }
279 
280  var blocks_across = componentInfo.Width_in_blocks;
281  var h_samp_factor = componentInfo.H_samp_factor;
282 
283  /* Count number of dummy blocks to be added at the right margin. */
284  var ndummy = blocks_across % h_samp_factor;
285  if (ndummy > 0)
286  {
287  ndummy = h_samp_factor - ndummy;
288  }
289 
290  var forward_DCT = m_cinfo.m_fdct.forward_DCT[ci];
291  /* Perform DCT for all non-dummy blocks in this iMCU row. Each call
292  * on forward_DCT processes a complete horizontal row of DCT blocks.
293  */
294  for (var block_row = 0; block_row < block_rows; block_row++)
295  {
296  forward_DCT(componentInfo, input_buf[ci],
297  buffer[block_row], block_row * componentInfo.DCT_v_scaled_size, 0, blocks_across);
298 
299  if (ndummy > 0)
300  {
301  /* Create dummy blocks at the right edge of the image. */
302  Array.Clear(buffer[block_row][blocks_across].data, 0, buffer[block_row][blocks_across].data.Length);
303 
304  var lastDC = buffer[block_row][blocks_across - 1][0];
305  for (var bi = 0; bi < ndummy; bi++)
306  {
307  buffer[block_row][blocks_across + bi][0] = lastDC;
308  }
309  }
310  }
311 
312  /* If at end of image, create dummy block rows as needed.
313  * The tricky part here is that within each MCU, we want the DC values
314  * of the dummy blocks to match the last real block's DC value.
315  * This squeezes a few more bytes out of the resulting file...
316  */
317  if (m_iMCU_row_num == last_iMCU_row)
318  {
319  blocks_across += ndummy; /* include lower right corner */
320  var MCUs_across = blocks_across / h_samp_factor;
321  for (var block_row = block_rows; block_row < componentInfo.V_samp_factor; block_row++)
322  {
323  for (var i = 0; i < blocks_across; i++)
324  {
325  Array.Clear(buffer[block_row][i].data, 0, buffer[block_row][i].data.Length);
326  }
327 
328  var thisOffset = 0;
329  var lastOffset = 0;
330  for (var MCUindex = 0; MCUindex < MCUs_across; MCUindex++)
331  {
332  var lastDC = buffer[block_row - 1][lastOffset + h_samp_factor - 1][0];
333  for (var bi = 0; bi < h_samp_factor; bi++)
334  {
335  buffer[block_row][thisOffset + bi][0] = lastDC;
336  }
337 
338  thisOffset += h_samp_factor; /* advance to next MCU in row */
339  lastOffset += h_samp_factor;
340  }
341  }
342  }
343  }
344 
345  /* NB: compress_output will increment iMCU_row_num if successful.
346  * A suspension return will result in redoing all the work above next time.
347  */
348 
349  /* Emit data to the entropy encoder, sharing code with subsequent passes */
350  return CompressOutput();
351  }

353  /// <summary>
354  /// Process some data in subsequent passes of a multi-pass case.
355  /// We process the equivalent of one fully interleaved MCU row ("iMCU" row)
356  /// per call, ie, v_samp_factor block rows for each component in the scan.
357  /// The data is obtained from the virtual arrays and fed to the entropy coder.
358  /// Returns true if the iMCU row is completed, false if suspended.
359  /// </summary>
360  private bool CompressOutput()
361  {
362  /* Align the virtual buffers for the components used in this scan.
363  */
364  var buffer = new JBlock[JpegConstants.MAX_COMPS_IN_SCAN][][];
365  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
366  {
367  var componentInfo = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[ci]];
368  buffer[ci] = m_whole_image[componentInfo.Component_index].Access(
369  m_iMCU_row_num * componentInfo.V_samp_factor, componentInfo.V_samp_factor);
370  }
371 
372  /* Loop to process one whole iMCU row */
373  for (var yoffset = m_MCU_vert_offset; yoffset < m_MCU_rows_per_iMCU_row; yoffset++)
374  {
375  for (var MCU_col_num = m_mcu_ctr; MCU_col_num < m_cinfo.m_MCUs_per_row; MCU_col_num++)
376  {
377  /* Construct list of pointers to DCT blocks belonging to this MCU */
378  var blkn = 0; /* index of current DCT block within MCU */
379  for (var ci = 0; ci < m_cinfo.m_comps_in_scan; ci++)
380  {
381  var componentInfo = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[ci]];
382  var start_col = MCU_col_num * componentInfo.MCU_width;
383  for (var yindex = 0; yindex < componentInfo.MCU_height; yindex++)
384  {
385  for (var xindex = 0; xindex < componentInfo.MCU_width; xindex++)
386  {
387  var bufLength = buffer[ci][yindex + yoffset].Length;
388  var start = start_col + xindex;
389  m_MCU_buffer[blkn] = new JBlock[bufLength - start];
390  for (var j = start; j < bufLength; j++)
391  {
392  m_MCU_buffer[blkn][j - start] = buffer[ci][yindex + yoffset][j];
393  }
394 
395  blkn++;
396  }
397  }
398  }
399 
400  /* Try to write the MCU. */
401  if (!m_cinfo.m_entropy.encodeMcu(m_MCU_buffer))
402  {
403  /* Suspension forced; update state counters and exit */
404  m_MCU_vert_offset = yoffset;
405  m_mcu_ctr = MCU_col_num;
406  return false;
407  }
408  }
409 
410  /* Completed an MCU row, but perhaps not an iMCU row */
411  m_mcu_ctr = 0;
412  }
413 
414  /* Completed the iMCU row, advance counters for next one */
415  m_iMCU_row_num++;
416  StartIMcuRow();
417  return true;
418  }

420  // Reset within-iMCU-row counters for a new row
421  private void StartIMcuRow()
422  {
423  /* In an interleaved scan, an MCU row is the same as an iMCU row.
424  * In a noninterleaved scan, an iMCU row has v_samp_factor MCU rows.
425  * But at the bottom of the image, process only what's left.
426  */
427  if (m_cinfo.m_comps_in_scan > 1)
428  {
429  m_MCU_rows_per_iMCU_row = 1;
430  }
431  else
432  {
433  if (m_iMCU_row_num < (m_cinfo.m_total_iMCU_rows - 1))
434  {
435  m_MCU_rows_per_iMCU_row = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[0]].V_samp_factor;
436  }
437  else
438  {
439  m_MCU_rows_per_iMCU_row = m_cinfo.Component_info[m_cinfo.m_cur_comp_info[0]].last_row_height;
440  }
441  }
442 
443  m_mcu_ctr = 0;
444  m_MCU_vert_offset = 0;
445  }
