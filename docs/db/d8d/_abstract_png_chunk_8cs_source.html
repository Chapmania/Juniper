174  /// <summary>
175  /// Creates one new blank chunk of the corresponding type, according to factoryMap (PngChunkUNKNOWN if not known)
176  /// </summary>
177  /// <param name="cid">Chunk Id</param>
178  /// <param name="info"></param>
179  /// <returns></returns>
180  internal static AbstractPngChunk FactoryFromId(string cid, ImageInfo info)
181  {
182  AbstractPngChunk chunk = null;
183  if (factoryMap is null)
184  {
185  _ = InitFactory();
186  }
187 
188  if (IsKnown(cid))
189  {
190  var t = factoryMap[cid];
191  if (t is null)
192  {
193  Console.Error.WriteLine("What?? " + cid);
194  }
195 
196  var cons = t.GetConstructor(new Type[] { typeof(ImageInfo) });
197  var o = cons.Invoke(new object[] { info });
198  chunk = (AbstractPngChunk)o;
199  }
200 
201  if (chunk is null)
202  {
203  chunk = new PngChunkUNKNOWN(cid, info);
204  }
205 
206  return chunk;
207  }
