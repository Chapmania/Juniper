151  internal static PngChunk Factory(ChunkRaw chunk, ImageInfo info) {
152  PngChunk c = FactoryFromId(Hjg.Pngcs.Chunks.ChunkHelper.ToString(chunk.IdBytes), info);
153  c.Length = chunk.Len;
154  c.ParseFromRaw(chunk);
155  return c;
156  }
157  /// <summary>
158  /// Creates one new blank chunk of the corresponding type, according to factoryMap (PngChunkUNKNOWN if not known)
159  /// </summary>
160  /// <param name="cid">Chunk Id</param>
161  /// <param name="info"></param>
162  /// <returns></returns>
163  internal static PngChunk FactoryFromId(String cid, ImageInfo info) {
164  PngChunk chunk = null;
165  if (factoryMap == null) initFactory();
166  if (isKnown(cid)) {
167  Type t = factoryMap[cid];
168  if (t == null) Console.Error.WriteLine("What?? " + cid);
169  System.Reflection.ConstructorInfo cons = t.GetConstructor(new Type[] { typeof(ImageInfo) });
170  object o = cons.Invoke(new object[] { info });
171  chunk = (PngChunk)o;
172  }
173  if (chunk == null)
174  chunk = new PngChunkUNKNOWN(cid, info);
175 
176  return chunk;
177  }
