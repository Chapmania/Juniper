231  private static int RaycastComparer(RaycastResult lhs, RaycastResult rhs)
232  {
233  if (lhs.module.eventCamera != null
234  && rhs.module.eventCamera != null)
235  {
236  return lhs.distance.CompareTo(rhs.distance);
237  }
238  else if (lhs.module.sortOrderPriority != rhs.module.sortOrderPriority)
239  {
240  return lhs.module.sortOrderPriority.CompareTo(rhs.module.sortOrderPriority);
241  }
242  else if (lhs.module.renderOrderPriority != rhs.module.renderOrderPriority)
243  {
244  return lhs.module.renderOrderPriority.CompareTo(rhs.module.renderOrderPriority);
245  }
246  else if (lhs.depth != rhs.depth)
247  {
248  return rhs.depth.CompareTo(lhs.depth);
249  }
250  else if (Mathf.Abs(lhs.distance - rhs.distance) > 0.00001f)
251  {
252  return lhs.distance.CompareTo(rhs.distance);
253  }
254  else
255  {
256  return lhs.index.CompareTo(rhs.index);
257  }

352  private RaycastResult UpdateRay(IPointerDevice pointer, PointerEventData evtData)
353  {
354  var ray = evtData.pointerCurrentRaycast;
355 
356  if (pointer.IsDragging)
357  {
358  var rot = Quaternion.FromToRotation(pointer.LastDirection, pointer.Direction);
359 
360  ray.gameObject = evtData.pointerPress;
361  ray.worldPosition = pointer.Origin + ray.distance * pointer.Direction;
362  ray.worldNormal = rot * ray.worldNormal;
363  ray.screenPosition = pointer.ScreenPoint;
364  }
365  else
366  {
367  RaycastAll(pointer, evtData);
368  if (m_RaycastResultCache.Count > 0)
369  {
370  ray = m_RaycastResultCache[0];
371  evtData.hovered.Clear();
372  foreach (var r in m_RaycastResultCache)
373  {
374  evtData.hovered.Add(r.gameObject);
375  }
376  }
377  else
378  {
379  ray.Clear();
380  ray.worldPosition = pointer.SmoothedWorldPoint;
381  ray.distance = pointer.MinimumPointerDistance;
382  ray.worldNormal = -pointer.Direction;
383  ray.screenPosition = pointer.ScreenPoint;
384  }
385  }

392  /// <summary>
393  /// Fire a raycast using all of the GraphicRaycasters in the system, plus the one
394  /// PhysicsRaycaster that is associated with the event Camera.
395  /// </summary>
396  /// <param name="pointer">Pointer.</param>
397  /// <param name="eventData">Event data.</param>
398  private void RaycastAll(IPointerDevice pointer, PointerEventData eventData)
399  {
400  m_RaycastResultCache.Clear();
401 
402  foreach (var canvas in ComponentExt.FindAll<Canvas>())
403  {
404  canvas.worldCamera = pointer.EventCamera;
405  }
406 
407  eventSystem.RaycastAll(eventData, m_RaycastResultCache);
408 
409  for (var i = 0; i < m_RaycastResultCache.Count; ++i)
410  {
411  var ray = m_RaycastResultCache[i];
412  if (ray.module is GraphicRaycaster)
413  {
414  var gfr = (GraphicRaycaster)ray.module;
415  var canv = gfr.GetComponent<Canvas>();
416  ray.worldNormal = -canv.transform.forward;
417 
418  var pos = (Vector3)ray.screenPosition;
419  if (canv.renderMode == RenderMode.WorldSpace)
420  {
421  pos.z = ray.distance + CanvasOffset;
422  }
423  ray.worldPosition = pointer.EventCamera.ScreenToWorldPoint(pos);
424 
425  m_RaycastResultCache[i] = ray;
426  }
427  }
428 
429  m_RaycastResultCache.Sort(RaycastComparer);
430  }
