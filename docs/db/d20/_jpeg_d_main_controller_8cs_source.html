12  /// <summary>
13  /// <para>Main buffer control (downsampled-data buffer)</para>
14  /// <para>
15  /// In the current system design, the main buffer need never be a full-image
16  /// buffer; any full-height buffers will be found inside the coefficient or
17  /// postprocessing controllers. Nonetheless, the main controller is not
18  /// trivial. Its responsibility is to provide context rows for upsampling/
19  /// rescaling, and doing this in an efficient fashion is a bit tricky.
20  /// </para>
21  /// <para>
22  /// Postprocessor input data is counted in "row groups". A row group
23  /// is defined to be (v_samp_factor * DCT_scaled_size / min_DCT_scaled_size)
24  /// sample rows of each component. (We require DCT_scaled_size values to be
25  /// chosen such that these numbers are integers. In practice DCT_scaled_size
26  /// values will likely be powers of two, so we actually have the stronger
27  /// condition that DCT_scaled_size / min_DCT_scaled_size is an integer.)
28  /// Upsampling will typically produce max_v_samp_factor pixel rows from each
29  /// row group (times any additional scale factor that the upsampler is
30  /// applying).
31  /// </para>
32  /// <para>
33  /// The coefficient controller will deliver data to us one iMCU row at a time;
34  /// each iMCU row contains v_samp_factor * DCT_scaled_size sample rows, or
35  /// exactly min_DCT_scaled_size row groups. (This amount of data corresponds
36  /// to one row of MCUs when the image is fully interleaved.) Note that the
37  /// number of sample rows varies across components, but the number of row
38  /// groups does not. Some garbage sample rows may be included in the last iMCU
39  /// row at the bottom of the image.
40  /// </para>
41  /// <para>
42  /// Depending on the vertical scaling algorithm used, the upsampler may need
43  /// access to the sample row(s) above and below its current input row group.
44  /// The upsampler is required to set need_context_rows true at global selection
45  /// time if so. When need_context_rows is false, this controller can simply
46  /// obtain one iMCU row at a time from the coefficient controller and dole it
47  /// out as row groups to the postprocessor.
48  /// </para>
49  /// <para>
50  /// When need_context_rows is true, this controller guarantees that the buffer
51  /// passed to postprocessing contains at least one row group's worth of samples
52  /// above and below the row group(s) being processed. Note that the context
53  /// rows "above" the first passed row group appear at negative row offsets in
54  /// the passed buffer. At the top and bottom of the image, the required
55  /// context rows are manufactured by duplicating the first or last real sample
56  /// row; this avoids having special cases in the upsampling inner loops.
57  /// </para>
58  /// <para>
59  /// The amount of context is fixed at one row group just because that's a
60  /// convenient number for this controller to work with. The existing
61  /// upsamplers really only need one sample row of context. An upsampler
62  /// supporting arbitrary output rescaling might wish for more than one row
63  /// group of context when shrinking the image; tough, we don't handle that.
64  /// (This is justified by the assumption that downsizing will be handled mostly
65  /// by adjusting the DCT_scaled_size values, so that the actual scale factor at
66  /// the upsample step needn't be much less than one.)
67  /// </para>
68  /// <para>
69  /// To provide the desired context, we have to retain the last two row groups
70  /// of one iMCU row while reading in the next iMCU row. (The last row group
71  /// can't be processed until we have another row group for its below-context,
72  /// and so we have to save the next-to-last group too for its above-context.)
73  /// We could do this most simply by copying data around in our buffer, but
74  /// that'd be very slow. We can avoid copying any data by creating a rather
75  /// strange pointer structure. Here's how it works. We allocate a workspace
76  /// consisting of M+2 row groups (where M = min_DCT_scaled_size is the number
77  /// of row groups per iMCU row). We create two sets of redundant pointers to
78  /// the workspace. Labeling the physical row groups 0 to M+1, the synthesized
79  /// pointer lists look like this:
80  /// M+1 M-1
81  /// master pointer --> 0 master pointer --> 0
82  /// 1 1
83  /// ... ...
84  /// M-3 M-3
85  /// M-2 M
86  /// M-1 M+1
87  /// M M-2
88  /// M+1 M-1
89  /// 0 0
90  /// We read alternate iMCU rows using each master pointer; thus the last two
91  /// row groups of the previous iMCU row remain un-overwritten in the workspace.
92  /// The pointer lists are set up so that the required context rows appear to
93  /// be adjacent to the proper places when we pass the pointer lists to the
94  /// upsampler.
95  /// </para>
96  /// <para>
97  /// The above pictures describe the normal state of the pointer lists.
98  /// At top and bottom of the image, we diddle the pointer lists to duplicate
99  /// the first or last sample row as necessary (this is cheaper than copying
100  /// sample rows around).
101  /// </para>
102  /// <para>
103  /// This scheme breaks down if M less than 2, ie, min_DCT_scaled_size is 1. In that
104  /// situation each iMCU row provides only one row group so the buffering logic
105  /// must be different (eg, we must read two iMCU rows before we can emit the
106  /// first row group). For now, we simply do not support providing context
107  /// rows when min_DCT_scaled_size is 1. That combination seems unlikely to
108  /// be worth providing --- if someone wants a 1/8th-size preview, they probably
109  /// want it quick and dirty, so a context-free upsampler is sufficient.
110  /// </para>
111  /// </summary>

145  public JpegDMainController(JpegDecompressStruct cinfo)
146  {
147  m_cinfo = cinfo;
148 
149  /* Allocate the workspace.
150  * ngroups is the number of row groups we need.
151  */
152  var ngroups = cinfo.min_DCT_v_scaled_size;
153  if (cinfo.m_upsample.NeedContextRows())
154  {
155  if (cinfo.min_DCT_v_scaled_size < 2) /* unsupported, see comments above */
156  {
157  cinfo.ErrExit(JMessageCode.JERR_NOTIMPL);
158  }
159 
160  AllocFunnyPointers(); /* Alloc space for xbuffer[] lists */
161  ngroups = cinfo.min_DCT_v_scaled_size + 2;
162  }
163 
164  for (var ci = 0; ci < cinfo.numComponents; ci++)
165  {
166  /* height of a row group of component */
167  var rgroup = (cinfo.CompInfo[ci].V_samp_factor * cinfo.CompInfo[ci].DCT_v_scaled_size) / cinfo.min_DCT_v_scaled_size;
168 
169  m_buffer[ci] = JpegCommonStruct.AllocJpegSamples(
170  cinfo.CompInfo[ci].Width_in_blocks * cinfo.CompInfo[ci].DCT_h_scaled_size,
171  rgroup * ngroups);
172  }

175  /// <summary>
176  /// Initialize for a processing pass.
177  /// </summary>
178  public void StartPass(JBufMode pass_mode)
179  {
180  switch (pass_mode)
181  {
182  case JBufMode.PassThrough:
183  if (m_cinfo.m_upsample.NeedContextRows())
184  {
185  m_dataProcessor = DataProcessor.context_main;
186  MakeFunnyPointers(); /* Create the xbuffer[] lists */
187  m_whichFunny = 0; /* Read first iMCU row into xbuffer[0] */
188  m_context_state = CTX_PREPARE_FOR_IMCU;
189  m_iMCU_row_ctr = 0;
190  }
191  else
192  {
193  /* Simple case with no context needed */
194  m_dataProcessor = DataProcessor.simple_main;
195  }
196  m_buffer_full = false; /* Mark buffer empty */
197  m_rowgroup_ctr = 0;
198  break;
199  case JBufMode.CrankDest:
200  /* For last pass of 2-pass quantization, just crank the postprocessor */
201  m_dataProcessor = DataProcessor.crank_post;
202  break;
203  default:
204  m_cinfo.ErrExit(JMessageCode.JERR_BAD_BUFFER_MODE);
205  break;
206  }
207  }

231  /// <summary>
232  /// Process some data.
233  /// This handles the simple case where no context is required.
234  /// </summary>
235  private void ProcessDataSimpleMain(byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
236  {
237  var cb = new ComponentBuffer[JpegConstants.MAX_COMPONENTS];
238  for (var i = 0; i < JpegConstants.MAX_COMPONENTS; i++)
239  {
240  cb[i] = new ComponentBuffer();
241  cb[i].SetBuffer(m_buffer[i], null, 0);
242  }
243 
244  /* Read input data if we haven't filled the main buffer yet */
245  if (!m_buffer_full)
246  {
247  if (m_cinfo.m_coef.DecompressData(cb) == ReadResult.JPEG_SUSPENDED)
248  {
249  /* suspension forced, can do nothing more */
250  return;
251  }
252 
253  /* OK, we have an iMCU row to work with */
254  m_buffer_full = true;
255  }
256 
257  /* There are always min_DCT_scaled_size row groups in an iMCU row. */
258  var rowgroups_avail = m_cinfo.min_DCT_v_scaled_size;
259 
260  /* Note: at the bottom of the image, we may pass extra garbage row groups
261  * to the postprocessor. The postprocessor has to check for bottom
262  * of image anyway (at row resolution), so no point in us doing it too.
263  */
264 
265  /* Feed the postprocessor */
266  m_cinfo.m_post.PostProcessData(cb, ref m_rowgroup_ctr, rowgroups_avail, output_buf, ref out_row_ctr, out_rows_avail);
267 
268  /* Has postprocessor consumed all the data yet? If so, mark buffer empty */
269  if (m_rowgroup_ctr >= rowgroups_avail)
270  {
271  m_buffer_full = false;
272  m_rowgroup_ctr = 0;
273  }
274  }

276  /// <summary>
277  /// Process some data.
278  /// This handles the case where context rows must be provided.
279  /// </summary>
280  private void ProcessDataContextMain(byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
281  {
282  var cb = new ComponentBuffer[m_cinfo.numComponents];
283  for (var i = 0; i < m_cinfo.numComponents; i++)
284  {
285  cb[i] = new ComponentBuffer();
286  cb[i].SetBuffer(m_buffer[i], m_funnyIndices[m_whichFunny][i], m_funnyOffsets[i]);
287  }
288 
289  /* Read input data if we haven't filled the main buffer yet */
290  if (!m_buffer_full)
291  {
292  if (m_cinfo.m_coef.DecompressData(cb) == ReadResult.JPEG_SUSPENDED)
293  {
294  /* suspension forced, can do nothing more */
295  return;
296  }
297 
298  /* OK, we have an iMCU row to work with */
299  m_buffer_full = true;
300 
301  /* count rows received */
302  m_iMCU_row_ctr++;
303  }
304 
305  /* Postprocessor typically will not swallow all the input data it is handed
306  * in one call (due to filling the output buffer first). Must be prepared
307  * to exit and restart.
308 
309 
310  This switch lets us keep track of how far we got.
311  * Note that each case falls through to the next on successful completion.
312  */
313  if (m_context_state == CTX_POSTPONED_ROW)
314  {
315  /* Call postprocessor using previously set pointers for postponed row */
316  m_cinfo.m_post.PostProcessData(cb, ref m_rowgroup_ctr,
317  m_rowgroups_avail, output_buf, ref out_row_ctr, out_rows_avail);
318 
319  if (m_rowgroup_ctr < m_rowgroups_avail)
320  {
321  /* Need to suspend */
322  return;
323  }
324 
325  m_context_state = CTX_PREPARE_FOR_IMCU;
326 
327  if (out_row_ctr >= out_rows_avail)
328  {
329  /* Postprocessor exactly filled output buf */
330  return;
331  }
332  }
333 
334  if (m_context_state == CTX_PREPARE_FOR_IMCU)
335  {
336  /* Prepare to process first M-1 row groups of this iMCU row */
337  m_rowgroup_ctr = 0;
338  m_rowgroups_avail = m_cinfo.min_DCT_v_scaled_size - 1;
339 
340  /* Check for bottom of image: if so, tweak pointers to "duplicate"
341  * the last sample row, and adjust rowgroups_avail to ignore padding rows.
342  */
343  if (m_iMCU_row_ctr == m_cinfo.m_total_iMCU_rows)
344  {
345  SetBottomPointers();
346  }
347 
348  m_context_state = CTX_PROCESS_IMCU;
349  }
350 
351  if (m_context_state == CTX_PROCESS_IMCU)
352  {
353  /* Call postprocessor using previously set pointers */
354  m_cinfo.m_post.PostProcessData(cb, ref m_rowgroup_ctr,
355  m_rowgroups_avail, output_buf, ref out_row_ctr, out_rows_avail);
356 
357  if (m_rowgroup_ctr < m_rowgroups_avail)
358  {
359  /* Need to suspend */
360  return;
361  }
362 
363  /* After the first iMCU, change wraparound pointers to normal state */
364  if (m_iMCU_row_ctr == 1)
365  {
366  SetWraparoundPointers();
367  }
368 
369  /* Prepare to load new iMCU row using other xbuffer list */
370  m_whichFunny ^= 1; /* 0=>1 or 1=>0 */
371  m_buffer_full = false;
372 
373  /* Still need to process last row group of this iMCU row, */
374  /* which is saved at index M+1 of the other xbuffer */
375  m_rowgroup_ctr = m_cinfo.min_DCT_v_scaled_size + 1;
376  m_rowgroups_avail = m_cinfo.min_DCT_v_scaled_size + 2;
377  m_context_state = CTX_POSTPONED_ROW;
378  }
379  }

396  private void AllocFunnyPointers()
397  {
398  var M = m_cinfo.min_DCT_v_scaled_size;
399  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
400  {
401  /* height of a row group of component */
402  var rgroup = (m_cinfo.CompInfo[ci].V_samp_factor * m_cinfo.CompInfo[ci].DCT_v_scaled_size) / m_cinfo.min_DCT_v_scaled_size;

412  /// <summary>
413  /// Create the funny pointer lists discussed in the comments above.
414  /// The actual workspace is already allocated (in main.buffer),
415  /// and the space for the pointer lists is allocated too.
416  /// This routine just fills in the curiously ordered lists.
417  /// This will be repeated at the beginning of each pass.
418  /// </summary>
419  private void MakeFunnyPointers()
420  {
421  var M = m_cinfo.min_DCT_v_scaled_size;
422  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
423  {
424  /* height of a row group of component */
425  var rgroup = (m_cinfo.CompInfo[ci].V_samp_factor * m_cinfo.CompInfo[ci].DCT_v_scaled_size) / m_cinfo.min_DCT_v_scaled_size;
426 
427  var ind0 = m_funnyIndices[0][ci];
428  var ind1 = m_funnyIndices[1][ci];
429 
430  /* First copy the workspace pointers as-is */
431  for (var i = 0; i < rgroup * (M + 2); i++)
432  {
433  ind0[i + rgroup] = i;
434  ind1[i + rgroup] = i;
435  }
436 
437  /* In the second list, put the last four row groups in swapped order */
438  for (var i = 0; i < rgroup * 2; i++)
439  {
440  ind1[(rgroup * (M - 1)) + i] = (rgroup * M) + i;
441  ind1[(rgroup * (M + 1)) + i] = (rgroup * (M - 2)) + i;
442  }
443 
444  /* The wraparound pointers at top and bottom will be filled later
445  * (see set_wraparound_pointers, below). Initially we want the "above"
446  * pointers to duplicate the first actual data line. This only needs
447  * to happen in xbuffer[0].
448  */
449  for (var i = 0; i < rgroup; i++)
450  {
451  ind0[i] = ind0[rgroup];
452  }
453  }
454  }
455 
456  /// <summary>
457  /// Set up the "wraparound" pointers at top and bottom of the pointer lists.
458  /// This changes the pointer list state from top-of-image to the normal state.
459  /// </summary>
460  private void SetWraparoundPointers()
461  {
462  var M = m_cinfo.min_DCT_v_scaled_size;
463  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
464  {
465  /* height of a row group of component */
466  var rgroup = (m_cinfo.CompInfo[ci].V_samp_factor * m_cinfo.CompInfo[ci].DCT_v_scaled_size) / m_cinfo.min_DCT_v_scaled_size;
467 
468  var ind0 = m_funnyIndices[0][ci];
469  var ind1 = m_funnyIndices[1][ci];
470 
471  for (var i = 0; i < rgroup; i++)
472  {
473  ind0[i] = ind0[(rgroup * (M + 2)) + i];
474  ind1[i] = ind1[(rgroup * (M + 2)) + i];
475 
476  ind0[(rgroup * (M + 3)) + i] = ind0[i + rgroup];
477  ind1[(rgroup * (M + 3)) + i] = ind1[i + rgroup];
478  }
479  }
480  }

482  /// <summary>
483  /// Change the pointer lists to duplicate the last sample row at the bottom
484  /// of the image. m_whichFunny indicates which m_funnyIndices holds the final iMCU row.
485  /// Also sets rowgroups_avail to indicate number of nondummy row groups in row.
486  /// </summary>
487  private void SetBottomPointers()
488  {
489  for (var ci = 0; ci < m_cinfo.numComponents; ci++)
490  {
491  /* Count sample rows in one iMCU row and in one row group */
492  var iMCUheight = m_cinfo.CompInfo[ci].V_samp_factor * m_cinfo.CompInfo[ci].DCT_v_scaled_size;
493  var rgroup = iMCUheight / m_cinfo.min_DCT_v_scaled_size;
494 
495  /* Count nondummy sample rows remaining for this component */
496  var rows_left = m_cinfo.CompInfo[ci].downsampled_height % iMCUheight;
497  if (rows_left == 0)
498  {
499  rows_left = iMCUheight;
500  }
501 
502  /* Count nondummy row groups. Should get same answer for each component,
503  * so we need only do it once.
504  */
505  if (ci == 0)
506  {
507  m_rowgroups_avail = ((rows_left - 1) / rgroup) + 1;
508  }
509 
510  /* Duplicate the last real sample row rgroup*2 times; this pads out the
511  * last partial rowgroup and ensures at least one full rowgroup of context.
512  */
513  for (var i = 0; i < rgroup * 2; i++)
514  {
515  m_funnyIndices[m_whichFunny][ci][rows_left + i + rgroup] = m_funnyIndices[m_whichFunny][ci][rows_left - 1 + rgroup];
516  }
517  }
518  }
