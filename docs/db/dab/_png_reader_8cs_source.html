10  /// <summary>
11  /// Reads a PNG image, line by line
12  /// </summary>
13  /// <remarks>
14  /// <para>The typical reading sequence is as follows:</para>
15  /// <para>1. At construction time, the header and IHDR chunk are read (basic image info)</para>
16  /// <para>2 (Optional) you can set some global options: UnpackedMode CrcCheckDisabled</para>
17  /// <para>3. (Optional) If you call GetMetadata() or or GetChunksLisk() before reading the pixels, the chunks before IDAT are automatically loaded and available</para>
18  /// <para>
19  /// 4a. The rows are read, one by one, with the <tt>ReadRowXXX</tt> methods: (ReadRowInt() , ReadRowByte(), etc)
20  /// in order, from 0 to nrows-1 (you can skip or repeat rows, but not go backwards)
21  /// </para>
22  /// <para>
23  /// 4b. Alternatively, you can read all rows, or a subset, in a single call: see ReadRowsInt(), ReadRowsByte()
24  /// In general this consumes more memory, but for interlaced images this is equally efficient, and more so if reading a small subset of rows.
25  /// </para>
26  /// <para>5. Read of the last row automatically loads the trailing chunks, and ends the reader.</para>
27  /// <para>6. End() forcibly finishes/aborts the reading and closes the stream</para>

479  /// <summary>
480  /// Reads chunkd from input stream, adds to ChunksList, and returns it.
481  /// If it's skipped, a PngChunkSkipped object is created
482  /// </summary>
483  /// <returns></returns>
484  private AbstractPngChunk ReadChunk(byte[] chunkid, int clen, bool skipforced)
485  {
486  if (clen < 0)
487  {
488  throw new PngjInputException("invalid chunk lenght: " + clen);
489  }
490  // skipChunksByIdSet is created lazyly, if fist IHDR has already been read
491  if (skipChunkIdsSet is null && CurrentChunkGroup > ChunksList.CHUNK_GROUP_0_IDHR)
492  {
493  skipChunkIdsSet = new Dictionary<string, int>();
494  if (SkipChunkIds is object)
495  {
496  foreach (var id in SkipChunkIds)
497  {
498  skipChunkIdsSet.Add(id, 1);
499  }
500  }
501  }
502 
503  var chunkidstr = ChunkHelper.ToString(chunkid);
504  var critical = ChunkHelper.IsCritical(chunkidstr);
505  var skip = skipforced;
506  if (MaxTotalBytesRead > 0 && clen + offset > MaxTotalBytesRead)
507  {
508  throw new PngjInputException("Maximum total bytes to read exceeeded: " + MaxTotalBytesRead + " offset:"
509  + offset + " clen=" + clen);
510  }
511  // an ancillary chunks can be skipped because of several reasons:
512  if (CurrentChunkGroup > ChunksList.CHUNK_GROUP_0_IDHR && !ChunkHelper.IsCritical(chunkidstr))
513  {
514  skip = skip || (SkipChunkMaxSize > 0 && clen >= SkipChunkMaxSize) || skipChunkIdsSet.ContainsKey(chunkidstr)
515  || (MaxBytesMetadata > 0 && clen > MaxBytesMetadata - bytesChunksLoaded)
516  || !ChunkHelper.ShouldLoad(chunkidstr, ChunkLoadBehaviour);
517  }
518 
519  AbstractPngChunk pngChunk;
520  if (skip)
521  {
522  PngHelperInternal.SkipBytes(inputStream, clen);
523  PngHelperInternal.ReadInt4(inputStream); // skip - we dont call PngHelperInternal.skipBytes(inputStream, clen + 4) for risk of overflow
524  pngChunk = new PngChunkSkipped(chunkidstr, ImgInfo, clen);
525  }
526  else
527  {
528  var chunk = new ChunkRaw(clen, chunkid, true);
529  _ = chunk.ReadChunkData(inputStream, critical);
530  pngChunk = AbstractPngChunk.Factory(chunk, ImgInfo);
531  if (!pngChunk.Crit)
532  {
533  bytesChunksLoaded += chunk.Len;
534  }
535  }
536 
537  pngChunk.Offset = offset - 8L;
538  chunksList.AppendReadChunk(pngChunk, CurrentChunkGroup);
539  offset += clen + 4L;
540  return pngChunk;
541  }

720  private void DecodeLastReadRowToInt(int[] buffer, int bytesRead)
721  { // see http://www.libpng.org/pub/png/spec/1.2/PNG-DataRep.html
722  if (ImgInfo.BitDepth <= 8)
723  {

875  private int ReadRowRaw(int nrow)
876  {
877  //
878  if (nrow == 0 && FirstChunksNotYetRead())
879  {
880  ReadFirstChunks();
881  }
882 
883  if (nrow == 0 && Interlaced)
884  {
885  Array.Clear(rowb, 0, rowb.Length); // new subimage: reset filters: this is enough, see the swap that happens lines
886  }
887  // below
888  var bytesRead = ImgInfo.BytesPerRow; // NOT including the filter byte
889  if (Interlaced)
890  {
891  if (nrow < 0 || nrow > deinterlacer.GetRows() || (nrow != 0 && nrow != deinterlacer.GetCurrRowSubimg() + 1))
892  {
893  throw new PngjInputException("invalid row in interlaced mode: " + nrow);
894  }
895 
896  deinterlacer.SetRow(nrow);
897  bytesRead = ((ImgInfo.BitspPixel * deinterlacer.GetPixelsToRead()) + 7) / 8;
898  if (bytesRead < 1)
899  {
900  throw new PngjInternalException("wtf??");
901  }
902  }
903  else
904  { // check for non interlaced
905  if (nrow < 0 || nrow >= ImgInfo.Rows || nrow != rowNum + 1)
906  {
907  throw new PngjInputException("invalid row: " + nrow);
908  }
909  }
910

912  // swap buffers
913  var tmp = rowb;
914  rowb = rowbprev;
915  rowbprev = tmp;

917  PngHelperInternal.ReadBytes(idatIstream, rowbfilter, 0, bytesRead + 1);
