11  /// <summary>
12  /// Reads a PNG image, line by line
13  /// </summary>
14  /// <remarks>
15  /// The typical reading sequence is as follows:
16  ///
17  /// 1. At construction time, the header and IHDR chunk are read (basic image info)
18  ///
19  /// 2 (Optional) you can set some global options: UnpackedMode CrcCheckDisabled
20  ///
21  /// 3. (Optional) If you call GetMetadata() or or GetChunksLisk() before reading the pixels, the chunks before IDAT are automatically loaded and available
22  ///
23  /// 4a. The rows are read, one by one, with the <tt>ReadRowXXX</tt> methods: (ReadRowInt() , ReadRowByte(), etc)
24  /// in order, from 0 to nrows-1 (you can skip or repeat rows, but not go backwards)
25  ///
26  /// 4b. Alternatively, you can read all rows, or a subset, in a single call: see ReadRowsInt(), ReadRowsByte()
27  /// In general this consumes more memory, but for interlaced images this is equally efficient, and more so if reading a small subset of rows.
28  ///
29  /// 5. Read of the last row automatically loads the trailing chunks, and ends the reader.
30  ///
31  /// 6. End() forcibly finishes/aborts the reading and closes the stream
32  ///
33  /// </remarks>

317  /// Reads chunks before first IDAT. Position before: after IDHR (crc included)
318  /// Position after: just after the first IDAT chunk id Returns length of first
319  /// IDAT chunk , -1 if not found
320  /// </summary>
321  ///
322  private void ReadFirstChunks() {
323  if (!FirstChunksNotYetRead())
324  return;
325  int clen = 0;
326  bool found = false;
327  byte[] chunkid = new byte[4]; // it's important to reallocate in each
328  this.CurrentChunkGroup = ChunksList.CHUNK_GROUP_1_AFTERIDHR;
329  while (!found) {
330  clen = PngHelperInternal.ReadInt4(inputStream);
331  offset += 4;
332  if (clen < 0)
333  break;
334  PngHelperInternal.ReadBytes(inputStream, chunkid, 0, 4);
335  offset += 4;
336  if (PngCsUtils.arraysEqual4(chunkid, Hjg.Pngcs.Chunks.ChunkHelper.b_IDAT)) {
337  found = true;
338  this.CurrentChunkGroup = ChunksList.CHUNK_GROUP_4_IDAT;
339  // add dummy idat chunk to list
340  chunksList.AppendReadChunk(new PngChunkIDAT(ImgInfo, clen, offset - 8), CurrentChunkGroup);
341  break;
342  } else if (PngCsUtils.arraysEqual4(chunkid, Hjg.Pngcs.Chunks.ChunkHelper.b_IEND)) {
343  throw new PngjInputException("END chunk found before image data (IDAT) at offset=" + offset);
344  }
345  String chunkids = ChunkHelper.ToString(chunkid);
346  if (chunkids.Equals(ChunkHelper.PLTE))
347  this.CurrentChunkGroup = ChunksList.CHUNK_GROUP_2_PLTE;
348  ReadChunk(chunkid, clen, false);
349  if (chunkids.Equals(ChunkHelper.PLTE))
350  this.CurrentChunkGroup = ChunksList.CHUNK_GROUP_3_AFTERPLTE;
351  }
352  int idatLen = found ? clen : -1;
353  if (idatLen < 0)
354  throw new PngjInputException("first idat chunk not found!");
355  iIdatCstream = new PngIDatChunkInputStream(inputStream, idatLen, offset);
356  idatIstream = ZlibStreamFactory.createZlibInputStream(iIdatCstream, true);
357  if (!crcEnabled)
358  iIdatCstream.DisableCrcCheck();
359  }

361  /// <summary>
362  /// Reads (and processes ... up to a point) chunks after last IDAT.
363  /// </summary>
364  ///
365  private void ReadLastChunks() {
366  CurrentChunkGroup = ChunksList.CHUNK_GROUP_5_AFTERIDAT;
367  // PngHelper.logdebug("idat ended? " + iIdatCstream.isEnded());
368  if (!iIdatCstream.IsEnded())
369  iIdatCstream.ForceChunkEnd();
370  int clen = iIdatCstream.GetLenLastChunk();
371  byte[] chunkid = iIdatCstream.GetIdLastChunk();
372  bool endfound = false;
373  bool first = true;
374  bool skip = false;
375  while (!endfound) {
376  skip = false;
377  if (!first) {
378  clen = PngHelperInternal.ReadInt4(inputStream);
379  offset += 4;
380  if (clen < 0)
381  throw new PngjInputException("bad len " + clen);
382  PngHelperInternal.ReadBytes(inputStream, chunkid, 0, 4);
383  offset += 4;
384  }
385  first = false;
386  if (PngCsUtils.arraysEqual4(chunkid, ChunkHelper.b_IDAT)) {
387  skip = true; // extra dummy (empty?) idat chunk, it can happen, ignore it
388  } else if (PngCsUtils.arraysEqual4(chunkid, ChunkHelper.b_IEND)) {
389  CurrentChunkGroup = ChunksList.CHUNK_GROUP_6_END;
390  endfound = true;
391  }
392  ReadChunk(chunkid, clen, skip);
393  }
394  if (!endfound)
395  throw new PngjInputException("end chunk not found - offset=" + offset);
396  // PngHelper.logdebug("end chunk found ok offset=" + offset);
397  }

399  /// <summary>
400  /// Reads chunkd from input stream, adds to ChunksList, and returns it.
401  /// If it's skipped, a PngChunkSkipped object is created
402  /// </summary>
403  /// <returns></returns>
404  private PngChunk ReadChunk(byte[] chunkid, int clen, bool skipforced) {
405  if (clen < 0) throw new PngjInputException("invalid chunk lenght: " + clen);
406  // skipChunksByIdSet is created lazyly, if fist IHDR has already been read
407  if (skipChunkIdsSet == null && CurrentChunkGroup > ChunksList.CHUNK_GROUP_0_IDHR) {
408  skipChunkIdsSet = new Dictionary<string, int>();
409  if (SkipChunkIds != null)
410  foreach (string id in SkipChunkIds) skipChunkIdsSet.Add(id, 1);
411  }
412 
413  String chunkidstr = ChunkHelper.ToString(chunkid);
414  PngChunk pngChunk = null;
415  bool critical = ChunkHelper.IsCritical(chunkidstr);
416  bool skip = skipforced;
417  if (MaxTotalBytesRead > 0 && clen + offset > MaxTotalBytesRead)
418  throw new PngjInputException("Maximum total bytes to read exceeeded: " + MaxTotalBytesRead + " offset:"
419  + offset + " clen=" + clen);
420  // an ancillary chunks can be skipped because of several reasons:
421  if (CurrentChunkGroup > ChunksList.CHUNK_GROUP_0_IDHR && !ChunkHelper.IsCritical(chunkidstr))
422  skip = skip || (SkipChunkMaxSize > 0 && clen >= SkipChunkMaxSize) || skipChunkIdsSet.ContainsKey(chunkidstr)
423  || (MaxBytesMetadata > 0 && clen > MaxBytesMetadata - bytesChunksLoaded)
424  || !ChunkHelper.ShouldLoad(chunkidstr, ChunkLoadBehaviour);
425 
426  if (skip) {
427  PngHelperInternal.SkipBytes(inputStream, clen);
428  PngHelperInternal.ReadInt4(inputStream); // skip - we dont call PngHelperInternal.skipBytes(inputStream, clen + 4) for risk of overflow
429  pngChunk = new PngChunkSkipped(chunkidstr, ImgInfo, clen);
430  } else {
431  ChunkRaw chunk = new ChunkRaw(clen, chunkid, true);
432  chunk.ReadChunkData(inputStream, crcEnabled || critical);
433  pngChunk = PngChunk.Factory(chunk, ImgInfo);
434  if (!pngChunk.Crit) {
435  bytesChunksLoaded += chunk.Len;
436  }
437 
438  }
439  pngChunk.Offset = offset - 8L;
440  chunksList.AppendReadChunk(pngChunk, CurrentChunkGroup);
441  offset += clen + 4L;
442  return pngChunk;
443  }

567  private void decodeLastReadRowToInt(int[] buffer, int bytesRead) { // see http://www.libpng.org/pub/png/spec/1.2/PNG-DataRep.html
568  if (ImgInfo.BitDepth <= 8) {
569  for (int i = 0, j = 1; i < bytesRead; i++)
570  buffer[i] = (rowb[j++]);
571  } else { // 16 bitspc
572  for (int i = 0, j = 1; j < bytesRead; i++)
573  buffer[i] = (rowb[j++] << 8) + rowb[j++];
574  }
575  if (ImgInfo.Packed && unpackedMode)
576  ImageLine.unpackInplaceInt(ImgInfo, buffer, buffer, false);
577  }

579  private void decodeLastReadRowToByte(byte[] buffer, int bytesRead) { // see http://www.libpng.org/pub/png/spec/1.2/PNG-DataRep.html
580  if (ImgInfo.BitDepth <= 8) {
581  Array.Copy(rowb, 1, buffer, 0, bytesRead);
582  } else { // 16 bitspc
583  for (int i = 0, j = 1; j < bytesRead; i++, j += 2)
584  buffer[i] = rowb[j]; // 16 bits in 1 byte: this discards the LSB!!!
585  }
586  if (ImgInfo.Packed && unpackedMode)
587  ImageLine.unpackInplaceByte(ImgInfo, buffer, buffer, false);

663  private int ReadRowRaw(int nrow) {
664  //
665  if (nrow == 0 && FirstChunksNotYetRead())
666  ReadFirstChunks();
667  if (nrow == 0 && interlaced)
668  Array.Clear(rowb, 0, rowb.Length); // new subimage: reset filters: this is enough, see the swap that happens lines
669  // below
670  int bytesRead = ImgInfo.BytesPerRow; // NOT including the filter byte
671  if (interlaced) {
672  if (nrow < 0 || nrow > deinterlacer.getRows() || (nrow != 0 && nrow != deinterlacer.getCurrRowSubimg() + 1))
673  throw new PngjInputException("invalid row in interlaced mode: " + nrow);
674  deinterlacer.setRow(nrow);
675  bytesRead = (ImgInfo.BitspPixel * deinterlacer.getPixelsToRead() + 7) / 8;
676  if (bytesRead < 1)
677  throw new PngjExceptionInternal("wtf??");
678  } else { // check for non interlaced
679  if (nrow < 0 || nrow >= ImgInfo.Rows || nrow != rowNum + 1)
680  throw new PngjInputException("invalid row: " + nrow);
681  }
682  rowNum = nrow;
683  // swap buffers
684  byte[] tmp = rowb;
685  rowb = rowbprev;
686  rowbprev = tmp;
687  // loads in rowbfilter "raw" bytes, with filter
688  PngHelperInternal.ReadBytes(idatIstream, rowbfilter, 0, bytesRead + 1);
689  offset = iIdatCstream.GetOffset();
690  if (offset < 0)
691  throw new PngjExceptionInternal("bad offset ??" + offset);
692  if (MaxTotalBytesRead > 0 && offset >= MaxTotalBytesRead)
693  throw new PngjInputException("Reading IDAT: Maximum total bytes to read exceeeded: " + MaxTotalBytesRead
694  + " offset:" + offset);
695  rowb[0] = 0;
696  UnfilterRow(bytesRead);
697  rowb[0] = rowbfilter[0];
698  if ((rowNum == ImgInfo.Rows - 1 && !interlaced) || (interlaced && deinterlacer.isAtLastRow()))
699  ReadLastAndClose();
700  return bytesRead;
701  }
