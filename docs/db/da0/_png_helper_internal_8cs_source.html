186  public static int FilterPaethPredictor(int a, int b, int c) {
187  // from http://www.libpng.org/pub/png/spec/1.2/PNG-Filters.html
188  // a = left, b = above, c = upper left
189  int p = a + b - c;// ; initial estimate
190  int pa = p >= a ? p - a : a - p;
191  int pb = p >= b ? p - b : b - p;
192  int pc = p >= c ? p - c : c - p;
193  // ; return nearest of a,b,c,
194  // ; breaking ties in order a,b,c.
195  if (pa <= pb && pa <= pc)
196  return a;
197  else if (pb <= pc)
198  return b;
199  else
200  return c;
201  }
