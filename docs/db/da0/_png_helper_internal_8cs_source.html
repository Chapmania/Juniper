319  public static int FilterPaethPredictor(int a, int b, int c)
320  {
321  // from http://www.libpng.org/pub/png/spec/1.2/PNG-Filters.html
322  // a = left, b = above, c = upper left
323  var p = a + b - c;// ; initial estimate
324  var pa = p >= a ? p - a : a - p;
325  var pb = p >= b ? p - b : b - p;
326  var pc = p >= c ? p - c : c - p;
327  // ; return nearest of a,b,c,
328  // ; breaking ties in order a,b,c.
329  if (pa <= pb && pa <= pc)
330  {
331  return a;
332  }
333  else if (pb <= pc)
334  {
335  return b;
336  }
337  else
338  {
339  return c;
