305  VuforiaRuntime.Instance.Deinit();
306  }
307  else
308  {
309  Debug.LogFormat("Initializing Vuforia ({0} of {1})", i, VUFORIA_INIT_ATTEMPTS);
310  }
311 
312  if (VuforiaRuntime.Instance.InitializationState == VuforiaRuntime.InitState.NOT_INITIALIZED)
313  {
314  VuforiaRuntime.Instance.InitVuforia();
315  VuforiaBehaviour.Instance.enabled = true;
316  }
317 
318  yield return WaitForNotInitializing;
319  }
320  }
321 
322  if (VuforiaRuntime.Instance.InitializationState == VuforiaRuntime.InitState.INITIALIZED)
323  {
324  yield return WaitForTracker;
325 
326  if (tracker.IsActive)
327  {
328  tracker.Stop();

330  }
331

333

335  {

339  }
