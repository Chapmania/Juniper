101  protected float RampIntensity(float rawIntensityEstimate)
102  {
103  const float linearRampThreshold = 0.8f;
104  const float middleGray = 0.18f;
105  const float inclination = 0.4f;
106  const float beta = (linearRampThreshold / inclination) - 1.0f;
107  const float alpha = (beta + 1.0f) / beta * linearRampThreshold;
108 
109  var normalizedIntensity = rawIntensityEstimate / middleGray;
110 
111  if (normalizedIntensity < 1.0f)
112  {
113  return normalizedIntensity * linearRampThreshold;
114  }
115  else
116  {
117  return alpha * (1.0f - (1.0f / ((beta * normalizedIntensity) + 1.0f)));
118  }
119  }
