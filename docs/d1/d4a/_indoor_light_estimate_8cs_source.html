123  const float linearRampThreshold = 0.8f;
124  const float middleGray = 0.18f;
125  const float inclination = 0.4f;
126  const float beta = (linearRampThreshold / inclination) - 1.0f;
127  const float alpha = (beta + 1.0f) / beta * linearRampThreshold;
128

132  {

134  }
135  else
