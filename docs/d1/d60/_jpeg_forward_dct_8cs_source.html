10  /// <summary>
11  /// <para>Forward DCT (also controls coefficient quantization)</para>
12  /// <para>
13  /// A forward DCT routine is given a pointer to an input sample array and
14  /// a pointer to a work area of type DCTELEM[]; the DCT is to be performed
15  /// in-place in that buffer. Type DCTELEM is int for 8-bit samples, INT32
16  /// for 12-bit samples. (NOTE: Floating-point DCT implementations use an
17  /// array of type FAST_FLOAT, instead.)
18  /// The input data is to be fetched from the sample array starting at a
19  /// specified column. (Any row offset needed will be applied to the array
20  /// pointer before it is passed to the FDCT code.)
21  /// Note that the number of samples fetched by the FDCT routine is
22  /// DCT_h_scaled_size * DCT_v_scaled_size.
23  /// The DCT outputs are returned scaled up by a factor of 8; they therefore
24  /// have a range of +-8K for 8-bit data, +-128K for 12-bit data. This
25  /// convention improves accuracy in integer implementations and saves some
26  /// work in floating-point ones.
27  /// </para>
28  /// <para>Each IDCT routine has its own ideas about the best dct_table element type.</para>
29  /// </summary>

380  var do_dct = this.do_dct[compptr.Component_index];
381  var divisors = m_dctTables[compptr.Component_index].int_array;
382  var workspace = new int[JpegConstants.DCTSIZE2]; /* work area for FDCT subroutine */
383  for (var bi = 0; bi < num_blocks; bi++, start_col += compptr.DCT_h_scaled_size)
384  {
385  /* Perform the DCT */
386  do_dct(workspace, sample_data, start_row, start_col);
387 
388  /* Quantize/descale the coefficients, and store into coef_blocks[] */
389  for (var i = 0; i < JpegConstants.DCTSIZE2; i++)
390  {
391  var qval = divisors[i];
392  var temp = workspace[i];
393 
394  if (temp < 0)
395  {
396  temp = -temp;
397  temp += qval >> 1; /* for rounding */
398 
399  if (temp >= qval)
400  {
401  temp /= qval;
402  }
403  else
404  {
405  temp = 0;
406  }
407 
408  temp = -temp;
409  }
410  else
411  {
412  temp += qval >> 1; /* for rounding */
413 
414  if (temp >= qval)
415  {
416  temp /= qval;
417  }
418  else
419  {
420  temp = 0;
421  }
422  }

458  /// <summary>
459  /// <para>
460  /// Perform the forward DCT on one block of samples.
461  /// NOTE: this code only copes with 8x8 DCTs.
462  /// </para>
463  /// <para>
464  /// A floating-point implementation of the
465  /// forward DCT (Discrete Cosine Transform).
466  /// </para>
467  /// <para>
468  /// This implementation should be more accurate than either of the integer
469  /// DCT implementations. However, it may not give the same results on all
470  /// machines because of differences in roundoff behavior. Speed will depend
471  /// on the hardware's floating point capacity.
472  /// </para>
473  /// <para>
474  /// A 2-D DCT can be done by 1-D DCT on each row followed by 1-D DCT
475  /// on each column. Direct algorithms are also available, but they are
476  /// much more complex and seem not to be any faster when reduced to code.
477  /// </para>
478  /// <para>
479  /// This implementation is based on Arai, Agui, and Nakajima's algorithm for
480  /// scaled DCT. Their original paper (Trans. IEICE E-71(11):1095) is in
481  /// Japanese, but the algorithm is described in the Pennebaker &amp; Mitchell
482  /// JPEG textbook (see REFERENCES section in file README). The following code
483  /// is based directly on figure 4-8 in P&amp;M.
484  /// While an 8-point DCT cannot be done in less than 11 multiplies, it is
485  /// possible to arrange the computation so that many of the multiplies are
486  /// simple scalings of the final outputs. These multiplies can then be
487  /// folded into the multiplications or divisions by the JPEG quantization
488  /// table entries. The AA&amp;N method leaves only 5 multiplies and 29 adds
489  /// to be done in the DCT itself.
490  /// The primary disadvantage of this method is that with a fixed-point
491  /// implementation, accuracy is lost due to imprecise representation of the
492  /// scaled quantization values. However, that problem does not arise if
493  /// we use floating point arithmetic.
494  /// </para>
495  /// </summary>
496  private static void JpegForwardDctFloat(float[] data, byte[][] sample_data, int start_row, int start_col)
497  {
498  /* Pass 1: process rows. */
499  var dataIndex = 0;
500  for (var ctr = 0; ctr < JpegConstants.DCTSIZE; ctr++)
501  {
502  var elem = sample_data[start_row + ctr];
503  var elemIndex = start_col;
504 
505  /* Load data into workspace */
506  float tmp0 = elem[elemIndex + 0] + elem[elemIndex + 7];
507  float tmp7 = elem[elemIndex + 0] - elem[elemIndex + 7];
508  float tmp1 = elem[elemIndex + 1] + elem[elemIndex + 6];
509  float tmp6 = elem[elemIndex + 1] - elem[elemIndex + 6];
510  float tmp2 = elem[elemIndex + 2] + elem[elemIndex + 5];

608  /// NOTE: this code only copes with 8x8 DCTs.
609  /// This file contains a fast, not so accurate integer implementation of the
610  /// forward DCT (Discrete Cosine Transform).
611  /// </para>
612  /// <para>
613  /// A 2-D DCT can be done by 1-D DCT on each row followed by 1-D DCT
614  /// on each column. Direct algorithms are also available, but they are
615  /// much more complex and seem not to be any faster when reduced to code.
616  /// </para>
617  /// <para>
618  /// This implementation is based on Arai, Agui, and Nakajima's algorithm for
619  /// scaled DCT. Their original paper (Trans. IEICE E-71(11):1095) is in
620  /// Japanese, but the algorithm is described in the Pennebaker &amp; Mitchell
621  /// JPEG textbook (see REFERENCES section in file README). The following code
622  /// is based directly on figure 4-8 in P&amp;M.
623  /// While an 8-point DCT cannot be done in less than 11 multiplies, it is
624  /// possible to arrange the computation so that many of the multiplies are
625  /// simple scalings of the final outputs. These multiplies can then be
626  /// folded into the multiplications or divisions by the JPEG quantization
627  /// table entries. The AA&amp;N method leaves only 5 multiplies and 29 adds
628  /// to be done in the DCT itself.
629  /// The primary disadvantage of this method is that with fixed-point math,
630  /// accuracy is lost due to imprecise representation of the scaled
631  /// quantization values. The smaller the quantization table entry, the less
632  /// precise the scaled value, so this implementation does worse with high-
633  /// quality-setting files than with low-quality ones.
634  /// </para>
635  /// <para>
636  /// Scaling decisions are generally the same as in the LL&amp;M algorithm;
637  /// see JpegForwardDctISlow for more details. However, we choose to descale
638  /// (right shift) multiplication products as soon as they are formed,
639  /// rather than carrying additional fractional bits into subsequent additions.
640  /// This compromises accuracy slightly, but it lets us save a few shifts.
641  /// More importantly, 16-bit arithmetic is then adequate (for 8-bit samples)
642  /// everywhere except in the multiplications proper; this saves a good deal
643  /// of work on 16-bit-int machines.
644  /// </para>
645  /// <para>
646  /// Again to save a few shifts, the intermediate results between pass 1 and
647  /// pass 2 are not upscaled, but are represented only to integral precision.
648  /// </para>
649  /// <para>
650  /// A final compromise is to represent the multiplicative constants to only
651  /// 8 fractional bits, rather than 13. This saves some shifting work on some
652  /// machines, and may also reduce the cost of multiplication (since there
653  /// are fewer one-bits in the constants).
654  /// </para>
655  /// </summary>

765  /// <summary>
766  /// <para>
767  /// Perform the forward DCT on one block of samples.
768  /// NOTE: this code only copes with 8x8 DCTs.
769  /// </para>
770  /// <para>
771  /// A slow-but-accurate integer implementation of the
772  /// forward DCT (Discrete Cosine Transform).
773  /// </para>
774  /// <para>
775  /// A 2-D DCT can be done by 1-D DCT on each row followed by 1-D DCT
776  /// on each column. Direct algorithms are also available, but they are
777  /// much more complex and seem not to be any faster when reduced to code.
778  /// </para>
779  /// <para>
780  /// This implementation is based on an algorithm described in
781  /// C. Loeffler, A. Ligtenberg and G. Moschytz, "Practical Fast 1-D DCT
782  /// Algorithms with 11 Multiplications", Proc. Int'l. Conf. on Acoustics,
783  /// Speech, and Signal Processing 1989 (ICASSP '89), pp. 988-991.
784  /// The primary algorithm described there uses 11 multiplies and 29 adds.
785  /// We use their alternate method with 12 multiplies and 32 adds.
786  /// The advantage of this method is that no data path contains more than one
787  /// multiplication; this allows a very simple and accurate implementation in
788  /// scaled fixed-point arithmetic, with a minimal number of shifts.
789  /// </para>
790  /// <para>The poop on this scaling stuff is as follows:</para>
791  /// <para>
792  /// Each 1-D DCT step produces outputs which are a factor of sqrt(N)
793  /// larger than the true DCT outputs. The final outputs are therefore
794  /// a factor of N larger than desired; since N=8 this can be cured by
795  /// a simple right shift at the end of the algorithm. The advantage of
796  /// this arrangement is that we save two multiplications per 1-D DCT,
797  /// because the y0 and y4 outputs need not be divided by sqrt(N).
798  /// In the IJG code, this factor of 8 is removed by the quantization
799  /// step, NOT here.
800  /// </para>
801  /// <para>
802  /// We have to do addition and subtraction of the integer inputs, which
803  /// is no problem, and multiplication by fractional constants, which is
804  /// a problem to do in integer arithmetic. We multiply all the constants
805  /// by CONST_SCALE and convert them to integer constants (thus retaining
806  /// SLOW_INTEGER_CONST_BITS bits of precision in the constants). After doing a
807  /// multiplication we have to divide the product by CONST_SCALE, with proper
808  /// rounding, to produce the correct output. This division can be done
809  /// cheaply as a right shift of SLOW_INTEGER_CONST_BITS bits. We postpone shifting
810  /// as long as possible so that partial sums can be added together with
811  /// full fractional precision.
812  /// </para>
813  /// <para>
814  /// The outputs of the first pass are scaled up by SLOW_INTEGER_PASS1_BITS bits so that
815  /// they are represented to better-than-integral precision. These outputs
816  /// require BITS_IN_JSAMPLE + SLOW_INTEGER_PASS1_BITS + 3 bits; this fits in a 16-bit word
817  /// with the recommended scaling. (For 12-bit sample data, the intermediate
818  /// array is int anyway.)
819  /// </para>
820  /// <para>
821  /// To avoid overflow of the 32-bit intermediate results in pass 2, we must
822  /// have BITS_IN_JSAMPLE + SLOW_INTEGER_CONST_BITS + SLOW_INTEGER_PASS1_BITS &lt;= 26. Error analysis
823  /// shows that the values given below are the most effective.
824  /// </para>
825  /// </summary>
826  private static void JpegForwardDctISlow(int[] data, byte[][] sample_data, int start_row, int start_col)
827  {
828  /* Pass 1: process rows. */
829  /* Note results are scaled up by sqrt(8) compared to a true DCT; */
830  /* furthermore, we scale the results by 2**SLOW_INTEGER_PASS1_BITS. */
831  var dataIndex = 0;
832  for (var ctr = 0; ctr < JpegConstants.DCTSIZE; ctr++)
833  {
834  var elem = sample_data[start_row + ctr];
835  var elemIndex = start_col;
836 
837  var tmp0 = elem[elemIndex + 0] + elem[elemIndex + 7];
838  var tmp1 = elem[elemIndex + 1] + elem[elemIndex + 6];
839  var tmp2 = elem[elemIndex + 2] + elem[elemIndex + 5];
840  var tmp3 = elem[elemIndex + 3] + elem[elemIndex + 4];
841 
842  /* Even part per LL&M figure 1 --- note that published figure is faulty;
843  * rotator "sqrt(2)*c1" should be "sqrt(2)*c6".
844  */
845 
846  var tmp10 = tmp0 + tmp3;
847  var tmp12 = tmp0 - tmp3;
848  var tmp11 = tmp1 + tmp2;
849  var tmp13 = tmp1 - tmp2;
850 
851  tmp0 = elem[elemIndex + 0] - elem[elemIndex + 7];
852  tmp1 = elem[elemIndex + 1] - elem[elemIndex + 6];
853  tmp2 = elem[elemIndex + 2] - elem[elemIndex + 5];
854  tmp3 = elem[elemIndex + 3] - elem[elemIndex + 4];
855 
856  data[dataIndex + 0] = (tmp10 + tmp11 - (8 * JpegConstants.CENTERJSAMPLE)) << SLOW_INTEGER_PASS1_BITS;
857  data[dataIndex + 4] = (tmp10 - tmp11) << SLOW_INTEGER_PASS1_BITS;
858 
859  var z1 = (tmp12 + tmp13) * SLOW_INTEGER_FIX_0_541196100;
860  /* Add fudge factor here for final descale. */
861  z1 += 1 << (SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS - 1);
862 
863  data[dataIndex + 2] = JpegUtils.RIGHT_SHIFT(z1 + (tmp12 * SLOW_INTEGER_FIX_0_765366865), /* c2-c6 */
864  SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
865  data[dataIndex + 6] = JpegUtils.DESCALE(z1 - (tmp13 * SLOW_INTEGER_FIX_1_847759065),
866  SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
867 
868  /* Odd part per figure 8 --- note paper omits factor of sqrt(2).
869  * cK represents cos(K*pi/16).
870  * i0..i3 in the paper are tmp4..tmp7 here.
871  */
872 
873  tmp12 = tmp0 + tmp2;
874  tmp13 = tmp1 + tmp3;
875 
876  z1 = (tmp12 + tmp13) * SLOW_INTEGER_FIX_1_175875602; /* c3 */
877  /* Add fudge factor here for final descale. */
878  z1 += 1 << (SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS - 1);
879 
880  tmp12 *= -SLOW_INTEGER_FIX_0_390180644; /* -c3+c5 */
881  tmp13 *= -SLOW_INTEGER_FIX_1_961570560; /* -c3-c5 */
882  tmp12 += z1;
883  tmp13 += z1;
884 
885  z1 = (tmp0 + tmp3) * -SLOW_INTEGER_FIX_0_899976223; /* -c3+c7 */
886  tmp0 *= SLOW_INTEGER_FIX_1_501321110; /* c1+c3-c5-c7 */
887  tmp3 *= SLOW_INTEGER_FIX_0_298631336; /* -c1+c3+c5-c7 */
888  tmp0 += z1 + tmp12;
889  tmp3 += z1 + tmp13;
890 
891  z1 = (tmp1 + tmp2) * -SLOW_INTEGER_FIX_2_562915447; /* -c1-c3 */
892  tmp1 *= SLOW_INTEGER_FIX_3_072711026; /* c1+c3+c5-c7 */
893  tmp2 *= SLOW_INTEGER_FIX_2_053119869; /* c1+c3-c5+c7 */
894  tmp1 += z1 + tmp13;
895  tmp2 += z1 + tmp12;
896 
897  data[dataIndex + 1] = JpegUtils.RIGHT_SHIFT(tmp0, SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
898  data[dataIndex + 3] = JpegUtils.RIGHT_SHIFT(tmp1, SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
899  data[dataIndex + 5] = JpegUtils.RIGHT_SHIFT(tmp2, SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
900  data[dataIndex + 7] = JpegUtils.RIGHT_SHIFT(tmp3, SLOW_INTEGER_CONST_BITS - SLOW_INTEGER_PASS1_BITS);
901 
902  dataIndex += JpegConstants.DCTSIZE; /* advance pointer to next row */
903  }
904 
905  /* Pass 2: process columns.
906  * We remove the SLOW_INTEGER_PASS1_BITS scaling, but leave the results scaled up
907  * by an overall factor of 8.
908  * cK represents sqrt(2) * cos(K*pi/16).
909  */
910 
911  dataIndex = 0;
912  for (var ctr = JpegConstants.DCTSIZE - 1; ctr >= 0; ctr--)
913  {
914  /* Even part per LL&M figure 1 --- note that published figure is faulty;
915  * rotator "sqrt(2)*c1" should be "sqrt(2)*c6".
916  */

918  var tmp1 = data[dataIndex + (JpegConstants.DCTSIZE * 1)] + data[dataIndex + (JpegConstants.DCTSIZE * 6)];
919  var tmp2 = data[dataIndex + (JpegConstants.DCTSIZE * 2)] + data[dataIndex + (JpegConstants.DCTSIZE * 5)];
920  var tmp3 = data[dataIndex + (JpegConstants.DCTSIZE * 3)] + data[dataIndex + (JpegConstants.DCTSIZE * 4)];
921 
922  /* Add fudge factor here for final descale. */
923  var tmp10 = tmp0 + tmp3 + (1 << (SLOW_INTEGER_PASS1_BITS - 1));
924  var tmp12 = tmp0 - tmp3;
925  var tmp11 = tmp1 + tmp2;
926  var tmp13 = tmp1 - tmp2;
927 
928  tmp0 = data[dataIndex + (JpegConstants.DCTSIZE * 0)] - data[dataIndex + (JpegConstants.DCTSIZE * 7)];
929  tmp1 = data[dataIndex + (JpegConstants.DCTSIZE * 1)] - data[dataIndex + (JpegConstants.DCTSIZE * 6)];
930  tmp2 = data[dataIndex + (JpegConstants.DCTSIZE * 2)] - data[dataIndex + (JpegConstants.DCTSIZE * 5)];
931  tmp3 = data[dataIndex + (JpegConstants.DCTSIZE * 3)] - data[dataIndex + (JpegConstants.DCTSIZE * 4)];
932

934  data[dataIndex + (JpegConstants.DCTSIZE * 4)] = JpegUtils.RIGHT_SHIFT(tmp10 - tmp11, SLOW_INTEGER_PASS1_BITS);
935 
936  var z1 = (tmp12 + tmp13) * SLOW_INTEGER_FIX_0_541196100; /* c6 */
937  /* Add fudge factor here for final descale. */
938  z1 += 1 << (SLOW_INTEGER_CONST_BITS + SLOW_INTEGER_PASS1_BITS - 1);
939 
940  data[dataIndex + (JpegConstants.DCTSIZE * 2)] = JpegUtils.RIGHT_SHIFT(
941  z1 + (tmp12 * SLOW_INTEGER_FIX_0_765366865),
942  SLOW_INTEGER_CONST_BITS + SLOW_INTEGER_PASS1_BITS);
943  data[dataIndex + (JpegConstants.DCTSIZE * 6)] = JpegUtils.RIGHT_SHIFT(
944  z1 - (tmp13 * SLOW_INTEGER_FIX_1_847759065),
945  SLOW_INTEGER_CONST_BITS + SLOW_INTEGER_PASS1_BITS);
946 
947  /* Odd part per figure 8 --- note paper omits factor of sqrt(2).
948  * i0..i3 in the paper are tmp4..tmp7 here.
949  */
950 
951  tmp12 = tmp0 + tmp2;
952  tmp13 = tmp1 + tmp3;
953 
954  z1 = (tmp12 + tmp13) * SLOW_INTEGER_FIX_1_175875602; /* c3 */
955  /* Add fudge factor here for final descale. */
956  z1 += 1 << (SLOW_INTEGER_CONST_BITS + SLOW_INTEGER_PASS1_BITS - 1);
