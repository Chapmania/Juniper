175  /// <summary>

178  /// </summary>

180  private IEnumerator StartGPSTrackingCoroutine()
181  {
182  if (Status == LocationServiceStatus.Stopped)
183  {
184  Location.Start(desiredAccuracy, updateDistance);
185  }
186 
187  yield return new WaitUntil(() =>
188  Status == LocationServiceStatus.Running
189  || Status == LocationServiceStatus.Failed);
190 
191  if (Status == LocationServiceStatus.Failed)
192  {
193  Debug.LogError("Could not start the GPS service");
194  enabled = false;
195  }
196  }

198  /// <summary>
199  /// If the time between reports has elapsed, retrieve a new GPS report, saving the results to
200  /// PlayerPrefs and firing the <see cref="onPositionUpdated"/> and <see
201  /// cref="PositionUpdated"/> events if one is found.
202  /// </summary>
203  public void Update()
204  {
205  if (Status == LocationServiceStatus.Running
206  && !UseFakeCoord
207  && Time.time >= nextUpdateTime)
208  {
209  nextUpdateTime += timeBetweenUpdates;
210  var newLocation = Location.lastData;
211  if (newLocation.timestamp > lastLocation.timestamp)
212  {
213  Coord = new LatLngPoint(newLocation.latitude, newLocation.longitude, newLocation.altitude);
214  PlayerPrefs.SetString(COORD_KEY, JsonConvert.SerializeObject(Coord));
215  lastLocation = newLocation;
216  OnPositionUpdated();
217  }
218  }

222  PrintDebugReport();
223  }
