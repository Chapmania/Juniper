359  {

361  // initialized to 0 the first time
362  if (filterStrat.ShouldTestAll(rown))
363  {
364  FilterRowNone();
365  ReportResultsForFilter(rown, FilterType.FILTER_NONE, true);
366  FilterRowSub();
367  ReportResultsForFilter(rown, FilterType.FILTER_SUB, true);
368  FilterRowUp();
369  ReportResultsForFilter(rown, FilterType.FILTER_UP, true);
370  FilterRowAverage();

372  FilterRowPaeth();
373  ReportResultsForFilter(rown, FilterType.FILTER_PAETH, true);
374  }

377  rowbfilter[0] = (byte)(int)filterType;
378  switch (filterType)
379  {
380  case Hjg.Pngcs.FilterType.FILTER_NONE:
381  FilterRowNone();
382  break;
383 
384  case Hjg.Pngcs.FilterType.FILTER_SUB:
385  FilterRowSub();
386  break;
387 
388  case Hjg.Pngcs.FilterType.FILTER_UP:
389  FilterRowUp();
390  break;
391 
392  case Hjg.Pngcs.FilterType.FILTER_AVERAGE:
393  FilterRowAverage();

484  /// <summary>
485  /// copy chunks from reader - copy_mask : see ChunksToWrite.COPY_XXX
486  /// If we are after idat, only considers those chunks after IDAT in PngReader
487  /// TODO: this should be more customizable
488  /// </summary>
489  ///
490  private void CopyChunks(PngReader reader, int copy_mask, bool onlyAfterIdat)
491  {
492  if (reader is null)
493  {
494  throw new ArgumentNullException(nameof(reader));
495  }
496 
497  var idatDone = CurrentChunkGroup >= ChunksList.CHUNK_GROUP_4_IDAT;
498  if (onlyAfterIdat && reader.CurrentChunkGroup < ChunksList.CHUNK_GROUP_6_END)
499  {
500  throw new PngjException("tried to copy last chunks but reader has not ended");
501  }
502 
503  foreach (var chunk in reader.GetChunksList().Chunks)
504  {
505  var group = chunk.ChunkGroup;
506  if (group < ChunksList.CHUNK_GROUP_4_IDAT && idatDone)
507  {
508  continue;
509  }
510 
511  var copy = false;
512  if (chunk.Crit)
513  {
514  if (chunk.Id.Equals(ChunkHelper.PLTE, StringComparison.Ordinal))
515  {
516  if (ImgInfo.Indexed && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_PALETTE))
517  {
518  copy = true;
519  }
520 
521  if (!ImgInfo.Greyscale && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL))
522  {
523  copy = true;
524  }
525  }
526  }
527  else
528  { // ancillary
529  var text = (chunk is AbstractPngChunkTextVar);
530  var safe = chunk.Safe;
531  // notice that these if are not exclusive
532  if (ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL))
533  {
534  copy = true;
535  }
536 
537  if (safe && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL_SAFE))
538  {
539  copy = true;
540  }
541 
542  if (chunk.Id.Equals(ChunkHelper.tRNS, StringComparison.Ordinal)
543  && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_TRANSPARENCY))
544  {
545  copy = true;
546  }
547 
548  if (chunk.Id.Equals(ChunkHelper.pHYs, StringComparison.Ordinal)
549  && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_PHYS))
550  {
551  copy = true;
552  }
553 
554  if (text
555  && ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_TEXTUAL))
556  {
557  copy = true;
558  }
559 
560  if (((ChunkHelper.MaskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALMOSTALL)
561  && !ChunkHelper.IsUnknown(chunk))
562  || text
563  || chunk.Id.Equals(ChunkHelper.hIST, StringComparison.Ordinal)
564  || chunk.Id.Equals(ChunkHelper.tIME, StringComparison.Ordinal)))
565  {
566  copy = true;
567  }
568 
569  if (chunk is PngChunkSkipped)
570  {
571  copy = false;
572  }
573  }
574 
575  if (copy)
576  {
577  _ = chunksList.Queue(AbstractPngChunk.CloneChunk(chunk, ImgInfo));
578  }
579  }
580  }
