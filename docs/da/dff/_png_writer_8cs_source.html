395  /// <summary>
396  /// copy chunks from reader - copy_mask : see ChunksToWrite.COPY_XXX
397  /// If we are after idat, only considers those chunks after IDAT in PngReader
398  /// TODO: this should be more customizable
399  /// </summary>
400  ///
401  private void CopyChunks(PngReader reader, int copy_mask, bool onlyAfterIdat) {
402  bool idatDone = CurrentChunkGroup >= ChunksList.CHUNK_GROUP_4_IDAT;
403  if (onlyAfterIdat && reader.CurrentChunkGroup < ChunksList.CHUNK_GROUP_6_END) throw new PngjException("tried to copy last chunks but reader has not ended");
404  foreach (PngChunk chunk in reader.GetChunksList().GetChunks()) {
405  int group = chunk.ChunkGroup;
406  if (group < ChunksList.CHUNK_GROUP_4_IDAT && idatDone)
407  continue;
408  bool copy = false;
409  if (chunk.Crit) {
410  if (chunk.Id.Equals(ChunkHelper.PLTE)) {
411  if (ImgInfo.Indexed && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_PALETTE))
412  copy = true;
413  if (!ImgInfo.Greyscale && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL))
414  copy = true;
415  }
416  } else { // ancillary
417  bool text = (chunk is PngChunkTextVar);
418  bool safe = chunk.Safe;
419  // notice that these if are not exclusive
420  if (ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL))
421  copy = true;
422  if (safe && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALL_SAFE))
423  copy = true;
424  if (chunk.Id.Equals(ChunkHelper.tRNS)
425  && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_TRANSPARENCY))
426  copy = true;
427  if (chunk.Id.Equals(ChunkHelper.pHYs) && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_PHYS))
428  copy = true;
429  if (text && ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_TEXTUAL))
430  copy = true;
431  if (ChunkHelper.maskMatch(copy_mask, ChunkCopyBehaviour.COPY_ALMOSTALL)
432  && !(ChunkHelper.IsUnknown(chunk) || text || chunk.Id.Equals(ChunkHelper.hIST) || chunk.Id
433  .Equals(ChunkHelper.tIME)))
434  copy = true;
435  if (chunk is PngChunkSkipped)
436  copy = false;
437  }
438  if (copy) {
439  chunksList.Queue(PngChunk.CloneChunk(chunk, ImgInfo));
440  }
441  }
442  }
