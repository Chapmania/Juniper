292  /// <summary>
293  /// Updates the pointer graphics if the option to show them has changed, and checks to see
294  /// how close we are to the target, updating the location of the arrow graphic if we haven't
295  /// found it yet, or activating the callback function if we have.
296  /// </summary>
297  public void Update()
298  {
299  if (showImage != wasShowImage)
300  {
301  image.SetActive(showImage);
302  }
303 
304  if (showPointer != wasShowPointer)
305  {
306  pointer.SetActive(showPointer);
307  }
308 
309  var origin = transform.position;
310  var fwd = transform.forward;
311 
312  if (!targetTrans.IsActivated())
313  {
314  targetTrans = null;
315  }
316 
317  if (Target != Vector3.zero)
318  {
319  var dir = (Target - origin).normalized;
320  var right = Vector3.Dot(dir, transform.right) * transform.right;
321  var up = Vector3.Dot(dir, transform.up) * transform.up;
322  var point = up + right;
323  pointer.rotation = Quaternion.LookRotation(point, transform.forward);
324 
325  var toTarget = DisplayManager.MainCamera.WorldToViewportPoint(Target);
326  toTarget.z = 0;
327  var targetDistance = toTarget.magnitude / 2;
328  arrow.localPosition = Mathf.Clamp01(targetDistance) * tagRange;
329 
330  var distance = 1 - Vector3.Dot(fwd, dir);
331  if (distance < minDistance && onTargetFound != null)
332  {
333  // The callback could call AttentionDirector::Target, so if we call onTargetFound
334  // before nulling it, we'll break the callback.
335  var cb = onTargetFound;
336  onTargetFound = null;
337  cb();
338  }
339  }
340 
341 #if UNITY_EDITOR
342  if (UnityEngine.Input.GetKeyDown(KeyCode.F)
343  && Find.Any(out RunningMovement mouse))
344  {
345  mouse.transform.LookAt(Target, Vector3.up);
346  }
347 #endif
348
