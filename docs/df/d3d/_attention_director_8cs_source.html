201  /// <summary>
202  /// Updates the pointer graphics if the option to show them has changed, and checks to see
203  /// how close we are to the target, updating the location of the arrow graphic if we haven't
204  /// found it yet, or activating the callback function if we have.
205  /// </summary>
206  public void Update()
207  {
208  if (showImage != wasShowImage)
209  {
210  image.SetActive(showImage);
211  }
212 
213  if (showPointer != wasShowPointer)
214  {
215  pointer.SetActive(showPointer);
216  }
217 
218  var origin = transform.position;
219  var fwd = transform.forward;
220 
221  if (!targetTrans.IsActivated())
222  {
223  targetTrans = null;
224  }
225 
226  if (Target != Vector3.zero)
227  {
228  var dir = (Target - origin).normalized;
229  var right = Vector3.Dot(dir, transform.right) * transform.right;
230  var up = Vector3.Dot(dir, transform.up) * transform.up;
231  var point = up + right;
232  pointer.rotation = Quaternion.LookRotation(point, transform.forward);
233 
234  var toTarget = DisplayManager.MainCamera.WorldToViewportPoint(Target);
235  toTarget.z = 0;
236  var targetDistance = toTarget.magnitude / 2;
237  arrow.localPosition = Mathf.Clamp01(targetDistance) * tagRange;
238 
239  var distance = 1 - Vector3.Dot(fwd, dir);
240  if (distance < minDistance && onTargetFound != null)
241  {
242  // The callback could call AttentionDirector::Target, so if we call onTargetFound
243  // before nulling it, we'll break the callback.
244  var cb = onTargetFound;
245  onTargetFound = null;
246  cb();
247  }
248  }
