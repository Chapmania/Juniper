202  /// <summary>
203  /// Updates the pointer graphics if the option to show them has changed, and checks to see

205  /// found it yet, or activating the callback function if we have.

208  {

219  var origin = transform.position;
220  var fwd = transform.forward;
221 
222  if (!targetTrans.IsActivated())

224  targetTrans = null;
225  }
226 
227  if (Target != Vector3.zero)
228  {
229  var dir = (Target - origin).normalized;
230  var right = Vector3.Dot(dir, transform.right) * transform.right;
231  var up = Vector3.Dot(dir, transform.up) * transform.up;
232  var point = up + right;

234 
235  var toTarget = DisplayManager.MainCamera.WorldToViewportPoint(Target);
236  toTarget.z = 0;

238  arrow.localPosition = Mathf.Clamp01(targetDistance) * tagRange;
239 
240  var distance = 1 - Vector3.Dot(fwd, dir);

242  {

244  // before nulling it, we'll break the callback.
245  var cb = onTargetFound;
