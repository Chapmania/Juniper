288  /// <summary>
289  /// If we aren't waiting on a response to a previous request, wait until the request time is
290  /// nigh, then issue the request. If we are waiting on a response to a previous request, then
291  /// check how long it has taken, and issue the TimeOut event if it has taken too long.
292  /// </summary>
293  protected override void Loop()
294  {
295  if (UntilNextRequestSeconds <= 0)
296  {
297  var now = DateTime.Now;
298  LastRequest = now;
299  NextRequest = now.AddSeconds(PollRate);
300  LastResponse = null;
301  OnReady(OnSuccess, OnServerError);
302  }
303 
304  if (IsTimedOut && !wasTimedOut)
305  {
306  OnTimedOut();
307  }
308  wasTimedOut = IsTimedOut;
309  }
