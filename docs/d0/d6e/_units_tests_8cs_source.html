413  private static string SigFig(float value, int sigfigs)
414  {
415  var q = (int)Mathf.Log10(value) + 1;
416  sigfigs -= q;
417  var p = Mathf.Pow(10, sigfigs);
418  var v = (Mathf.Round(value * p) / p).ToString();
419  if (sigfigs > 0)
420  {
421  var i = v.IndexOf(".", System.StringComparison.Ordinal);
422  if (i == -1)
423  {
424  v += ".";
425  i = v.Length - 1;
426  }
427 
428  var z = sigfigs + i + 1;
429  while (v.Length < z)
430  {
431  v += "0";
432  }
433  }
434  return v;
435  }
