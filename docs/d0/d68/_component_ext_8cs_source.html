131  /// forward-slash delimited names. Use ".." to select the parent transform.
132  /// </summary>

134  /// <param name="parent"></param>
135  /// <param name="path"> </param>
136  /// <returns></returns>
137  public static Transform Query(this Component parent, string path)
138  {
139  var parts = path.Split('/');
140  var top = new List<Transform>(5) { parent.transform };
141  for (var i = 0; i < parts.Length; ++i)
142  {
143  var part = parts[i];
144  if (part != ".")
145  {
146  var next = new List<Transform>(5);
147 
148  foreach (var here in top)
149  {
150  if (here == null)
151  {
152  next.Add((from scene in MasterSceneController.AllScenes
153  from gameObject in scene.GetRootGameObjects()
154  where gameObject.name == part
155  select gameObject.transform)
156  .FirstOrDefault());
157  }
158  else if (part == "*")
159  {
160  next.AddRange(here.Children());
161  }
162  else if (part == "..")
163  {
164  next.Add(here.parent);
165  }
166  else if (part.Length > 0)
167  {
168  var child = here.Find(part);
169  if (child != null)
170  {
171  next.Add(child);
172  }
173  }
174  else
175  {
176  next.Add(null);
177  }
178  }
179 
180  if (next.Count == 0 && i < parts.Length - 1)
181  {
182  return null;
183  }
184  else

186  top = next;
187  }
188  }
189  }
