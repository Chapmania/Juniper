34  /// <summary>
35  /// Per-pass setup.
36  /// This is called at the beginning of each output pass. We determine which
37  /// modules will be active during this pass and give them appropriate
38  /// start_pass calls. We also set is_dummy_pass to indicate whether this
39  /// is a "real" output pass or a dummy pass for color quantization.
40  /// (In the latter case, we will crank the pass to completion.)
41  /// </summary>
42  public void PrepareForOutputPass()
43  {
44  if (m_is_dummy_pass)
45  {
46  /* Final pass of 2-pass quantization */
47  m_is_dummy_pass = false;
48  m_cinfo.m_cquantize.StartPass(false);
49  m_cinfo.m_post.StartPass(JBufMode.CrankDest);
50  m_cinfo.m_main.StartPass(JBufMode.CrankDest);
51  }
52  else
53  {
54  if (m_cinfo.quantizeColors && m_cinfo.m_colormap is null)
55  {
56  /* Select new quantization method */
57  if (m_cinfo.twoPassQuantize && m_cinfo.enable2PassQuant)
58  {
59  m_cinfo.m_cquantize = m_quantizer_2pass;
60  m_is_dummy_pass = true;
61  }
62  else if (m_cinfo.enable1PassQuant)
63  {
64  m_cinfo.m_cquantize = m_quantizer_1pass;
65  }
66  else
67  {
68  m_cinfo.ErrExit(JMessageCode.JERR_MODE_CHANGE);
69  }
70  }
71 
72  m_cinfo.m_idct.StartPass();
73  m_cinfo.m_coef.StartOutputPass();
74 
75  if (!m_cinfo.rawDataOut)
76  {
77  m_cinfo.m_upsample.StartPass();
78 
79  if (m_cinfo.quantizeColors)
80  {
81  m_cinfo.m_cquantize.StartPass(m_is_dummy_pass);
82  }
83 
84  m_cinfo.m_post.StartPass(m_is_dummy_pass ? JBufMode.SaveAndPass : JBufMode.PassThrough);
85  m_cinfo.m_main.StartPass(JBufMode.PassThrough);
86  }
87  }
88 
89  /* Set up progress monitor's pass info if present */
90  if (m_cinfo.prog is object)
91  {
92  m_cinfo.prog.CompletedPasses = m_pass_number;
93  m_cinfo.prog.TotalPasses = m_pass_number + (m_is_dummy_pass ? 2 : 1);
94 
95  /* In buffered-image mode, we assume one more output pass if EOI not
96  * yet reached, but no more passes if EOI has been reached.
97  */
98  if (m_cinfo.bufferedImage && !m_cinfo.m_inputctl.EOIReached())
99  {
100  m_cinfo.prog.TotalPasses += (m_cinfo.enable2PassQuant ? 2 : 1);
101  }
102  }
103  }

123  /// <summary>
124  /// <para>
125  /// Master selection of decompression modules.
126  /// This is done once at jpeg_start_decompress time. We determine
127  /// which modules will be used and give them appropriate initialization calls.
128  /// We also initialize the decompressor input side to begin consuming data.
129  /// </para>
130  /// <para>
131  /// Since jpeg_read_header has finished, we know what is in the SOF
132  /// and (first) SOS markers. We also have all the application parameter
133  /// settings.
134  /// </para>
135  /// </summary>
136  private void MasterSelection()
137  {
138  /* For now, precision must match compiled-in value... */
139  if (m_cinfo.m_dataPrecision != JpegConstants.BITS_IN_JSAMPLE)
140  {
141  m_cinfo.ErrExit(JMessageCode.JERR_BAD_PRECISION, m_cinfo.m_dataPrecision);
142  }
143 
144  /* Initialize dimensions and other stuff */
145  m_cinfo.JpegCalcOutputDimensions();
146  PrepareRangeLimitTable();
147 
148  /* Sanity check on image dimensions */
149  if (m_cinfo.outputHeight <= 0 || m_cinfo.outputWidth <= 0 ||
150  m_cinfo.outColorComponents <= 0)
151  {
152  m_cinfo.ErrExit(JMessageCode.JERR_EMPTY_IMAGE);
153  }
154 
155  /* Width of an output scanline must be representable as int. */
156  long samplesperrow = m_cinfo.outputWidth * m_cinfo.outColorComponents;
157  var jd_samplesperrow = (int)samplesperrow;
158  if (jd_samplesperrow != samplesperrow)
159  {
160  m_cinfo.ErrExit(JMessageCode.JERR_WIDTH_OVERFLOW);
161  }
162 
163  /* Initialize my private state */
164  m_pass_number = 0;
165  m_using_merged_upsample = m_cinfo.UseMergedUpSample();
166 
167  /* Color quantizer selection */
168  m_quantizer_1pass = null;
169  m_quantizer_2pass = null;
170 
171  /* No mode changes if not using buffered-image mode. */
172  if (!m_cinfo.quantizeColors || !m_cinfo.bufferedImage)
173  {
174  m_cinfo.enable1PassQuant = false;
175  m_cinfo.enableExternalQuant = false;
176  m_cinfo.enable2PassQuant = false;
177  }
178 
179  if (m_cinfo.quantizeColors)
180  {
181  if (m_cinfo.rawDataOut)
182  {
183  m_cinfo.ErrExit(JMessageCode.JERR_NOTIMPL);
184  }
185 
186  /* 2-pass quantizer only works in 3-component color space. */
187  if (m_cinfo.outColorComponents != 3)
188  {
189  m_cinfo.enable1PassQuant = true;
190  m_cinfo.enableExternalQuant = false;
191  m_cinfo.enable2PassQuant = false;
192  m_cinfo.m_colormap = null;
193  }
194  else if (m_cinfo.m_colormap is object)
195  {
196  m_cinfo.enableExternalQuant = true;
197  }
198  else if (m_cinfo.twoPassQuantize)
199  {
200  m_cinfo.enable2PassQuant = true;
201  }
202  else
203  {
204  m_cinfo.enable1PassQuant = true;
205  }
206 
207  if (m_cinfo.enable1PassQuant)
208  {
209  m_cinfo.m_cquantize = new My1PassCQuantizer(m_cinfo);
210  m_quantizer_1pass = m_cinfo.m_cquantize;
211  }
212 
213  /* We use the 2-pass code to map to external colormaps. */
214  if (m_cinfo.enable2PassQuant || m_cinfo.enableExternalQuant)
215  {
216  m_cinfo.m_cquantize = new My2PassCQuantizer(m_cinfo);
217  m_quantizer_2pass = m_cinfo.m_cquantize;
218  }
219  /* If both quantizers are initialized, the 2-pass one is left active;
220  * this is necessary for starting with quantization to an external map.
221  */
222  }
223 
224  /* Post-processing: in particular, color conversion first */
225  if (!m_cinfo.rawDataOut)
226  {
227  if (m_using_merged_upsample)
228  {
229  /* does color conversion too */
230  m_cinfo.m_upsample = new MyMergedUpSampler(m_cinfo);
231  }
232  else
233  {
234  m_cinfo.m_cconvert = new JpegColorDeconverter(m_cinfo);
235  m_cinfo.m_upsample = new MyUpSampler(m_cinfo);
236  }
237 
238  m_cinfo.m_post = new JpegDPostController(m_cinfo, m_cinfo.enable2PassQuant);
239  }
240 
241  /* Inverse DCT */
242  m_cinfo.m_idct = new JpegInverseDct(m_cinfo);
243 
244  if (m_cinfo.arithCode)
245  {
246  m_cinfo.m_entropy = new ArithEntropyDecoder(m_cinfo);
247  }
248  else
249  {
250  m_cinfo.m_entropy = new HuffmanEntropyDecoder(m_cinfo);
251  }
252 
253  /* Initialize principal buffer controllers. */
254  var use_c_buffer = m_cinfo.m_inputctl.HasMultipleScans() || m_cinfo.bufferedImage;
255  m_cinfo.m_coef = new JpegDCoefController(m_cinfo, use_c_buffer);
256 
257  if (!m_cinfo.rawDataOut)
258  {
259  m_cinfo.m_main = new JpegDMainController(m_cinfo);
260  }
261 
262  /* Initialize input side of decompressor to consume first scan. */
263  m_cinfo.m_inputctl.StartInputPass();
264 
265  /* If jpeg_start_decompress will read the whole file, initialize
266  * progress monitoring appropriately. The input step is counted
267  * as one pass.
268  */
269  if (m_cinfo.prog is object && !m_cinfo.bufferedImage && m_cinfo.m_inputctl.HasMultipleScans())
270  {
271  /* Estimate number of scans to set pass_limit. */
272  int nscans;
273  if (m_cinfo.progressiveMode)
274  {
275  /* Arbitrarily estimate 2 interleaved DC scans + 3 AC scans/component. */
276  nscans = 2 + (3 * m_cinfo.numComponents);
277  }
278  else
279  {
280  /* For a non progressive multiscan file, estimate 1 scan per component. */
281  nscans = m_cinfo.numComponents;
282  }
283 
284  m_cinfo.prog.PassCounter = 0;
285  m_cinfo.prog.PassLimit = m_cinfo.m_total_iMCU_rows * nscans;
286  m_cinfo.prog.CompletedPasses = 0;
287  m_cinfo.prog.TotalPasses = (m_cinfo.enable2PassQuant ? 3 : 2);
288 
289  /* Count the input pass as done */
290  m_pass_number++;
291  }
292  }

294  /// <summary>
295  /// <para>Allocate and fill in the sample_range_limit table.</para>
296  /// <para>
297  /// Several decompression processes need to range-limit values to the range
298  /// 0..MAXJSAMPLE; the input value may fall somewhat outside this range
299  /// due to noise introduced by quantization, roundoff error, etc. These
300  /// processes are inner loops and need to be as fast as possible. On most
301  /// machines, particularly CPUs with pipelines or instruction prefetch,
302  /// a (subscript-check-less) C table lookup
303  /// x = sample_range_limit[x];
304  /// is faster than explicit tests
305  /// <c>
306  /// if (x &amp; 0)
307  /// x = 0;
308  /// else if (x > MAXJSAMPLE)
309  /// x = MAXJSAMPLE;
310  /// </c>
311  /// These processes all use a common table prepared by the routine below.
312  /// </para>
313  /// <para>
314  /// For most steps we can mathematically guarantee that the initial value
315  /// of x is within 2*(MAXJSAMPLE+1) of the legal range, so a table running
316  /// from -2*(MAXJSAMPLE+1) to 3*MAXJSAMPLE+2 is sufficient.But for the
317  /// initial limiting step(just after the IDCT), a wildly out-of-range value
318  /// is possible if the input data is corrupt.To avoid any chance of indexing
319  /// off the end of memory and getting a bad-pointer trap, we perform the
320  /// post-IDCT limiting thus:
321  /// <c>x = (sample_range_limit - SUBSET)[(x + CENTER) &amp; MASK];</c>
322  /// where MASK is 2 bits wider than legal sample data, ie 10 bits for 8-bit
323  /// samples. Under normal circumstances this is more than enough range and
324  /// a correct output will be generated; with bogus input data the mask will
325  /// cause wraparound, and we will safely generate a bogus-but-in-range output.
326  /// For the post-IDCT step, we want to convert the data from signed to unsigned
327  /// representation by adding CENTERJSAMPLE at the same time that we limit it.
328  /// This is accomplished with SUBSET = CENTER - CENTERJSAMPLE.
329  /// </para>
330  /// <para>
331  /// Note that the table is allocated in near data space on PCs; it's small
332  /// enough and used often enough to justify this.
333  /// </para>
334  /// </summary>
335  private void PrepareRangeLimitTable()
336  {
337  var table = new byte[5 * (JpegConstants.MAXJSAMPLE + 1)];
338  /* First segment of range limit table: limit[x] = 0 for x < 0 */
339 
340  /* allow negative subscripts of simple table */
341  const int tableOffset = 2 * (JpegConstants.MAXJSAMPLE + 1);
342  m_cinfo.m_sample_range_limit = table;
343  m_cinfo.m_sampleRangeLimitOffset = tableOffset;
344 
345  /* Main part of range limit table: limit[x] = x */
346  int i;
347  for (i = 0; i <= JpegConstants.MAXJSAMPLE; i++)
348  {
349  table[tableOffset + i] = (byte)i;
350  }
351 
352  /* End of range limit table: limit[x] = MAXJSAMPLE for x > MAXJSAMPLE */
353  for (; i < 3 * (JpegConstants.MAXJSAMPLE + 1); i++)
354  {
355  table[tableOffset + i] = JpegConstants.MAXJSAMPLE;
356  }
357  }
358  }
