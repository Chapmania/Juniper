103  /// <param name="rawIntensityEstimate">Raw intensity estimate.</param>
104  protected float RampIntensity(float rawIntensityEstimate)
105  {
106  const float linearRampThreshold = 0.8f;
107  const float middleGray = 0.18f;
108  const float inclination = 0.4f;
109  const float beta = (linearRampThreshold / inclination) - 1.0f;
110  const float alpha = (beta + 1.0f) / beta * linearRampThreshold;
111 
112  var normalizedIntensity = rawIntensityEstimate / middleGray;
113 
114  if (normalizedIntensity < 1.0f)
115  {
116  return normalizedIntensity * linearRampThreshold;
117  }
118  else
119  {
120  return alpha * (1.0f - (1.0f / ((beta * normalizedIntensity) + 1.0f)));
121  }
122  }
