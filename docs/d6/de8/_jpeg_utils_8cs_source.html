114  /* We assume that right shift corresponds to signed division by 2 with
115  * rounding towards minus infinity. This is correct for typical "arithmetic
116  * shift" instructions that shift in copies of the sign bit.
117  * RIGHT_SHIFT provides a proper signed right shift of an int quantity.
118  * It is only applied with constant shift counts. SHIFT_TEMPS must be
119  * included in the variables of any routine using RIGHT_SHIFT.
120  */
121  public static int RIGHT_SHIFT(int x, int shft)
122  {
123  return x >> shft;
124  }
125 
126  /* Descale and correctly round an int value that's scaled by N bits.
127  * We assume RIGHT_SHIFT rounds towards minus infinity, so adding
128  * the fudge factor is correct for either sign of X.
129  */
130  public static int DESCALE(int x, int n)
131  {
132  return RIGHT_SHIFT(x + (1 << (n - 1)), n);
133  }
