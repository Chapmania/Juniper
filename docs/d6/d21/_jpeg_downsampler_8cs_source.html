1 /*
2  * This file contains downsampling routines.
3  *
4  * Downsampling input data is counted in "row groups". A row group
5  * is defined to be max_v_samp_factor pixel rows of each component,
6  * from which the downsampler produces v_samp_factor sample rows.
7  * A single row group is processed in each call to the downsampler module.
8  *
9  * The downsampler is responsible for edge-expansion of its output data
10  * to fill an integral number of DCT blocks horizontally. The source buffer
11  * may be modified if it is helpful for this purpose (the source buffer is
12  * allocated wide enough to correspond to the desired output width).
13  * The caller (the prep controller) is responsible for vertical padding.
14  *
15  * The downsampler may request "context rows" by setting need_context_rows
16  * during startup. In this case, the input arrays will contain at least
17  * one row group's worth of pixels above and below the passed-in data;
18  * the caller will create dummy rows at image top and bottom by replicating
19  * the first or last real pixel row.
20  *
21  * An excellent reference for image resampling is
22  * Digital Image Warping, George Wolberg, 1990.
23  * Pub. by IEEE Computer Society Press, Los Alamitos, CA. ISBN 0-8186-8944-7.
24  *
25  * The downsampling algorithm used here is a simple average of the source
26  * pixels covered by the output pixel. The hi-falutin sampling literature
27  * refers to this as a "box filter". In general the characteristics of a box
28  * filter are not very good, but for the specific cases we normally use (1:1
29  * and 2:1 ratios) the box is equivalent to a "triangle filter" which is not
30  * nearly so bad. If you intend to use other sampling ratios, you'd be well
31  * advised to improve this code.
32  *
33  * A simple input-smoothing capability is provided. This is mainly intended
34  * for cleaning up color-dithered GIF input files (if you find it inadequate,
35  * we suggest using an external filtering program such as pnmconvol). When
36  * enabled, each input pixel P is replaced by a weighted sum of itself and its
37  * eight neighbors. P's weight is 1-8*SF and each neighbor's weight is SF,
38  * where SF = (smoothing_factor / 1024).
39  * Currently, smoothing is only supported for 2h2v sampling factors.
40  */

74  public JpegDownsampler(JpegCompressStruct cinfo)
75  {
76  m_cinfo = cinfo;
77  m_need_context_rows = false;
78 
79  if (cinfo.CIR601sampling)
80  {
81  cinfo.ErrExit(JMessageCode.JERR_CCIR601_NOTIMPL);
82  }
83 
84  /* Verify we can handle the sampling factors, and set up method pointers */
85  var smoothok = true;
86  for (var ci = 0; ci < cinfo.m_num_components; ci++)
87  {
88  var componentInfo = cinfo.Component_info[ci];
89 
90  /* Compute size of an "output group" for DCT scaling. This many samples
91  * are to be converted from max_h_samp_factor * max_v_samp_factor pixels.
92  */
93  var h_out_group = (componentInfo.H_samp_factor * componentInfo.DCT_h_scaled_size) /
94  m_cinfo.min_DCT_h_scaled_size;
95  var v_out_group = (componentInfo.V_samp_factor * componentInfo.DCT_v_scaled_size) /
96  m_cinfo.min_DCT_v_scaled_size;
97  var h_in_group = m_cinfo.m_max_h_samp_factor;
98  var v_in_group = m_cinfo.m_max_v_samp_factor;
99  rowgroup_height[ci] = v_out_group; /* save for use later */
100  if (h_in_group == h_out_group && v_in_group == v_out_group)
101  {
102  if (cinfo.m_input_smoothing != 0)
103  {
104  m_downSamplers[ci] = DownSampleMethod.fullsize_smooth_downsampler;
105  m_need_context_rows = true;
106  }
107  else
108  {
109  m_downSamplers[ci] = DownSampleMethod.fullsize_downsampler;
110  }
111  }
112  else if (h_in_group == h_out_group * 2 && v_in_group == v_out_group)
113  {
114  smoothok = false;
115  m_downSamplers[ci] = DownSampleMethod.h2v1_downsampler;
116  }
117  else if (h_in_group == h_out_group * 2 && v_in_group == v_out_group * 2)
118  {
119  if (cinfo.m_input_smoothing != 0)
120  {
121  m_downSamplers[ci] = DownSampleMethod.h2v2_smooth_downsampler;
122  m_need_context_rows = true;
123  }
124  else
125  {
126  m_downSamplers[ci] = DownSampleMethod.h2v2_downsampler;
127  }
128  }
129  else if ((h_in_group % h_out_group) == 0 && (v_in_group % v_out_group) == 0)
130  {
131  smoothok = false;
132  m_downSamplers[ci] = DownSampleMethod.int_downsampler;
133  h_expand[ci] = (byte)(h_in_group / h_out_group);
134  v_expand[ci] = (byte)(v_in_group / v_out_group);
135  }
136  else
137  {
138  cinfo.ErrExit(JMessageCode.JERR_FRACT_SAMPLE_NOTIMPL);
139  }
140  }
141 
142  if (cinfo.m_input_smoothing != 0 && !smoothok)
143  {
144  cinfo.TraceMS(0, JMessageCode.JTRC_SMOOTH_NOTIMPL);
145  }
146  }

148  /// <summary>
149  /// <para>Do downsampling for a whole row group (all components).</para>
150  /// <para>In this version we simply downsample each component independently.</para>
151  /// </summary>
152  public void Downsample(byte[][][] input_buf, int in_row_index, byte[][][] output_buf, int out_row_group_index)
153  {
154  for (var ci = 0; ci < m_cinfo.m_num_components; ci++)
155  {
156  var outIndex = out_row_group_index * rowgroup_height[ci];
157  switch (m_downSamplers[ci])
158  {
159  case DownSampleMethod.fullsize_smooth_downsampler:
160  FullsizeSmoothDownsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
161  break;
162 
163  case DownSampleMethod.fullsize_downsampler:
164  FullsizeDownsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
165  break;
166 
167  case DownSampleMethod.h2v1_downsampler:
168  H2V1Downsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
169  break;
170 
171  case DownSampleMethod.h2v2_smooth_downsampler:
172  H2V2SmoothDownsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
173  break;
174 
175  case DownSampleMethod.h2v2_downsampler:
176  H2V2Downsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
177  break;
178 
179  case DownSampleMethod.int_downsampler:
180  IntDownsample(ci, input_buf[ci], in_row_index, output_buf[ci], outIndex);
181  break;
182  }
183  }

191  /// <summary>
192  /// Downsample pixel values of a single component.
193  /// One row group is processed per call.
194  /// This version handles arbitrary integral sampling ratios, without smoothing.
195  /// Note that this version is not actually used for customary sampling ratios.
196  /// </summary>
197  private void IntDownsample(int componentIndex, byte[][] input_data, int startInputRow, byte[][] output_data, int startOutRow)
198  {
199  /* Expand input data enough to let all the output samples be generated
200  * by the standard loop. Special-casing padded output would be more
201  * efficient.
202  */
203  var compptr = m_cinfo.Component_info[componentIndex];
204  var output_cols = compptr.Width_in_blocks * compptr.DCT_h_scaled_size;
205  int h_expand = this.h_expand[compptr.Component_index];
206  ExpandRightEdge(input_data, startInputRow, m_cinfo.m_max_v_samp_factor, m_cinfo.m_image_width, output_cols * h_expand);
207 
208  int v_expand = this.v_expand[compptr.Component_index];
209  var numpix = h_expand * v_expand;
210  var numpix2 = numpix / 2;
211  var inrow = 0;
212  for (var outrow = 0; inrow < m_cinfo.m_max_v_samp_factor; outrow++)
213  {
214  for (int outcol = 0, outcol_h = 0; outcol < output_cols; outcol++, outcol_h += h_expand)
215  {
216  var outvalue = 0;
217  for (var v = 0; v < v_expand; v++)
218  {
219  for (var h = 0; h < h_expand; h++)
220  {
221  outvalue += input_data[startInputRow + inrow + v][outcol_h + h];
222  }
223  }
224 
225  output_data[startOutRow + outrow][outcol] = (byte)((outvalue + numpix2) / numpix);
226  }
227 
228  inrow += v_expand;
229  }
230  }

248  /// <summary>
249  /// <para>
250  /// Downsample pixel values of a single component.
251  /// This version handles the common case of 2:1 horizontal and 1:1 vertical,
252  /// without smoothing.
253  /// </para>
254  /// <para>
255  /// A note about the "bias" calculations: when rounding fractional values to
256  /// integer, we do not want to always round 0.5 up to the next integer.
257  /// If we did that, we'd introduce a noticeable bias towards larger values.
258  /// Instead, this code is arranged so that 0.5 will be rounded up or down at
259  /// alternate pixel locations (a simple ordered dither pattern).
260  /// </para>
261  /// </summary>
262  private void H2V1Downsample(int componentIndex, byte[][] input_data, int startInputRow, byte[][] output_data, int startOutRow)
263  {
264  /* Expand input data enough to let all the output samples be generated
265  * by the standard loop. Special-casing padded output would be more
266  * efficient.
267  */
268  var compptr = m_cinfo.Component_info[componentIndex];
269  var output_cols = compptr.Width_in_blocks * compptr.DCT_h_scaled_size;
270  ExpandRightEdge(input_data, startInputRow, m_cinfo.m_max_v_samp_factor, m_cinfo.m_image_width, output_cols * 2);
271 
272  for (var outrow = 0; outrow < m_cinfo.m_max_v_samp_factor; outrow++)
273  {
274  /* bias = 0,1,0,1,... for successive samples */
275  var bias = 0;
276  var inputColumn = 0;
277  for (var outcol = 0; outcol < output_cols; outcol++)
278  {
279  output_data[startOutRow + outrow][outcol] = (byte)(
280  (input_data[startInputRow + outrow][inputColumn] +
281  input_data[startInputRow + outrow][inputColumn + 1] + bias) >> 1);
282 
283  bias ^= 1; /* 0=>1, 1=>0 */
284  inputColumn += 2;
285  }
286  }
287  }

326  /// <summary>
327  /// Downsample pixel values of a single component.
328  /// This version handles the standard case of 2:1 horizontal and 2:1 vertical,
329  /// with smoothing. One row of context is required.
330  /// </summary>
331  private void H2V2SmoothDownsample(int componentIndex, byte[][] input_data, int startInputRow, byte[][] output_data, int startOutRow)
332  {
333  /* Expand input data enough to let all the output samples be generated
334  * by the standard loop. Special-casing padded output would be more
335  * efficient.
336  */
337  var compptr = m_cinfo.Component_info[componentIndex];
338  var output_cols = compptr.Width_in_blocks * compptr.DCT_h_scaled_size;
339  ExpandRightEdge(input_data, startInputRow - 1, m_cinfo.m_max_v_samp_factor + 2, m_cinfo.m_image_width, output_cols * 2);
340 
341  /* We don't bother to form the individual "smoothed" input pixel values;
342  * we can directly compute the output which is the average of the four
343  * smoothed values. Each of the four member pixels contributes a fraction
344  * (1-8*SF) to its own smoothed image and a fraction SF to each of the three
345  * other smoothed pixels, therefore a total fraction (1-5*SF)/4 to the final
346  * output. The four corner-adjacent neighbor pixels contribute a fraction
347  * SF to just one smoothed pixel, or SF/4 to the final output; while the
348  * eight edge-adjacent neighbors contribute SF to each of two smoothed
349  * pixels, or SF/2 overall. In order to use integer arithmetic, these
350  * factors are scaled by 2^16 = 65536.
351  * Also recall that SF = smoothing_factor / 1024.
352  */
353 
354  var memberscale = 16384 - (m_cinfo.m_input_smoothing * 80); /* scaled (1-5*SF)/4 */
355  var neighscale = m_cinfo.m_input_smoothing * 16; /* scaled SF/4 */
356 
357  var inrow = 0;
358  for (var outrow = 0; inrow < m_cinfo.m_max_v_samp_factor; outrow++)
359  {
360  var outIndex = 0;
361  var inIndex0 = 0;
362  var inIndex1 = 0;
363  var aboveIndex = 0;
364  var belowIndex = 0;
365 
366  /* Special case for first column: pretend column -1 is same as column 0 */
367  var membersum = input_data[startInputRow + inrow][inIndex0] +
368  input_data[startInputRow + inrow][inIndex0 + 1] +
369  input_data[startInputRow + inrow + 1][inIndex1] +
370  input_data[startInputRow + inrow + 1][inIndex1 + 1];
371 
372  var neighsum = input_data[startInputRow + inrow - 1][aboveIndex] +
373  input_data[startInputRow + inrow - 1][aboveIndex + 1] +
374  input_data[startInputRow + inrow + 2][belowIndex] +
375  input_data[startInputRow + inrow + 2][belowIndex + 1] +
376  input_data[startInputRow + inrow][inIndex0] +
377  input_data[startInputRow + inrow][inIndex0 + 2] +
378  input_data[startInputRow + inrow + 1][inIndex1] +
379  input_data[startInputRow + inrow + 1][inIndex1 + 2];
380 
381  neighsum += neighsum;
382  neighsum += input_data[startInputRow + inrow - 1][aboveIndex] +
383  input_data[startInputRow + inrow - 1][aboveIndex + 2] +
384  input_data[startInputRow + inrow + 2][belowIndex] +
385  input_data[startInputRow + inrow + 2][belowIndex + 2];
386 
387  membersum = (membersum * memberscale) + (neighsum * neighscale);
388  output_data[startOutRow + outrow][outIndex] = (byte)((membersum + 32768) >> 16);
389  outIndex++;
390 
391  inIndex0 += 2;
392  inIndex1 += 2;
393  aboveIndex += 2;
394  belowIndex += 2;
395 
396  for (var colctr = output_cols - 2; colctr > 0; colctr--)
397  {
398  /* sum of pixels directly mapped to this output element */
399  membersum = input_data[startInputRow + inrow][inIndex0] +
400  input_data[startInputRow + inrow][inIndex0 + 1] +
401  input_data[startInputRow + inrow + 1][inIndex1] +
402  input_data[startInputRow + inrow + 1][inIndex1 + 1];
403 
404  /* sum of edge-neighbor pixels */
405  neighsum = input_data[startInputRow + inrow - 1][aboveIndex] +
406  input_data[startInputRow + inrow - 1][aboveIndex + 1] +
407  input_data[startInputRow + inrow + 2][belowIndex] +
408  input_data[startInputRow + inrow + 2][belowIndex + 1] +
409  input_data[startInputRow + inrow][inIndex0 - 1] +
410  input_data[startInputRow + inrow][inIndex0 + 2] +
411  input_data[startInputRow + inrow + 1][inIndex1 - 1] +
412  input_data[startInputRow + inrow + 1][inIndex1 + 2];
413 
414  /* The edge-neighbors count twice as much as corner-neighbors */
415  neighsum += neighsum;
416 
417  /* Add in the corner-neighbors */
418  neighsum += input_data[startInputRow + inrow - 1][aboveIndex - 1] +
419  input_data[startInputRow + inrow - 1][aboveIndex + 2] +
420  input_data[startInputRow + inrow + 2][belowIndex - 1] +
421  input_data[startInputRow + inrow + 2][belowIndex + 2];
422 
423  /* form final output scaled up by 2^16 */
424  membersum = (membersum * memberscale) + (neighsum * neighscale);
425 
426  /* round, descale and output it */
427  output_data[startOutRow + outrow][outIndex] = (byte)((membersum + 32768) >> 16);
428  outIndex++;
429 
430  inIndex0 += 2;
431  inIndex1 += 2;
432  aboveIndex += 2;
433  belowIndex += 2;
434  }
435 
436  /* Special case for last column */
437  membersum = input_data[startInputRow + inrow][inIndex0] +
438  input_data[startInputRow + inrow][inIndex0 + 1] +
439  input_data[startInputRow + inrow + 1][inIndex1] +
440  input_data[startInputRow + inrow + 1][inIndex1 + 1];
441 
442  neighsum = input_data[startInputRow + inrow - 1][aboveIndex] +
443  input_data[startInputRow + inrow - 1][aboveIndex + 1] +
444  input_data[startInputRow + inrow + 2][belowIndex] +
445  input_data[startInputRow + inrow + 2][belowIndex + 1] +
446  input_data[startInputRow + inrow][inIndex0 - 1] +
447  input_data[startInputRow + inrow][inIndex0 + 1] +
448  input_data[startInputRow + inrow + 1][inIndex1 - 1] +
449  input_data[startInputRow + inrow + 1][inIndex1 + 1];
450 
451  neighsum += neighsum;
452  neighsum += input_data[startInputRow + inrow - 1][aboveIndex - 1] +
453  input_data[startInputRow + inrow - 1][aboveIndex + 1] +
454  input_data[startInputRow + inrow + 2][belowIndex - 1] +
455  input_data[startInputRow + inrow + 2][belowIndex + 1];
456 
457  membersum = (membersum * memberscale) + (neighsum * neighscale);
458  output_data[startOutRow + outrow][outIndex] = (byte)((membersum + 32768) >> 16);
459 
460  inrow += 2;
461  }

486  var neighscale = m_cinfo.m_input_smoothing * 64; /* scaled SF */

491  var inIndex = 0;
492  var aboveIndex = 0;
493  var belowIndex = 0;
494

497  input_data[startInputRow + outrow + 1][belowIndex] +
498  input_data[startInputRow + outrow][inIndex];
499

501  belowIndex++;
502

505

517  colsum = nextcolsum;
518

522
