1 /*
2  * This file contains the decompression postprocessing controller.
3  * This controller manages the upsampling, color conversion, and color
4  * quantization/reduction steps; specifically, it controls the buffering
5  * between upsample/color conversion and color quantization/reduction.
6  *
7  * If no color quantization/reduction is required, then this module has no
8  * work to do, and it just hands off to the upsample/color conversion code.
9  * An integrated upsample/convert/quantize process would replace this module
10  * entirely.
11  */

16  /// Decompression postprocessing (color quantization buffer control)
17  /// </summary>
18  internal class JpegDPostController

20  private enum ProcessorType
21  {
22  OnePass,

24  Upsample,

32  /* Color quantization source buffer: this holds output data from
33  * the upsample/color conversion step to be passed to the quantizer.
34  * For two-pass color quantization, we need a full-image buffer;
35  * for one-pass operation, a strip buffer is sufficient.
36  */
37  private readonly JVirtArray<byte> m_whole_image; /* virtual array, or null if one-pass */
38  private byte[][] m_buffer; /* strip buffer, or current strip of virtual */
39  private readonly int m_strip_height; /* buffer size in rows */
40  /* for two-pass mode only: */
41  private int m_starting_row; /* row # of first row in current strip */
42  private int m_next_row; /* index of next row to fill/empty in strip */
43 
44  /// <summary>
45  /// Initialize postprocessing controller.
46  /// </summary>
47  public JpegDPostController(JpegDecompressStruct cinfo, bool need_full_buffer)
48  {
49  m_cinfo = cinfo;
50 
51  /* Create the quantization buffer, if needed */
52  if (cinfo.quantizeColors)
53  {
54  /* The buffer strip height is max_v_samp_factor, which is typically
55  * an efficient number of rows for upsampling to return.
56  * (In the presence of output rescaling, we might want to be smarter?)
57  */
58  m_strip_height = cinfo.m_maxVSampleFactor;
59 
60  if (need_full_buffer)
61  {
62  /* Two-pass color quantization: need full-image storage. */
63  /* We round up the number of rows to a multiple of the strip height. */
64  m_whole_image = JpegCommonStruct.CreateSamplesArray(
65  cinfo.outputWidth * cinfo.outColorComponents,
66  JpegUtils.jround_up(cinfo.outputHeight, m_strip_height));
67  m_whole_image.ErrorProcessor = cinfo;
68  }
69  else
70  {
71  /* One-pass color quantization: just make a strip buffer. */
72  m_buffer = JpegCommonStruct.AllocJpegSamples(
73  cinfo.outputWidth * cinfo.outColorComponents, m_strip_height);
74  }
75  }
76  }

78  /// <summary>
79  /// Initialize for a processing pass.
80  /// </summary>
81  public void StartPass(JBufMode pass_mode)
82  {
83  switch (pass_mode)
84  {
85  case JBufMode.PassThrough:
86  if (m_cinfo.quantizeColors)
87  {
88  /* Single-pass processing with color quantization. */
89  m_processor = ProcessorType.OnePass;
90  /* We could be doing buffered-image output before starting a 2-pass
91  * color quantization; in that case, jinit_d_post_controller did not
92  * allocate a strip buffer. Use the virtual-array buffer as workspace.
93  */
94  if (m_buffer is null)
95  {
96  m_buffer = m_whole_image.Access(0, m_strip_height);
97  }
98  }
99  else
100  {
101  /* For single-pass processing without color quantization,
102  * I have no work to do; just call the upsampler directly.
103  */
104  m_processor = ProcessorType.Upsample;
105  }
106  break;
107  case JBufMode.SaveAndPass:
108  /* First pass of 2-pass quantization */
109  if (m_whole_image is null)
110  {
111  m_cinfo.ErrExit(JMessageCode.JERR_BAD_BUFFER_MODE);
112  }
113 
114  m_processor = ProcessorType.PrePass;
115  break;
116  case JBufMode.CrankDest:
117  /* Second pass of 2-pass quantization */
118  if (m_whole_image is null)
119  {
120  m_cinfo.ErrExit(JMessageCode.JERR_BAD_BUFFER_MODE);
121  }
122 
123  m_processor = ProcessorType.SecondPass;
124  break;
125  default:
126  m_cinfo.ErrExit(JMessageCode.JERR_BAD_BUFFER_MODE);
127  break;
128  }
129 
130  m_starting_row = m_next_row = 0;
131  }

133  public void PostProcessData(ComponentBuffer[] input_buf, ref int in_row_group_ctr, int in_row_groups_avail, byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
134  {
135  switch (m_processor)
136  {
137  case ProcessorType.OnePass:
138  PostProcess1Pass(input_buf, ref in_row_group_ctr, in_row_groups_avail, output_buf, ref out_row_ctr, out_rows_avail);
139  break;
140  case ProcessorType.PrePass:
141  PostProcessPrepass(input_buf, ref in_row_group_ctr, in_row_groups_avail, ref out_row_ctr);
142  break;
143  case ProcessorType.Upsample:
144  m_cinfo.m_upsample.UpSample(input_buf, ref in_row_group_ctr, in_row_groups_avail, output_buf, ref out_row_ctr, out_rows_avail);
145  break;
146  case ProcessorType.SecondPass:
147  PostProcess2Pass(output_buf, ref out_row_ctr, out_rows_avail);
148  break;
149  default:
150  m_cinfo.ErrExit(JMessageCode.JERR_NOTIMPL);
151  break;
152  }
153  }

155  /// <summary>
156  /// Process some data in the one-pass (strip buffer) case.
157  /// This is used for color precision reduction as well as one-pass quantization.
158  /// </summary>
159  private void PostProcess1Pass(ComponentBuffer[] input_buf, ref int in_row_group_ctr, int in_row_groups_avail, byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
160  {
161  /* Fill the buffer, but not more than what we can dump out in one go. */
162  /* Note we rely on the upsampler to detect bottom of image. */
163  var max_rows = out_rows_avail - out_row_ctr;
164  if (max_rows > m_strip_height)
165  {
166  max_rows = m_strip_height;
167  }
168 
169  var num_rows = 0;
170  m_cinfo.m_upsample.UpSample(input_buf, ref in_row_group_ctr, in_row_groups_avail, m_buffer, ref num_rows, max_rows);
171 
172  /* Quantize and emit data. */
173  m_cinfo.m_cquantize.ColorQuantize(m_buffer, 0, output_buf, out_row_ctr, num_rows);
174  out_row_ctr += num_rows;
175  }

177  /// <summary>
178  /// Process some data in the first pass of 2-pass quantization.
179  /// </summary>
180  private void PostProcessPrepass(ComponentBuffer[] input_buf, ref int in_row_group_ctr, int in_row_groups_avail, ref int out_row_ctr)
181  {
182  int old_next_row, num_rows;
183 
184  /* Reposition virtual buffer if at start of strip. */
185  if (m_next_row == 0)
186  {
187  m_buffer = m_whole_image.Access(m_starting_row, m_strip_height);
188  }
189 
190  /* Upsample some data (up to a strip height's worth). */
191  old_next_row = m_next_row;
192  m_cinfo.m_upsample.UpSample(input_buf, ref in_row_group_ctr, in_row_groups_avail, m_buffer, ref m_next_row, m_strip_height);
193 
194  /* Allow quantizer to scan new data. No data is emitted, */
195  /* but we advance out_row_ctr so outer loop can tell when we're done. */
196  if (m_next_row > old_next_row)
197  {
198  num_rows = m_next_row - old_next_row;
199  m_cinfo.m_cquantize.ColorQuantize(m_buffer, old_next_row, null, 0, num_rows);
200  out_row_ctr += num_rows;
201  }
202 
203  /* Advance if we filled the strip. */
204  if (m_next_row >= m_strip_height)
205  {
206  m_starting_row += m_strip_height;
207  m_next_row = 0;
208  }
209  }

211  /// <summary>
212  /// Process some data in the second pass of 2-pass quantization.
213  /// </summary>
214  private void PostProcess2Pass(byte[][] output_buf, ref int out_row_ctr, int out_rows_avail)
215  {
216  int num_rows, max_rows;
217 
218  /* Reposition virtual buffer if at start of strip. */
219  if (m_next_row == 0)
220  {
221  m_buffer = m_whole_image.Access(m_starting_row, m_strip_height);
222  }
223 
224  /* Determine number of rows to emit. */
225  num_rows = m_strip_height - m_next_row; /* available in strip */
226  max_rows = out_rows_avail - out_row_ctr; /* available in output area */
227  if (num_rows > max_rows)
228  {
229  num_rows = max_rows;
230  }
231 
232  /* We have to check bottom of image here, can't depend on upsampler. */
233  max_rows = m_cinfo.outputHeight - m_starting_row;
234  if (num_rows > max_rows)
235  {
236  num_rows = max_rows;
237  }
238 
239  /* Quantize and emit data. */
240  m_cinfo.m_cquantize.ColorQuantize(m_buffer, m_next_row, output_buf, out_row_ctr, num_rows);
241  out_row_ctr += num_rows;
242 
243  /* Advance if we filled the strip. */
244  m_next_row += num_rows;
245  if (m_next_row >= m_strip_height)
246  {
247  m_starting_row += m_strip_height;
248  m_next_row = 0;
249  }
250  }
251  }
