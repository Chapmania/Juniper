169  // Creation of 2-D block arrays.
170  private static JBlock[][] AllocJpegBlocks(int blocksPerRow, int numberOfRows)
171  {
172  var result = new JBlock[numberOfRows][];
173  for (var i = 0; i < numberOfRows; ++i)
174  {
175  result[i] = new JBlock[blocksPerRow];
176  for (var j = 0; j < blocksPerRow; ++j)
177  {
178  result[i][j] = new JBlock();

180  }

188  /// <summary>
189  /// <para>
190  /// Abort processing of a JPEG compression or decompression operation,
191  /// but don't destroy the object itself.
192  /// </para>
193  /// <para>
194  /// Closing a data source or destination, if necessary, is the
195  /// application's responsibility.
196  /// </para>
197  /// </summary>
198  public void JpegAbort()
199  {
200  /* Reset overall state for possible reuse of object */
201  if (IsDecompressor)
202  {
203  globalState = JpegState.DSTATE_START;
204 
205  /* Try to keep application from accessing now-deleted marker list.
206  * A bit kludgy to do it here, but this is the most central place.
207  */
208  if (this is JpegDecompressStruct s)
209  {
210  s.m_marker_list = null;
211  }
212  }
213  else
214  {
215  globalState = JpegState.CSTATE_START;
216  }
217  }
