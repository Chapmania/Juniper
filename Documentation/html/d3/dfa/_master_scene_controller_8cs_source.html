576  private IEnumerator LoadScenePathCoroutine(string path, IProgressReceiver prog)
577  {
578  prog?.SetProgress(0);
579 
580  var sceneName = GetSceneNameFromPath(path);
581 
582  var split = prog.Split(2);
583  var sceneLoadProg = split[0];
584  var subSceneLoadProg = split[1];
585 
586  if (IsScenePathLoaded(path))
587  {
588  sceneLoadProg?.SetProgress(1);
589  yield return sceneName + " already loaded.";
590  }
591  else
592  {
593  var op = LoadScene(path, sceneName);
594  while (!op.IsComplete())
595  {
596  sceneLoadProg?.SetProgress(op.Progress);
597  yield return sceneName + " " + (prog?.Progress)?.ToString("P1") ?? "N/A";
598  }
599  }
600 
601  Scene? scene = null;
602  while (scene == null)
603  {
604  scene = GetScene(sceneName, path);
605  yield return sceneName + " " + (prog?.Progress)?.ToString("P1") ?? "N/A";
606  }
607 
608  var toLoad = from root in scene.Value.GetRootGameObjects()
609  from subScene in root.GetComponentsInChildren<SubSceneController>(true)
610  select subScene;
611 
612  subSceneLoadProg.ForEach(toLoad, (ss, p) => ss.Load(p), Debug.LogException);
613 
614  while (prog?.IsComplete() == false)
615  {
616  yield return sceneName + " " + (prog?.Progress)?.ToString("P1") ?? "N/A";
617  }
618  }
