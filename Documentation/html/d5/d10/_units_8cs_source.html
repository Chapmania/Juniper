375  /// <summary>
376  /// Loops a circular value around an range. Unity has this function already, but we
377  /// reimplement it here to make sure the Units subsystem does not depend on Unity.
378  /// </summary>
379  /// <param name="value"></param>
380  /// <param name="range"></param>
381  /// <returns></returns>
382  internal static float Repeat(float value, float range)
383  {
384  while (value < 0)
385  {
386  value += range;
387  }
388  while (value > range)
389  {
390  value -= range;
391  }
392  return value;
393  }

842  /// A look-up to quickly find the conversion function for each unit of measure pairing
843  /// without having to use reflection every time.
844  /// </summary>
845  private static readonly Dictionary<UnitOfMeasure, Dictionary<UnitOfMeasure, MethodInfo>> Conversions;

847  /// <summary>
848  /// Iterate over the <see cref="UnitOfMeasure"/> values and find all of the conversions that
849  /// are defined in the <see cref="Units"/> namespace.
850  /// </summary>
851  static Units()
852  {
853  var u = typeof(Units);
854  var um = typeof(UnitOfMeasure);
855  var f = typeof(float);
856  Conversions =
857  (from fromType in u.GetNestedTypes(BindingFlags.Public)
858  where Enum.IsDefined(um, fromType.Name)
859  group (from toMethod in fromType.GetMethods()
860  where toMethod.ReturnType == f
861  && Enum.IsDefined(um, toMethod.Name)
862  let p = toMethod.GetParameters()
863  where p.Length == 1
864  && p[0].ParameterType == f
865  select toMethod)
866  .ToDictionary(
867  to => (UnitOfMeasure)Enum.Parse(um, to.Name))
868  by fromType.Name)
869  .ToDictionary(
870  from => (UnitOfMeasure)Enum.Parse(um, from.Key),
871  from => from.First());
872  }

874  /// <summary>
875  /// For a given <see cref="UnitOfMeasure"/>, find any <see cref="Category"/> in any <see
876  /// cref="SystemOfMeasure"/> which has a matching Category in the given SystemOfMeasure, then
877  /// retrieve the UnitOfMeasure that fulfills that category in that SystemOfMeasure.
878  /// </summary>
879  /// <param name="fromUnit"></param>
880  /// <param name="toSystem"></param>
881  /// <returns></returns>
882  private static UnitOfMeasure FindConversion(UnitOfMeasure fromUnit, SystemOfMeasure toSystem)
883  {
884  if (!Conversions.ContainsKey(fromUnit))
885  {
886  throw new ArgumentException($"Unit type {fromUnit} not recognized for conversion");
887  }
888  else if (!TypeGroups.ContainsKey(fromUnit))
889  {
890  throw new ArgumentException($"Unit type {fromUnit} has not had type groupings defined");
891  }
892  else if (!SystemUnits.ContainsKey(toSystem))
893  {
894  throw new ArgumentException($"System type {toSystem} not specified");
895  }
896  else
897  {
898  foreach (var g in TypeGroups[fromUnit])
899  {
900  foreach (var s in SystemUnits[toSystem])
901  {
902  if (g == s.Key)
903  {
904  return s.Value;
905  }
906  }
907  }
908 
909  throw new ArgumentException($"Unit type {fromUnit} does not have a matching conversion in system {toSystem}");
910  }
911  }

915  /// </summary>
916  /// <param name="value"></param>
917  /// <param name="fromUnit"></param>
918  /// <param name="toUnit"></param>
919  /// <returns></returns>
920  public static float Convert(this float value, UnitOfMeasure fromUnit, UnitOfMeasure toUnit)
921  {
922  if (fromUnit == toUnit)
923  {
924  return value;
925  }
926  else if (Conversions.TryGetValue(fromUnit, out var forFromUnit)
927  && Conversions[fromUnit].TryGetValue(toUnit, out var conversion))
928  {

930  }
931  else
932  {
933  throw new InvalidCastException($"Cannot convert {value} in {fromUnit} to {toUnit}.");
