310  /// <summary>
311  /// The camera field of view along the large screen axis in portrait mode, the narrow axis in
312  /// landscape mode.
313  /// </summary>
314  /// <value>The vertical field of view.</value>
315  public float VerticalFieldOfView
316  {
317  get
318  {
319  if (Screen.width < Screen.height)
320  {
321  var tan = Mathf.Tan(Units.Degrees.Radians(DEFAULT_FOV / 2));
322  var max = Mathf.Max(Screen.width, Screen.height);
323  var min = Mathf.Min(Screen.width, Screen.height);
324 
325  float aspect;
326  if (Screen.height < Screen.width)
327  {
328  aspect = min / max;
329  }
330  else
331  {
332  aspect = max / min;
333  }
334 
335  return 2 * Units.Radians.Degrees(Mathf.Atan(tan * aspect));
336  }
337  else
338  {
339  return DEFAULT_FOV;
340  }
341  }
342  }
